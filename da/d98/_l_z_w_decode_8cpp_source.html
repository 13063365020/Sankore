91 int LZWDecode::getCode()
92 {
93  int c = 0;
94  int code = 0;
95 
96  while (_readBits < _bitsToRead)
97  {
98  if( _curSymbolIndex < _encoded.size() )
99  {
100  c = _encoded[_curSymbolIndex++];
101  }
102  else
103  {
104  return EOF;
105  }
106  _readBuf = (_readBuf << 8) | (c & 0xff);
107  _readBits += 8;
108  }
109  code = (_readBuf >> (_readBits - _bitsToRead)) & ((1 << _bitsToRead) - 1);
110  _readBits -= _bitsToRead;
111  return code;
112 }

115 bool LZWDecode::decode(std::string & encoded)
116 {
117  _curSymbolIndex = 0;
118  _encoded = encoded;
119 
120  // LZW decoding
121  std::string decoded;
122  struct DecodingTable
123  {
124  int length;
125  int head;
126  unsigned tail;
127  } decTable[4097];
128 
129  int prevCode = 0;
130  int newChar = 0;
131  unsigned curSequence[4097];
132  int nextLength = 0;
133 
134  clearTable();
135  while(1)
136  {
137  int code = getCode();
138  if( code == EOF || code == 257 )
139  {
140  // finish
141  break;
142  }
143  if( code == 256 )
144  {
145  clearTable();
146  continue;
147  }
148  if( _nextCode >= 4997 )
149  {
150  std::cout<<"Bad LZW stream - unexpected clearTable\n";
151  clearTable();
152  continue;
153  }
154  nextLength = _curSequenceLength + 1;
155  if( code < 256 )
156  {
157  curSequence[ 0 ] = code;
158  _curSequenceLength = 1;
159  }
160  else if( code < _nextCode )
161  {
162  //lets take sequence from table
163  _curSequenceLength = decTable[code].length;
164  int j = code;
165  for( int i = _curSequenceLength - 1; i > 0; i--)
166  {
167  curSequence[ i ] = decTable[j].tail;
168  j = decTable[ j ].head;
169  }
170  curSequence[0] = j;
171  }
172  else if( code == _nextCode )
173  {
174  curSequence[ _curSequenceLength ] = newChar;
175  ++_curSequenceLength;
176  }
177  else
178  {
179  std::cout<<"Bad LZW stream - unexpected code "<<code<<"\n";
180  break;
181  }
182  newChar = curSequence[0];
183  if( _first )
184  {
185  _first = false;
186  }
187  else
188  {
189  // lets build decoding table
190  decTable[ _nextCode ].length = nextLength;
191  decTable[ _nextCode ].head = prevCode;
192  decTable[ _nextCode ].tail = newChar;
193  ++ _nextCode;
194  // processing of PDF LZW parameter
195  if (_nextCode + _earlyChange == 512)
196  {
197  _bitsToRead = 10;
198  }
199  else if (_nextCode + _earlyChange == 1024)
200  {
201  _bitsToRead = 11;
202  }
203  else if (_nextCode + _earlyChange == 2048)
204  {
205  _bitsToRead = 12;
206  }
207  }
208  prevCode = code;
209  // put current sequence to output stream
210  for(int i = 0;i < _curSequenceLength;i++)
211  {
212  decoded += (char)curSequence[ i ];
213  }
214  }
215  encoded = decoded;
216 
217  // if predictor exists for that object, then lets decode it
218  if( _predict )
219  {
220  _predict->decode(encoded);
221  }
222  return true;
223 }
