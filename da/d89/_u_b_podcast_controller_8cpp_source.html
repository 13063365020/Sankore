223  int offsetY = (videoFrameSize.height() - scaledWidgetSize.height()) / 2;

258 void UBPodcastController::start()
259 {
260  if (mRecordingState == Stopped)
261  {
262  mInitialized = false;
263 
264  QSize recommendedSize(1024, 768);
265 
266  int fullBitRate = UBSettings::settings()->podcastWindowsMediaBitsPerSecond->get().toInt();
267 
268  if (mSmallVideoSizeAction && mSmallVideoSizeAction->isChecked())
269  {
270  recommendedSize = QSize(640, 480);
271  mVideoBitsPerSecondAtStart = fullBitRate / 4;
272  }
273  else if (mMediumVideoSizeAction && mMediumVideoSizeAction->isChecked())
274  {
275  recommendedSize = QSize(1024, 768);
276  mVideoBitsPerSecondAtStart = fullBitRate / 2;
277  }
278  else if (mFullVideoSizeAction && mFullVideoSizeAction->isChecked())
279  {
280  recommendedSize = UBApplication::boardController->controlView()->size();
281  mVideoBitsPerSecondAtStart = fullBitRate;
282  }
283 
284  QSize scaledboardSize = UBApplication::boardController->controlView()->size();
285  scaledboardSize.scale(recommendedSize, Qt::KeepAspectRatio);
286 
287  // Video width/height should be a multiple of 4
288 
289  int width = scaledboardSize.width();
290  int height = scaledboardSize.height();
291 
292  if (width % 4 != 0)
293  width = ((width / 4) * 4);
294 
295  if (height % 4 != 0)
296  height = ((height / 4) * 4);
297 
298  mVideoFrameSizeAtStart = QSize(width, height);
299 
300  applicationMainModeChanged(UBApplication::applicationController->displayMode());
301 
302 #ifdef Q_WS_WIN
303  mVideoEncoder = new UBWindowsMediaVideoEncoder(this); //deleted on stop
304 #elif defined(Q_WS_MAC)
305  mVideoEncoder = new UBQuickTimeVideoEncoder(this); //deleted on stop
306 #endif
307 
308  if (mVideoEncoder)
309  {
310  connect(mVideoEncoder, SIGNAL(encodingStatus(const QString&)), this, SLOT(encodingStatus(const QString&)));
311  connect(mVideoEncoder, SIGNAL(encodingFinished(bool)), this, SLOT(encodingFinished(bool)));
312 
313  if(mRecordingPalette)
314  {
315  connect(mVideoEncoder, SIGNAL(audioLevelChanged(quint8))
316  , mRecordingPalette, SLOT(audioLevelChanged(quint8)));
317  }
318 
319  mVideoEncoder->setRecordAudio(!mNoAudioInputDeviceAction->isChecked());
320 
321  QString recordingDevice = "";
322 
323  if (!mNoAudioInputDeviceAction->isChecked() && !mDefaultAudioInputDeviceAction->isChecked())
324  {
325  foreach(QAction* audioDevice, mAudioInputDevicesActions)
326  {
327  if (audioDevice->isChecked())
328  {
329  recordingDevice = audioDevice->text();
330  break;
331  }
332  }
333  }
334 
335  mVideoEncoder->setAudioRecordingDevice(recordingDevice);
336 
337  mVideoEncoder->setFramesPerSecond(mVideoFramesPerSecondAtStart);
338  mVideoEncoder->setVideoSize(mVideoFrameSizeAtStart);
339  mVideoEncoder->setVideoBitsPerSecond(mVideoBitsPerSecondAtStart);
340 
341  mPartNumber = 0;
342 
343  mPodcastRecordingPath = UBSettings::settings()->userPodcastRecordingDirectory();
344 
345  qDebug() << "mPodcastRecordingPath: " << mPodcastRecordingPath;
346 
347  QString videoFileName;
348 
349  if (mIntranetPublicationAction && mIntranetPublicationAction->isChecked())
350  {
351  videoFileName = mPodcastRecordingPath + "/" + "Podcast-"
352  + QDateTime::currentDateTime().toString("yyyyMMddhhmmss")
353  + "-" + UBPlatformUtils::computerName() + "." + mVideoEncoder->videoFileExtension();
354  }
355  else
356  {
357  videoFileName = mPodcastRecordingPath + "/" + tr("Sankore Cast") + "." + mVideoEncoder->videoFileExtension();
358  }
359 
360  videoFileName = UBFileSystemUtils::nextAvailableFileName(videoFileName, " ");
361 
362  mVideoEncoder->setVideoFileName(videoFileName);
363 
364  mLatestCapture = QImage(mVideoFrameSizeAtStart, QImage::Format_RGB32); //0xffRRGGBB
365 
366  mRecordStartTime = QTime::currentTime();
367 
368  mRecordingProgressTimerEventID = startTimer(100);
369 
370  if(mVideoEncoder->start())
371  {
372  applicationMainModeChanged(UBApplication::applicationController->displayMode());
373 
374  setRecordingState(Recording);
375 
376  if (mSourceScene)
377  {
378  processScenePaintEvent();
379  }
380  else if (mSourceWidget)
381  {
382  processWidgetPaintEvent();
383  }
384  }
385  else
386  {
387  UBApplication::showMessage(tr("Failed to start encoder ..."), false);
388  }
389  }
390  else
391  {
392  UBApplication::showMessage(tr("No Podcast encoder available ..."), false);
393  }
394  }

466 void UBPodcastController::processWidgetPaintEvent()
467 {
468  if(mRecordingState != Recording)
469  return;
470 
471  QRect repaintRect;
472 
473  if (!mInitialized)
474  {
475  mWidgetRepaintRectQueue.clear();
476  repaintRect = mSourceWidget->geometry();
477 
478  mLatestCapture.fill(sBackgroundColor);
479 
480  mInitialized = true;
481  }
482  else
483  {
484  while(mWidgetRepaintRectQueue.size() > 0)
485  {
486  repaintRect = repaintRect.unite(mWidgetRepaintRectQueue.dequeue());
487  }
488  }
489 
490  if (!repaintRect.isNull())
491  {
492  mIsGrabbing = true;
493 
494  QPainter p(&mLatestCapture);
495  p.setTransform(mViewToVideoTransform);
496  p.setRenderHints(QPainter::Antialiasing);
497  p.setRenderHints(QPainter::SmoothPixmapTransform);
498 
499  mSourceWidget->render(&p, repaintRect.topLeft(), QRegion(repaintRect), QWidget::DrawChildren);
500 
501  mIsGrabbing = false;
502 
503  sendLatestPixmapToEncoder();
504  }
505 }

592 void UBPodcastController::processScenePaintEvent()
593 {
594  if(mRecordingState != Recording)
595  return;
596 
597  UBBoardView *bv = qobject_cast<UBBoardView *>(mSourceWidget);
598 
599  if(!bv)
600  return;
601 
602  QRectF repaintRect;
603 
604  if (!mInitialized)
605  {
606  mSceneRepaintRectQueue.clear();
607  repaintRect = bv->mapToScene(QRect(0, 0, bv->width(), bv->height())).boundingRect();
608 
609  if (bv->scene()->isDarkBackground())
610  mLatestCapture.fill(Qt::black);
611  else
612  mLatestCapture.fill(Qt::white);
613 
614  mLatestCapture.fill(Qt::white);
615 
616  mInitialized = true;
617  }
618  else
619  {
620  while(mSceneRepaintRectQueue.size() > 0)
621  {
622  repaintRect = repaintRect.unite(mSceneRepaintRectQueue.dequeue());
623  }
624  }
625 
626  if (!repaintRect.isNull())
627  {
628  UBGraphicsScene *scene = bv->scene();
629 
630  QPainter p(&mLatestCapture);
631 
632  p.setTransform(mViewToVideoTransform);
633  p.setTransform(bv->viewportTransform(), true);
634 
635  p.setRenderHints(QPainter::Antialiasing);
636  p.setRenderHints(QPainter::SmoothPixmapTransform);
637 
638  repaintRect.adjust(-1, -1, 1, 1);
639 
640  p.setClipRect(repaintRect);
641 
642  if (scene->isDarkBackground())
643  p.fillRect(repaintRect, Qt::black);
644  else
645  p.fillRect(repaintRect, Qt::white);
646 
647  scene->setRenderingContext(UBGraphicsScene::Podcast);
648 
649  scene->render(&p, repaintRect, repaintRect);
650 
651  scene->setRenderingContext(UBGraphicsScene::Screen);
652 
653  sendLatestPixmapToEncoder();
654  }
