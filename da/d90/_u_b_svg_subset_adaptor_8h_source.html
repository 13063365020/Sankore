202  inline QString pointsToSvgPointsAttribute(QVector<QPointF> points)
203  {
204  UBGeometryUtils::crashPointList(points);
205 
206  int pointsCount = points.size();
207  QString svgPoints;
208 
209  int length = 0;
210  QString sBuf;
211 
212  for(int j = 0; j < pointsCount; j++)
213  {
214  sBuf = "%1,%2 ";
215  const QPointF & point = points.at(j);
216 
217  QString temp1 = "%1", temp2 = "%2";
218 
219  temp1 = temp1.arg(point.x());
220  temp2 = temp2.arg(point.y());
221 
222  QLocale loc(QLocale::C);
223  sBuf = sBuf.arg(loc.toFloat(temp1)).arg(loc.toFloat(temp2));
224 
225  svgPoints.insert(length, sBuf);
226  length += sBuf.length();
227  }
228  return svgPoints;
229  }
