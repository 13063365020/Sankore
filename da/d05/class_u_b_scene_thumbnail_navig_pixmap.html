if(bCanDelete && p.x() >= 0 && p.x() <= BUTTONSIZE && p.y() >= 0 && p.y() <= BUTTONSIZE)
deletePage();

duplicatePage();

moveUpPage();

moveDownPage();
event->accept();
}
