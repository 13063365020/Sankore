73 QFont UBGraphicsTextItemDelegate::createDefaultFont()
74 {

77  QString fFamily = UBSettings::settings()->fontFamily();
78  if (!fFamily.isEmpty())
79  textFormat.setFontFamily(fFamily);
80

84

90  int pointSize = UBSettings::settings()->fontPointSize();

128 {
129  fontDialog.setOption(QFontDialog::DontUseNativeDialog);

132  fontDialog.setStyleSheet("background-color: white;");
133  }
134 
135  QListView *fontNameListView;
136  QList<QListView*> listViews = fontDialog.findChildren<QListView*>();
137  if (listViews.count() > 0)
138  {
139  fontNameListView = listViews.at(0);
140  foreach (QListView* listView, listViews)
141  {
142  if (listView->pos().x() < fontNameListView->pos().x())
143  fontNameListView = listView;
144  }
145  }
146  if (fontNameListView)
147  {
148  QStringListModel *stringListModel = dynamic_cast<QStringListModel*>(fontNameListView->model());
149  if (stringListModel)
150  {
151  QStringList dialogFontNames = stringListModel->stringList();
152  QStringList safeWebFontNames;
153  safeWebFontNames.append("Arial");
154  safeWebFontNames.append("Arial Black");
155  safeWebFontNames.append("Comic Sans MS");
156  safeWebFontNames.append("Courier New");
157  safeWebFontNames.append("Georgia");
158  safeWebFontNames.append("Impact");
159  safeWebFontNames.append("Times New Roman");
160  safeWebFontNames.append("Trebuchet MS");
161  safeWebFontNames.append("Verdana");
162 
163  QStringList customFontList = UBResources::resources()->customFontList();
164  int index = 0;
165  foreach (QString dialogFontName, dialogFontNames){
166  if (safeWebFontNames.contains(dialogFontName, Qt::CaseInsensitive) || customFontList.contains(dialogFontName, Qt::CaseSensitive))
167  index++;
168  else
169  stringListModel->removeRow(index);
170  }
171  }
172  }
173  QList<QComboBox*> comboBoxes = fontDialog.findChildren<QComboBox*>();
174  if (comboBoxes.count() > 0)
175  comboBoxes.at(0)->setEnabled(false);
176 }

331 void UBGraphicsTextItemDelegate::ChangeTextSize(qreal factor, textChangeMode changeMode)
332 {
333  if (scaleSize == changeMode)
334  {
335  if (1 == factor)
336  return;
337  }
338  else
339  if (0 == factor)
340  return;
341 
342  UBGraphicsTextItem *item = dynamic_cast<UBGraphicsTextItem*>(delegated());
343 
344  if (item && (QString() == item->toPlainText()))
345  return;
346 
347  QTextCursor cursor = delegated()->textCursor();
348  QTextCharFormat textFormat;
349 
350  int anchorPos = cursor.anchor();
351  int cursorPos = cursor.position();
352 
353  if (0 == anchorPos-cursorPos)
354  {
355  cursor.setPosition (0, QTextCursor::MoveAnchor);
356  cursor.setPosition (cursor.document()->characterCount()-1, QTextCursor::KeepAnchor);
357  }
358 
359  int startPos = qMin(cursor.anchor(), cursor.position());
360  int endPos = qMax(cursor.anchor(), cursor.position());
361 
362  QFont curFont;
363  bool bEndofTheSameBlock;
364  int iBlockLen;
365  int iPointSize;
366  int iNextPointSize;
367  int iCursorPos = startPos;
368 
369  // we search continuous blocks of the text with the same PointSize and allpy new settings for them.
370  cursor.setPosition (startPos, QTextCursor::MoveAnchor);
371  while(iCursorPos < endPos)
372  {
373  bEndofTheSameBlock = false;
374  iBlockLen = 0;
375 
376  cursor.setPosition (iCursorPos+1, QTextCursor::KeepAnchor);
377  iPointSize = cursor.charFormat().font().pointSize();
378  curFont = cursor.charFormat().font();
379  cursor.setPosition (iCursorPos, QTextCursor::KeepAnchor);
380 
381 
382  do
383  {
384  iBlockLen++;
385 
386  cursor.setPosition (iCursorPos+iBlockLen+1, QTextCursor::KeepAnchor);
387  iNextPointSize = cursor.charFormat().font().pointSize();
388 
389  cursor.setPosition (iCursorPos+iBlockLen, QTextCursor::KeepAnchor);
390  if ((iPointSize != iNextPointSize)||(iCursorPos+iBlockLen >= endPos))
391  bEndofTheSameBlock = true;
392 
393  }while(!bEndofTheSameBlock);
394 
395 
396  //setting new parameners
397  int iNewPointSize = (changeSize == changeMode) ? (iPointSize + factor) : (iPointSize * factor);
398  curFont.setPointSize( (iNewPointSize > 0)?iNewPointSize:1);
399  textFormat.setFont(curFont);
400  cursor.mergeCharFormat(textFormat);
401 
402  iCursorPos += iBlockLen;
403  cursor.setPosition (iCursorPos, QTextCursor::MoveAnchor);
404  }
405 
406  UBSettings::settings()->setFontPointSize(iPointSize);
407  //returning initial selection
408  cursor.setPosition (anchorPos, QTextCursor::MoveAnchor);
409  cursor.setPosition (cursorPos, QTextCursor::KeepAnchor);
410 
411  delegated()->setTextCursor(cursor);
412 }
