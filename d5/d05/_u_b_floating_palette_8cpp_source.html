159 void UBFloatingPalette::moveInsideParent(const QPoint &position)
160 {
161  QWidget *parent = parentWidget();
162 
163  if (parent)
164  {
165  int margin = UBSettings::boardMargin - border();
166  qreal newX = qMax(margin, qMin(parent->width() - getParentRightOffset() - width() - margin, position.x()));
167  qreal newY = qMax(margin, qMin(parent->height() - height() - margin, position.y()));
168 
169  if (!mCustomPosition && !mIsMoving)
170  {
171  if (mDefaultPosition == Qt::TopLeftCorner || mDefaultPosition == Qt::BottomLeftCorner)
172  {
173  newX = margin;
174  }
175  else
176  {
177  newX = qMax(margin, parent->width() - getParentRightOffset() - width() - margin);
178  }
179  }
180  move(newX, newY);
181  minimizePalette(QPoint(newX, newY));
182  }
183  else
184  {
185  move(position);
186  }
187 }

211 void UBFloatingPalette::adjustSizeAndPosition(bool pUp, bool resetPosition)
212 {
213  QSize newPreferredSize = preferredSize();
214 
215  foreach (UBFloatingPalette* palette, mAssociatedPalette)
216  {
217  QSize palettePreferredSize = palette->preferredSize();
218  newPreferredSize.setWidth(newPreferredSize.expandedTo(palettePreferredSize).width());
219  }
220  QSize previousSize = size();
221  int biggerHeight = preferredSize().height() - previousSize.height();
222  if ((pUp && (biggerHeight > 0))
223  || (!pUp && (biggerHeight < 0)))
224  {
225  move(pos().x(), pos().y() - biggerHeight);
226  }
227 
228  if (newPreferredSize != size())
229  {
230  resize(newPreferredSize);
231  moveInsideParent(pos());
232  foreach(UBFloatingPalette* palette, mAssociatedPalette)
233  {
234  palette->move(pos().x(), palette->pos().y());
235  palette->resize(newPreferredSize.width(), palette->size().height());
236  }
237  }
238 
239  if (parentWidget() && resetPosition)
240  move((parentWidget()->width() - width()) / 2, (parentWidget()->height() - height()) / 5);
241 }

257 void UBFloatingPalette::paintEvent(QPaintEvent *)
258 {
259  QPainter painter(this);
260  painter.setRenderHint(QPainter::Antialiasing);
261  painter.setPen(Qt::NoPen);
262  painter.setBrush(mBackgroundBrush);
263 
264  if(mbGrip)

267  QPainterPath borderPath;

269  borderPath.addRoundedRect(border(), border(), width() - 2 * border(), height() - 2 * border(), radius(), radius());
270  painter.drawPath(borderPath);
271  painter.setBrush(mBackgroundBrush);
272  painter.drawRoundedRect(border(), border(), width() - 2 * border(), height() - 2 * border(), radius(), radius());
273  }

285 void UBFloatingPalette::minimizePalette(const QPoint& pos)
286 {
287  if(!mCanBeMinimized)
288  {
289  // If this floating palette cannot be minimized, we exit this method.
290  return;
291  }
292 
293  if(mMinimizedLocation == eMinimizedLocation_None)
294  {
295  // Verify if we have to minimize this palette
296  if(pos.x() == 5)
297  {
298  mMinimizedLocation = eMinimizedLocation_Left;
299  }
300 // else if(pos.y() == 5)
301 // {
302 // mMinimizedLocation = eMinimizedLocation_Top;
303 // }
304  else if(pos.x() == parentWidget()->width() - getParentRightOffset() - width() - 5)
305  {
306  mMinimizedLocation = eMinimizedLocation_Right;
307  }
308 // else if(pos.y() == parentSize.height() - height() - 5)
309 // {
310 // mMinimizedLocation = eMinimizedLocation_Bottom;
311 // }
312 
313  // Minimize the Palette
314  if(mMinimizedLocation != eMinimizedLocation_None)
315  {
316  emit minimizeStart(mMinimizedLocation);
317  }
318  }
319  else
320  {
321  // Restore the palette
322  if(pos.x() > 5 &&
323  pos.y() > 5 &&
324  pos.x() < parentWidget()->width() - getParentRightOffset() - width() - 5 &&
325  pos.y() < parentWidget()->size().height() - height() - 5)
326  {
327  mMinimizedLocation = eMinimizedLocation_None;
328  emit maximizeStart();
329  }
330  }
331 }
