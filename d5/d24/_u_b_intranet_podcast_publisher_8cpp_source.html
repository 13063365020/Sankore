126 QString UBIntranetPodcastPublisher::metadata()
127 {

130  /*
131  <podcast>
132  <title>{title}</title>
133  <description>{description}</description>
134  <author>{author}</author>
135  <pubDate>{publicationISODateTime}</pubDate>
136  <duration>{duration}</duration>
137  <fileName>{fileName}</fileName>

139  <fileSize>{fileSize}</fileSize>
140  </podcast>
141  */
142 
143  QString date = UBStringUtils::toUtcIsoDateTime(QDateTime::currentDateTime());
144  QString duration = QString("%1").arg((mVideoDurationInMs / 1000) + 1);
145 
146  QFileInfo fi(mVideoFilePath);
147  QString fileName = fi.fileName();
148  QString computerName = UBPlatformUtils::computerName();
149  QString fileSize = QString("%1").arg(fi.size());
150 
151  QString templatePath = UBPlatformUtils::applicationResourcesDirectory() + "/etc/intranet-podcast-metadata.template";
152  QString templateContent = UBFileSystemUtils::readTextFile(templatePath);
153 
154  return templateContent.replace("{title}", mTitle)
155  .replace("{description}", mDescription)
156  .replace("{author}", mAuthor)
157  .replace("{publicationISODateTime}", date)
158  .replace("{duration}", duration)
159  .replace("{fileName}", fileName)
160  .replace("{computerName}", computerName)
161  .replace("{fileSize}", fileSize)
