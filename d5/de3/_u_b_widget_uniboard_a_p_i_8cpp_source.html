489 void UBWidgetUniboardAPI::ProcessDropEvent(QGraphicsSceneDragDropEvent *event)
490 {
491  const QMimeData *pMimeData = event->mimeData();
492 
493  QString destFileName;
494  QString contentType;
495  bool downloaded = false;
496 
497  QGraphicsView *tmpView = mGraphicsWidget->scene()->views().at(0);
498  QPoint dropPoint(mGraphicsWidget->mapFromScene(tmpView->mapToScene(event->pos().toPoint())).toPoint());
499  Qt::DropActions dropActions = event->possibleActions();
500  Qt::MouseButtons dropMouseButtons = event->buttons();
501  Qt::KeyboardModifiers dropModifiers = event->modifiers();
502  QMimeData *dropMimeData = new QMimeData;
503  qDebug() << event->possibleActions();
504 
505 
506  if (pMimeData->hasHtml()) { //Dropping element from web browser
507  QString qsHtml = pMimeData->html();
508  QString url = UBApplication::urlFromHtml(qsHtml);
509 
510  if(!url.isEmpty()) {
511  QString str = "test string";
512 
513  QMimeData mimeData;
514  mimeData.setData(tMimeText, str.toAscii());
515 
516  sDownloadFileDesc desc;
517  desc.dest = sDownloadFileDesc::graphicsWidget;
518  desc.modal = true;
519  desc.srcUrl = url;
520  desc.currentSize = 0;
521  desc.name = QFileInfo(url).fileName();
522  desc.totalSize = 0; // The total size will be retrieved during the download
523 
524  desc.dropPoint = event->pos().toPoint(); //Passing pure event point. No modifications
525  desc.dropActions = dropActions;
526  desc.dropMouseButtons = dropMouseButtons;
527  desc.dropModifiers = dropModifiers;
528 
529  registerIDWidget(UBDownloadManager::downloadManager()->addFileToDownload(desc));
530 
531  }
532 
533  } else if (pMimeData->hasUrls()) { //Local file processing
534  QUrl curUrl = pMimeData->urls().first();
535  QString sUrl = curUrl.toString();
536 
537  if (sUrl.startsWith("file://") || sUrl.startsWith("/")) {
538  QString fileName = curUrl.toLocalFile();
539  QString extention = UBFileSystemUtils::extension(fileName);
540  contentType = UBFileSystemUtils::mimeTypeFromFileName(fileName);
541 
542  if (supportedTypeHeader(contentType)) {
543  destFileName = getObjDir() + QUuid::createUuid().toString() + "." + extention;
544 
545  if (!UBFileSystemUtils::copyFile(fileName, destFileName)) {
546  qDebug() << "can't copy from" << fileName << "to" << destFileName;
547  return;
548  }
549  downloaded = true;
550 
551  }
552  }
553  }
554  qDebug() << destFileName;
555  QString mimeText = createMimeText(downloaded, contentType, destFileName);
556  dropMimeData->setData(tMimeText, mimeText.toAscii());
557 
558  event->setMimeData(dropMimeData);
559 }

561 void UBWidgetUniboardAPI::onDownloadFinished(bool pSuccess, sDownloadFileDesc desc, QByteArray pData)
562 {
563  //if widget recieves is waiting for this id then process
564  if (!takeIDWidget(desc.id))
565  return;
566 
567  if (!pSuccess) {
568  qDebug() << "can't download the whole data. An error occured";
569  return;
570  }
571 
572  QString contentType = desc.contentTypeHeader;
573  QString extention = UBFileSystemUtils::fileExtensionFromMimeType(contentType);
574 
575  if (!supportedTypeHeader(contentType)) {
576  qDebug() << "actions for mime type" << contentType << "are not supported";
577  return;
578  }
579 
580  QString objDir = getObjDir();
581  if (!QDir().exists(objDir)) {
582  if (!QDir().mkpath(objDir)) {
583  qDebug() << "can't create objects directory path. Check the permissions";
584  return;
585  }
586  }
587 
588  QString destFileName = objDir + QUuid::createUuid() + "." + extention;
589  QFile destFile(destFileName);
590 
591  if (!destFile.open(QIODevice::WriteOnly)) {
592  qDebug() << "can't open" << destFileName << "for wrighting";
593  return;
594  }
595 
596  if (destFile.write(pData) == -1) {
597  qDebug() << "can't implement data writing";
598  return;
599  }
600 
601  QGraphicsView *tmpView = mGraphicsWidget->scene()->views().at(0);
602  QPoint dropPoint(mGraphicsWidget->mapFromScene(tmpView->mapToScene(desc.dropPoint)).toPoint());
603 
604  QMimeData dropMimeData;
605  QString mimeText = createMimeText(true, contentType, destFileName);
606  dropMimeData.setData(tMimeText, mimeText.toAscii());
607 
608  destFile.close();
609 
610  //To make js interpreter accept drop event we need to generate move event first.
611  QDragMoveEvent pseudoMove(dropPoint, desc.dropActions, &dropMimeData, desc.dropMouseButtons, desc.dropModifiers);
612  QApplication::sendEvent(mGraphicsWidget,&pseudoMove);
613 
614  QDropEvent readyEvent(dropPoint, desc.dropActions, &dropMimeData, desc.dropMouseButtons, desc.dropModifiers);
615  //sending event to destination either it had been downloaded or not
616  QApplication::sendEvent(mGraphicsWidget,&readyEvent);
617  readyEvent.acceptProposedAction();
618 }

620 QString UBWidgetUniboardAPI::createMimeText(bool downloaded, const QString &mimeType, const QString &fileName)
621 {
622  QString mimeXml;
623  QXmlStreamWriter writer(&mimeXml);
624  writer.setAutoFormatting(true);
625  writer.writeStartDocument();
626  writer.writeStartElement(tMainSection);
627 
628  writer.writeTextElement(tReady, boolToStr(downloaded));
629 
630  if (downloaded) {
631  if (!mimeType.isEmpty()) {
632  writer.writeTextElement(tType, mimeType); //writing type of element
633  }
634  if (!QFile::exists(fileName)) {
635  qDebug() << "file" << fileName << "doesn't exist";
636  return QString();
637  }
638 
639  QString relatedFileName = fileName;
640  relatedFileName = relatedFileName.remove(mGraphicsWidget->getOwnFolder().toLocalFile() + "/");
641  writer.writeTextElement(tPath, relatedFileName); //writing path to created object
642  }
643 
644  writer.writeEndElement();
645  writer.writeEndDocument();
646 
647  return mimeXml;
648 }
