190 void UBSettings::ValidateKeyboardPaletteKeyBtnSize()
191 {
192  // if boardKeyboardPaletteKeyBtnSize is not initialized, or supportedKeyboardSizes not initialized or empty
193  if( !boardKeyboardPaletteKeyBtnSize ||
194  !supportedKeyboardSizes ||
195  supportedKeyboardSizes->size() == 0 ) return;
196 
197  // get original size value
198  QString origValue = boardKeyboardPaletteKeyBtnSize->get().toString();
199 
200  // parse available size values, for make sure original value is valid
201  for(int i = 0; i < supportedKeyboardSizes->size(); i++)
202  {
203  int compareCode = QString::compare(origValue, supportedKeyboardSizes->at(i));
204  if(compareCode == 0) return;
205  }
206 
207  // if original value is invalid, than set it value to first value from avaliable list
208  boardKeyboardPaletteKeyBtnSize->set(supportedKeyboardSizes->at(0));
209 }

480 qreal UBSettings::currentPenWidth()

482  qreal width = 0;

484  switch (penWidthIndex())
