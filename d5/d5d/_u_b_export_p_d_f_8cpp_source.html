75 void UBExportPDF::persistsDocument(UBDocumentProxy* pDocumentProxy, const QString& filename)
76 {
77  QPrinter pdfPrinter;
78 
79  qDebug() << "exporting document to PDF" << filename;
80 
81  pdfPrinter.setOutputFormat(QPrinter::PdfFormat);
82  pdfPrinter.setResolution(UBSettings::settings()->pdfResolution->get().toInt());
83  pdfPrinter.setOutputFileName(filename);
84  pdfPrinter.setFullPage(true);
85 
86  //need to calculate screen resolution
87  QDesktopWidget* desktop = UBApplication::desktop();
88  int dpiCommon = (desktop->physicalDpiX() + desktop->physicalDpiY()) / 2;
89  float scaleFactor = 72.0f / dpiCommon;
90 
91  QPainter pdfPainter;
92  bool painterNeedsBegin = true;
93 
94  int existingPageCount = pDocumentProxy->pageCount();
95 
96  for(int pageIndex = 0 ; pageIndex < existingPageCount; pageIndex++)
97  {
98  UBGraphicsScene* scene = UBPersistenceManager::persistenceManager()->loadDocumentScene(pDocumentProxy, pageIndex);
99  UBApplication::showMessage(tr("Exporting page %1 of %2").arg(pageIndex + 1).arg(existingPageCount));
100  // set background to white, no crossing for PDF output
101  bool isDark = scene->isDarkBackground();
102  bool isCrossed = scene->isCrossedBackground();
103  scene->setBackground(false, false);
104 
105  QSize pageSize = scene->nominalSize();
106 
107  // set high res rendering
108  scene->setRenderingQuality(UBItem::RenderingQualityHigh);
109  scene->setRenderingContext(UBGraphicsScene::NonScreen);
110 
111  //setting page size to appropriate value
112  pdfPrinter.setPaperSize(QSizeF(pageSize.width()*scaleFactor, pageSize.height()*scaleFactor), QPrinter::Point);
113  if(painterNeedsBegin) painterNeedsBegin = !pdfPainter.begin(&pdfPrinter);
114  //render to PDF
115  scene->render(&pdfPainter, QRectF(), scene->normalizedSceneRect());
116 
117  if (pageIndex < existingPageCount - 1) pdfPrinter.newPage();
118 
119  //restore screen rendering quality
120  scene->setRenderingContext(UBGraphicsScene::Screen);
121  scene->setRenderingQuality(UBItem::RenderingQualityNormal);
122 
123  //restore background state
124  scene->setBackground(isDark, isCrossed);
125  }
126  if(!painterNeedsBegin) pdfPainter.end();
127 }
