100 qreal UBZLayerController::generateZLevel(itemLayerType::Enum key)
101 {
102 
103  if (!scopeMap.contains(key)) {
104  qDebug() << "Number is out of layer scope";
105  return errorNumber;
106  }
107 
108  qreal result = scopeMap.value(key).curValue;
109  qreal top = scopeMap.value(key).topLimit;
110  qreal incrementalStep = scopeMap.value(key).incStep;
111 
112  result += incrementalStep;
113  if (result >= top) {
114  // If not only one variable presents in the scope, notify that values for scope are over
115  if (scopeMap.value(key).topLimit != scopeMap.value(key).bottomLimit) {
116  qDebug() << "new values are over for the scope" << key;
117  }
118  result = top - incrementalStep;
119  }
120 
121  scopeMap[key].curValue = result;
122 
123  return result;

137 qreal UBZLayerController::changeZLevelTo(QGraphicsItem *item, moveDestination dest)
138 {
139  itemLayerType::Enum curItemLayerType = typeForData(item);
140  if (curItemLayerType == itemLayerType::NoLayer) {
141  qDebug() << "item's layer is out of the scope. Can't implement z-layer changing operation";
142  return errorNum();
143  }
144 
145  //select only items wiht the same z-level as item's one and push it to sortedItems QMultiMap
146  QMultiMap<qreal, QGraphicsItem*> sortedItems;
147  if (mScene->items().count()) {
148  foreach (QGraphicsItem *tmpItem, mScene->items()) {
149  if (typeForData(tmpItem) == curItemLayerType) {
150 
151  sortedItems.insert(tmpItem->data(UBGraphicsItemData::ItemOwnZValue).toReal(), tmpItem);
152  }
153  }
154  }
155 
156  //If only one item itself - do nothing, return it's z-value
157  if (sortedItems.count() == 1 && sortedItems.values().first() == item) {
158  qDebug() << "only one item exists in layer. Have nothing to change";
159  return item->data(UBGraphicsItemData::ItemOwnZValue).toReal();
160  }
161 
162  QMapIterator<qreal, QGraphicsItem*>iCurElement(sortedItems);
163 
164  if (dest == up) {
165  if (iCurElement.findNext(item)) {
166  if (iCurElement.hasNext()) {
167  qreal nextZ = iCurElement.peekNext().value()->data(UBGraphicsItemData::ItemOwnZValue).toReal();
168  UBGraphicsItem::assignZValue(iCurElement.peekNext().value(), item->data(UBGraphicsItemData::ItemOwnZValue).toReal());
169  UBGraphicsItem::assignZValue(item, nextZ);
170 
171  iCurElement.next();
172 
173  while (iCurElement.hasNext() && iCurElement.peekNext().value()->data(UBGraphicsItemData::ItemOwnZValue).toReal() == nextZ) {
174  UBGraphicsItem::assignZValue(iCurElement.next().value(), nextZ);
175  }
176  }
177  }
178 
179  } else if (dest == top) {
180  if (iCurElement.findNext(item)) {
181  if (iCurElement.hasNext()) {
182  UBGraphicsItem::assignZValue(item, generateZLevel(item));
183  }
184  }
185 
186  } else if (dest == down) {
187  iCurElement.toBack();
188  if (iCurElement.findPrevious(item)) {
189  if (iCurElement.hasPrevious()) {
190  qreal nextZ = iCurElement.peekPrevious().value()->data(UBGraphicsItemData::ItemOwnZValue).toReal();
191  UBGraphicsItem::assignZValue(iCurElement.peekPrevious().value(), item->data(UBGraphicsItemData::ItemOwnZValue).toReal());
192  UBGraphicsItem::assignZValue(item, nextZ);
193 
194  while (iCurElement.hasNext() && iCurElement.peekNext().value()->data(UBGraphicsItemData::ItemOwnZValue).toReal() == nextZ) {
195  UBGraphicsItem::assignZValue(iCurElement.next().value(), nextZ);
196  }
197  }
198  }
199 
200  } else if (dest == bottom) {
201  iCurElement.toBack();
202  if (iCurElement.findPrevious(item)) {
203  if (iCurElement.hasPrevious()) {
204  qreal oldz = item->data(UBGraphicsItemData::ItemOwnZValue).toReal();
205  iCurElement.toFront();
206  qreal nextZ = iCurElement.next().value()->data(UBGraphicsItemData::ItemOwnZValue).toReal();
207 
208  ItemLayerTypeData curItemLayerTypeData = scopeMap.value(curItemLayerType);
209 
210  //if we have some free space between lowest graphics item and layer's bottom bound,
211  //insert element close to first element in layer
212  if (nextZ > curItemLayerTypeData.bottomLimit + curItemLayerTypeData.incStep) {
213  qreal result = nextZ - curItemLayerTypeData.incStep;
214  UBGraphicsItem::assignZValue(item, result);
215  } else {
216  UBGraphicsItem::assignZValue(item, nextZ);
217 
218  bool doubleGap = false; //to detect if we can finish rundown since we can insert item to the free space
219 
220  while (iCurElement.peekNext().value() != item) {
221  qreal curZ = iCurElement.value()->data(UBGraphicsItemData::ItemOwnZValue).toReal();
222  qreal curNextZ = iCurElement.peekNext().value()->data(UBGraphicsItemData::ItemOwnZValue).toReal();
223  if (curNextZ - curZ >= 2 * curItemLayerTypeData.incStep) {
224  UBGraphicsItem::assignZValue(iCurElement.value(), curZ + curItemLayerTypeData.incStep);
225  doubleGap = true;
226  break;
227  } else {
228  UBGraphicsItem::assignZValue(iCurElement.value(), curNextZ);
229  iCurElement.next();
230  }
231  }
232  if (!doubleGap) {
233 
234  UBGraphicsItem::assignZValue(iCurElement.value(), oldz);
235 
236  while (iCurElement.hasNext() && (iCurElement.peekNext().value()->data(UBGraphicsItemData::ItemOwnZValue).toReal() == oldz)) {
237  UBGraphicsItem::assignZValue(iCurElement.next().value(), oldz);
238  }
239  }
240  }
241  }
242  }
243  }
244 
245 
246  //clear selection of the item and then select it again to activate selectionChangeProcessing()
247  item->scene()->clearSelection();
248  item->setSelected(true);
249 
250  //Return new z value assigned to item
251  return item->data(UBGraphicsItemData::ItemOwnZValue).toReal();
252

458 bool UBGraphicsScene::inputDeviceMove(const QPointF& scenePos, const qreal& pressure)
459 {
460  bool accepted = false;
461 
462  UBDrawingController *dc = UBDrawingController::drawingController();
463  UBStylusTool::Enum currentTool = (UBStylusTool::Enum)dc->stylusTool();
464 
465  QPointF position = QPointF(scenePos);
466 
467  if (currentTool == UBStylusTool::Eraser)
468  {
469  drawEraser(position);
470  accepted = true;
471  }
472 
473  if (mInputDeviceIsPressed)
474  {
475  if (dc->isDrawingTool())
476  {
477  qreal width = 0;
478 
479  if (currentTool != UBStylusTool::Line){
480  // Handle the pressure
481  width = dc->currentToolWidth() * pressure;
482  }else{
483  // Ignore pressure for line tool
484  width = dc->currentToolWidth();
485  }
486 
487  width /= UBApplication::boardController->systemScaleFactor();
488  width /= UBApplication::boardController->currentZoom();
489 
490  if (currentTool == UBStylusTool::Line || dc->mActiveRuler)
491  {
492  if (UBDrawingController::drawingController()->stylusTool() != UBStylusTool::Marker)
493  if(NULL != mpLastPolygon && NULL != mCurrentStroke && mAddedItems.size() > 0){
494  UBCoreGraphicsScene::removeItemFromDeletion(mpLastPolygon);
495  mAddedItems.remove(mpLastPolygon);
496  mCurrentStroke->remove(mpLastPolygon);
497  if (mCurrentStroke->polygons().empty()){
498  delete mCurrentStroke;
499  mCurrentStroke = NULL;
500  }
501  removeItem(mpLastPolygon);
502  mPreviousPolygonItems.removeAll(mpLastPolygon);
503  }
504 
505  // ------------------------------------------------------------------------
506  // Here we wanna make sure that the Line will 'grip' at i*45, i*90 degrees
507  // ------------------------------------------------------------------------
508 
509  QLineF radius(mPreviousPoint, position);
510  qreal angle = radius.angle();
511  angle = qRound(angle / 45) * 45;
512  qreal radiusLength = radius.length();
513  QPointF newPosition(
514  mPreviousPoint.x() + radiusLength * cos((angle * PI) / 180),
515  mPreviousPoint.y() - radiusLength * sin((angle * PI) / 180));
516  QLineF chord(position, newPosition);
517  if (chord.length() < qMin((int)16, (int)(radiusLength / 20)))
518  position = newPosition;
519  }
520 
521  if(dc->mActiveRuler){
522  dc->mActiveRuler->DrawLine(position, width);
523  }else{
524  drawLineTo(position, width, UBDrawingController::drawingController()->stylusTool() == UBStylusTool::Line);
525  }
526  }
527  else if (currentTool == UBStylusTool::Eraser)
528  {
529  qreal eraserWidth = UBSettings::settings()->currentEraserWidth();
530  eraserWidth /= UBApplication::boardController->systemScaleFactor();
531  eraserWidth /= UBApplication::boardController->currentZoom();
532 
533  eraseLineTo(position, eraserWidth);
534  }
535  else if (currentTool == UBStylusTool::Pointer)
536  {
537  drawPointer(position);
538  }
539 
540  accepted = true;
541  }
542 
543  return accepted;
544 }

780 void UBGraphicsScene::eraseLineTo(const QPointF &pEndPoint, const qreal &pWidth)
781 {
782  const QLineF line(mPreviousPoint, pEndPoint);
783  mPreviousPoint = pEndPoint;
784 
785  const QPolygonF eraserPolygon = UBGeometryUtils::lineToPolygon(line, pWidth);
786  const QRectF eraserBoundingRect = eraserPolygon.boundingRect();
787 
788  QPainterPath eraserPath;
789  eraserPath.addPolygon(eraserPolygon);
790 
791  // Get all the items that are intersecting with the eraser path
792  QList<QGraphicsItem*> collidItems = items(eraserBoundingRect, Qt::IntersectsItemBoundingRect);
793 
794  QList<UBGraphicsPolygonItem*> intersectedItems;
795 
796  typedef QList<QPolygonF> POLYGONSLIST;
797  QList<POLYGONSLIST> intersectedPolygons;
798 
799  #pragma omp parallel for
800  for(int i=0; i<collidItems.size(); i++)
801  {
802  UBGraphicsPolygonItem *pi = qgraphicsitem_cast<UBGraphicsPolygonItem*>(collidItems[i]);
803  if(pi == NULL)
804  continue;
805 
806  QPainterPath itemPainterPath;
807  itemPainterPath.addPolygon(pi->sceneTransform().map(pi->polygon()));
808 
809  if (eraserPath.contains(itemPainterPath))
810  {
811  #pragma omp critical
812  {
813  // Compete remove item
814  intersectedItems << pi;
815  intersectedPolygons << QList<QPolygonF>();
816  }
817  }
818  else if (eraserPath.intersects(itemPainterPath))
819  {
820 
821  QPainterPath newPath = itemPainterPath.subtracted(eraserPath);
822  #pragma omp critical
823  {
824  intersectedItems << pi;
825  intersectedPolygons << newPath.simplified().toFillPolygons(pi->sceneTransform().inverted());
826  }
827  }
828  }
829 
830  for(int i=0; i<intersectedItems.size(); i++)
831  {
832  // item who intersects with eraser
833  UBGraphicsPolygonItem *intersectedPolygonItem = intersectedItems[i];
834 
835  if (!intersectedPolygons[i].empty())
836  {
837  // intersected polygons generated as QList<QPolygon> QPainterPath::toFillPolygons(),
838  // so each intersectedPolygonItem has one or couple of QPolygons who should be removed from it.
839  for(int j = 0; j < intersectedPolygons[i].size(); j++)
840  {
841  // create small polygon from couple of polygons to replace particular erased polygon
842  UBGraphicsPolygonItem* polygonItem = new UBGraphicsPolygonItem(intersectedPolygons[i][j], intersectedPolygonItem->parentItem());
843 
844  intersectedPolygonItem->copyItemParameters(polygonItem);
845  polygonItem->setStroke(intersectedPolygonItem->stroke());
846  polygonItem->setStrokesGroup(intersectedPolygonItem->strokesGroup());
847  intersectedPolygonItem->strokesGroup()->addToGroup(polygonItem);
848  mAddedItems << polygonItem;
849  }
850  }
851 
852  //remove full polygon item for replace it by couple of polygons which creates the same stroke without a part intersects with eraser
853  mRemovedItems << intersectedPolygonItem;
854 
855  QTransform t;
856  bool bApplyTransform = false;
857  if (intersectedPolygonItem->strokesGroup())
858  {
859  if (intersectedPolygonItem->strokesGroup()->parentItem())
860  {
861  bApplyTransform = true;
862  t = intersectedPolygonItem->sceneTransform();
863  }
864  intersectedPolygonItem->strokesGroup()->removeFromGroup(intersectedPolygonItem);
865  }
866 
867  removeItem(intersectedPolygonItem);
868  if (bApplyTransform)
869  intersectedPolygonItem->setTransform(t);
870 
871 
872  removeItem(intersectedPolygonItem);
873 
874  if (bApplyTransform)
875  intersectedPolygonItem->setTransform(t);
876  }
877 
878  if (!intersectedItems.empty())
879  setModified(true);
880 }
