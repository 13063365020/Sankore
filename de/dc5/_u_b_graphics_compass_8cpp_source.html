232 void UBGraphicsCompass::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
233 {
234  if (UBDrawingController::drawingController ()->stylusTool() != UBStylusTool::Selector &&
235  UBDrawingController::drawingController ()->stylusTool() != UBStylusTool::Play)
236  return;
237 
238  if (!mResizing && !mRotating && !mDrawing)
239  {
240  QGraphicsRectItem::mouseMoveEvent(event);
241  mDrewCenterCross = false;
242  }
243  else
244  {
245  if (mResizing)
246  {
247  QPointF delta = event->pos() - event->lastPos();
248  if (rect().width() + delta.x() < sMinRadius)
249  delta.setX(sMinRadius - rect().width());
250  setRect(QRectF(rect().topLeft(), QSizeF(rect().width() + delta.x(), rect().height())));
251  }
252  else
253  {
254  QLineF currentLine(needlePosition(), event->pos());
255  QLineF lastLine(needlePosition(), event->lastPos());
256  qreal deltaAngle = currentLine.angleTo(lastLine);
257  if (deltaAngle > 180)
258  deltaAngle -= 360;
259  else if (deltaAngle < -180)
260  deltaAngle += 360;
261  rotateAroundNeedle(deltaAngle);
262 
263  if (mDrawing)
264  {
265  mSpanAngleInDegrees += deltaAngle;
266  if (mSpanAngleInDegrees >= 1080)
267  mSpanAngleInDegrees -= 360;
268  else if (mSpanAngleInDegrees < -1080)
269  mSpanAngleInDegrees += 360;
270  drawArc();
271  }
272  }
273  event->accept();
274  }
275 }

415 void UBGraphicsCompass::paintRadiusDisplay(QPainter *painter)
416 {
417  qreal radiusInCentimeters = rect().width() / (mPixelsPerMillimeter * 10);
418  QString format = rect().width() >= sDisplayRadiusUnitMinLength ? "%1 cm" : "%1";
419  QString radiusText = QString(format).arg(radiusInCentimeters, 0, 'f', 1);
420 
421  bool onPencilArm = rect().width() > sDisplayRadiusOnPencilArmMinLength;
422 
423  painter->save();
424  painter->setFont(font());
425  QFontMetricsF fm(painter->font());
426  QPointF textCenter;
427 
428  if (onPencilArm)
429  textCenter = QPointF(rect().right() - sPencilBaseLength - sPencilLength - fm.width(radiusText) / 2 - 24 - 8, rect().center().y());
430  else
431  textCenter = QPointF((rect().left() + sNeedleLength + sNeedleBaseLength + hingeRect().left()) / 2, rect().center().y());
432 
433  painter->translate(textCenter);
434  qreal angle = angleInDegrees();
435  if (angle > 180)
436  angle -= 360;
437  else if (angle < -180)
438  angle += 360;
439  if (angle <= -90 || angle > 90)
440  painter->rotate(180);
441  painter->drawText(
442  QRectF(
443  - fm.width(radiusText) / 2,
444  - rect().height() / 2,
445  fm.width(radiusText),

449  painter->restore();
