100 void UBGraphicsTextItem::mousePressEvent(QGraphicsSceneMouseEvent *event)
101 {
102  // It is a cludge...
103  if (UBStylusTool::Play == UBDrawingController::drawingController()->stylusTool())
104  {
105  QGraphicsTextItem::mousePressEvent(event);
106  event->accept();
107  clearFocus();
108  return;
109  }
110 
111  if (Delegate())
112  {
113  Delegate()->mousePressEvent(event);
114  if (Delegate() && parentItem() && UBGraphicsGroupContainerItem::Type == parentItem()->type())
115  {
116  UBGraphicsGroupContainerItem *group = qgraphicsitem_cast<UBGraphicsGroupContainerItem*>(parentItem());
117  if (group)
118  {
119  QGraphicsItem *curItem = group->getCurrentItem();
120  if (curItem && this != curItem)
121  {
122  group->deselectCurrentItem();
123  }
124  group->setCurrentItem(this);
125  this->setSelected(true);
126  Delegate()->positionHandles();
127  }
128 
129  }
130  else
131  {
132  Delegate()->getToolBarItem()->show();
133  }
134 
135  }
136 
137  if (!data(UBGraphicsItemData::ItemEditable).toBool())
138  return;
139 
140  int elapsed = mLastMousePressTime.msecsTo(QTime::currentTime());
141 
142  if (elapsed < UBApplication::app()->doubleClickInterval())
143  {
144  mMultiClickState++;
145  if (mMultiClickState > 3)
146  mMultiClickState = 1;
147  }
148  else
149  {
150  mMultiClickState = 1;
151  }
152 
153  mLastMousePressTime = QTime::currentTime();
154 
155  if (mMultiClickState == 1)
156  {
157 // setTextInteractionFlags(Qt::TextEditorInteraction);
158  QGraphicsTextItem::mousePressEvent(event);
159  setFocus();
160  }
161  else if (mMultiClickState == 2)
162  {
163  QTextCursor tc= textCursor();
164  tc.select(QTextCursor::WordUnderCursor);
165  setTextCursor(tc);
166  }
167  else if (mMultiClickState == 3)
168  {
169  QTextCursor tc= textCursor();
170  tc.select(QTextCursor::Document);
171  setTextCursor(tc);
172  }
173  else
174  {
175  mMultiClickState = 0;
176  }
177 }

187 void UBGraphicsTextItem::mouseReleaseEvent(QGraphicsSceneMouseEvent *event)
188 {
189  // scene()->itemAt(pos) returns 0 if pos is not over text, but over text item, but mouse press comes.
190  // It is a cludge...
191  if (UBStylusTool::Play == UBDrawingController::drawingController()->stylusTool())
192  {
193  QPointF distance = event->pos() - event->lastPos();
194  if( fabs(distance.x()) < 1 && fabs(distance.y()) < 1 )
195  Delegate()->mouseReleaseEvent(event);
196  else
197  QGraphicsTextItem::mouseReleaseEvent(event);
198 
199  event->accept();
200  clearFocus();
201  return;
202  }
203 
204  if (mMultiClickState == 1)
205  {
206  if (Delegate())
207  Delegate()->mouseReleaseEvent(event);
208 
209  QGraphicsTextItem::mouseReleaseEvent(event);
210  }
211  else
212  {
213  event->accept();
214  }

218 {
