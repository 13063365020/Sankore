43 void UBThumbnailAdaptor::generateMissingThumbnails(UBDocumentProxy* proxy)
44 {
45  int existingPageCount = proxy->pageCount();
46 
47  for (int iPageNo = 0; iPageNo < existingPageCount; ++iPageNo)
48  {
49  QString thumbFileName = proxy->persistencePath() + UBFileSystemUtils::digitFileFormat("/page%1.thumbnail.jpg", iPageNo);
50 
51  QFile thumbFile(thumbFileName);
52 
53  if (!thumbFile.exists())
54  {
55  bool displayMessage = (existingPageCount > 5);
56 
57  int thumbCount = 0;
58 
59  UBGraphicsScene* scene = UBSvgSubsetAdaptor::loadScene(proxy, iPageNo);
60 
61  if (scene)
62  {
63  thumbCount++;
64 
65  if (displayMessage && thumbCount == 1)
66  UBApplication::showMessage(tr("Generating preview thumbnails ..."));
67 
68  persistScene(proxy, scene, iPageNo);
69  }
70 
71  if (displayMessage && thumbCount > 0)
72  UBApplication::showMessage(tr("%1 thumbnails generated ...").arg(thumbCount));
73 
74  }
75  }
76 }

78 const QPixmap* UBThumbnailAdaptor::get(UBDocumentProxy* proxy, int pageIndex)
79 {
80  QString fileName = proxy->persistencePath() + UBFileSystemUtils::digitFileFormat("/page%1.thumbnail.jpg", pageIndex);
81 
82  QFile file(fileName);
83  if (!file.exists())
84  {
85  generateMissingThumbnails(proxy);
86  }
87 
88  QPixmap* pix = new QPixmap();

90  {

94 #else

96 #endif
97  }
98  return pix;
99 }

131 void UBThumbnailAdaptor::persistScene(UBDocumentProxy* proxy, UBGraphicsScene* pScene, int pageIndex, bool overrideModified)
132 {
133  QString fileName = proxy->persistencePath() + UBFileSystemUtils::digitFileFormat("/page%1.thumbnail.jpg", pageIndex);
134 
135  QFile thumbFile(fileName);
136 
137  if (pScene->isModified() || overrideModified || !thumbFile.exists() || UBApplication::boardController->paletteManager()->teacherGuideDockWidget()->teacherGuideWidget()->isModified())
138  {
139  qreal nominalWidth = pScene->nominalSize().width();
140  qreal nominalHeight = pScene->nominalSize().height();
141  qreal ratio = nominalWidth / nominalHeight;
142  QRectF sceneRect = pScene->normalizedSceneRect(ratio);
143 
144  qreal width = UBSettings::maxThumbnailWidth;
145  qreal height = width / ratio;
146 
147  QImage thumb(width, height, QImage::Format_ARGB32);
148 
149  QRectF imageRect(0, 0, width, height);
150 
151  QPainter painter(&thumb);
152  painter.setRenderHint(QPainter::Antialiasing, true);
153  painter.setRenderHint(QPainter::SmoothPixmapTransform, true);
154 
155  if (pScene->isDarkBackground())
156  {
157  painter.fillRect(imageRect, Qt::black);
158  }
159  else
160  {
161  painter.fillRect(imageRect, Qt::white);
162  }
163 
164  pScene->setRenderingQuality(UBItem::RenderingQualityHigh);
165 
166  pScene->render(&painter, imageRect, sceneRect, Qt::KeepAspectRatio);
167 
168  if(UBApplication::boardController->paletteManager()->teacherGuideDockWidget()->teacherGuideWidget()->isModified()){
169  QPixmap toque(":images/toque.svg");
170  painter.setOpacity(0.6);
171  painter.drawPixmap(QPoint(width - toque.width(),0),toque);
172  }
173 
174  pScene->setRenderingQuality(UBItem::RenderingQualityNormal);
175 
176  thumb.scaled(width, height, Qt::KeepAspectRatio, Qt::SmoothTransformation).save(fileName, "JPG");
177  }
178 }
