585  {
586  qDebug() << j << qslCookieVals.at(j);

588  {

611 {
612  if (QFile::exists(mTmpZipFile))
613  {
614  QFile f(mTmpZipFile);
615  if (f.open(QIODevice::ReadOnly))
616  {
617  QFileInfo fi(f);
618  QByteArray ba = f.readAll();
619  QString boundary,data, multipartHeader;
620  QByteArray datatoSend;
621 
622  boundary = "---WebKitFormBoundaryDKBTgA53MiyWrzLY";
623  multipartHeader = "multipart/form-data; boundary="+boundary;
624 
625  data="--"+boundary+mCrlf;

627  data+="--"+boundary+mCrlf;
628  data+="Content-Disposition: form-data; name=\"description\"" + mCrlf + mCrlf + mDocInfos.description.remove("\n") + mCrlf;
629  data+="--"+boundary+mCrlf;
630  data+="Content-Disposition: form-data; name=\"file\"; filename=\""+ fi.fileName() +"\""+mCrlf;
631  data+="Content-Type: application/octet-stream"+mCrlf+mCrlf;
632  datatoSend=data.toAscii(); // convert data string to byte array for request
633  datatoSend += ba;
634  datatoSend += mCrlf;
635  datatoSend += QString("--%0--%1").arg(boundary).arg(mCrlf);
636 
637  QNetworkRequest request(QUrl(QString(DOCPUBLICATION_URL).toAscii().constData()));
638 
639  request.setHeader(QNetworkRequest::ContentTypeHeader, multipartHeader);
640  request.setHeader(QNetworkRequest::ContentLengthHeader,datatoSend.size());

643  request.setRawHeader("Host", "planete.sankore.org");
644  request.setRawHeader("Accept", "*/*");
645  request.setRawHeader("Accept-Language", "en-US,*");
