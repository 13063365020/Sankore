349 bool UBWebController::hasEmbeddedContent()
350 {
351  bool bHasContent = false;
352  if(mCurrentWebBrowser){
353  QString html = (*mCurrentWebBrowser)->currentTabWebView()->webPage()->mainFrame()->toHtml();
354 
355  // search the presence of "+oembed"
356  QString query = "\\+oembed([^>]*)>";
357  QRegExp exp(query);
358  exp.indexIn(html);
359  QStringList results = exp.capturedTexts();
360  if(2 <= results.size() && "" != results.at(1)){
361  // An embedded content has been found, no need to check the other ones
362  bHasContent = true;
363  }else{
364  QList<QUrl> contentUrls;
365  lookForEmbedContent(&html, "embed", "src", &contentUrls);
366  lookForEmbedContent(&html, "video", "src", &contentUrls);
367  lookForEmbedContent(&html, "object", "data", &contentUrls);
368 
369  // TODO: check the hidden iFrame
370 
371  if(!contentUrls.empty()){
372  bHasContent = true;
373  }
374  }
375  }
376 
377  return bHasContent;

380 QPixmap UBWebController::captureCurrentPage()
381 {
382  QPixmap pix;
383 
384  if (mCurrentWebBrowser
385  && (*mCurrentWebBrowser)
386  && (*mCurrentWebBrowser)->currentTabWebView()
387  && (*mCurrentWebBrowser)->currentTabWebView()->page()
388  && (*mCurrentWebBrowser)->currentTabWebView()->page()->mainFrame())
389  {
390  QWebFrame* frame = (*mCurrentWebBrowser)->currentTabWebView()->page()->mainFrame();
391  QSize size = frame->contentsSize();
392 
393  qDebug() << size;
394 
395  QVariant top = frame->evaluateJavaScript("document.getElementsByTagName('body')[0].clientTop");
396  QVariant left = frame->evaluateJavaScript("document.getElementsByTagName('body')[0].clientLeft");
397  QVariant width = frame->evaluateJavaScript("document.getElementsByTagName('body')[0].clientWidth");
398  QVariant height = frame->evaluateJavaScript("document.getElementsByTagName('body')[0].clientHeight");
399 
400  QSize vieportSize = (*mCurrentWebBrowser)->currentTabWebView()->page()->viewportSize();
401  (*mCurrentWebBrowser)->currentTabWebView()->page()->setViewportSize(frame->contentsSize());
402  pix = QPixmap(frame->geometry().width(), frame->geometry().height());
403 
404  {
405  QPainter p(&pix);
406  frame->render(&p);
407  }
408 
409  if (left.isValid() && top.isValid() && width.isValid() && height.isValid())
410  {
411  bool okLeft, okTop, okWidth, okHeight;
412 
413  int iLeft = left.toInt(&okLeft) * frame->zoomFactor();
414  int iTop = top.toInt(&okTop) * frame->zoomFactor();
415  int iWidth = width.toInt(&okWidth) * frame->zoomFactor();
416  int iHeight = height.toInt(&okHeight) * frame->zoomFactor();
417 
418  if(okLeft && okTop && okWidth && okHeight)
419  {
420  pix = pix.copy(iLeft, iTop, iWidth, iHeight);
421  }
422  }
423 
424 
425  (*mCurrentWebBrowser)->currentTabWebView()->page()->setViewportSize(vieportSize);
426  }
427 
428  return pix;
429 }

612 void UBWebController::lookForEmbedContent(QString* pHtml, QString tag, QString attribute, QList<QUrl> *pList)
613 {
614  if(NULL != pHtml && NULL != pList){
615  QVector<QString> urlsFound;
616  // Check for <embed> content
617  QRegExp exp(QString("<%0(.*)").arg(tag));
618  exp.indexIn(*pHtml);
619  QStringList strl = exp.capturedTexts();
620  if(2 <= strl.size() && strl.at(1) != ""){
621  // Here we call this regular expression:
622  // src\s?=\s?['"]([^'"]*)['"]
623  // It says: give me all characters that are after src=" (or src = ")
624  QRegExp src(QString("%0\\s?=\\s?['\"]([^'\"]*)['\"]").arg(attribute));
625  for(int i=1; i<strl.size(); i++){
626  src.indexIn(strl.at(i));
627  QStringList urls = src.capturedTexts();
628  if(2 <= urls.size() && urls.at(1) != "" && !urlsFound.contains(urls.at(1))){
629  urlsFound << urls.at(1);
630  pList->append(QUrl(urls.at(1)));
631  }
632  }
633  }
634  }
