294 void UBPlatformUtils::bringPreviousProcessToFront()
295 {
296  ProcessSerialNumber previousProcessPSN = {0, kNoProcess};
297 
298  CFArrayRef apps = CopyLaunchedApplicationsInFrontToBackOrder();
299  if (apps != NULL)
300  {
301  if (CFArrayGetCount(apps) > 1)
302  {
303  SInt64 psn64;
304  CFDictionaryRef appInfo = (CFDictionaryRef)CFArrayGetValueAtIndex(apps, 1);
305  CFNumberRef psn = (CFNumberRef)CFDictionaryGetValue(appInfo, CFSTR("PSN"));
306  if (psn != NULL)
307  {
308  CFNumberGetValue(psn, kCFNumberSInt64Type, &psn64);
309  previousProcessPSN.highLongOfPSN = psn64 >> 32;
310  previousProcessPSN.lowLongOfPSN = psn64 & 0xFFFFFFFF;
311  }
312  }
313  CFRelease(apps);
314  }
315  else
316  {
317  // On 10.4, we can't get the apps in front to back order, so we default to Finder
318  OSStatus status;
319  ProcessSerialNumber psn = {0, kNoProcess};
320  while ((status = GetNextProcess(&psn)) == noErr)
321  {
322  CFDictionaryRef processInfo = ProcessInformationCopyDictionary(&psn, kProcessDictionaryIncludeAllInformationMask);
323  if (processInfo != NULL)
324  {
325  CFStringRef bundleIdentifier = (CFStringRef)CFDictionaryGetValue(processInfo, kCFBundleIdentifierKey);
326  if (bundleIdentifier && CFStringCompare(CFSTR("com.apple.finder"), bundleIdentifier, kCFCompareCaseInsensitive) == kCFCompareEqualTo)
327  {
328  previousProcessPSN.highLongOfPSN = psn.highLongOfPSN;
329  previousProcessPSN.lowLongOfPSN = psn.lowLongOfPSN;
330  }
331  CFRelease(processInfo);
332  }
333  }
334  }
335 
336  SetFrontProcess(&previousProcessPSN);
337 }

368 QPixmap qpixmapFromIconRef(IconRef iconRef, int size) {
369  OSErr result;
370  int iconSize;
371  OSType elementType;
372 
373  // Determine elementType and iconSize
374  if (size <= 16) {
375  elementType = kSmall32BitData;
376  iconSize = 16;
377  } else if (size <= 32) {
378  elementType = kLarge32BitData;
379  iconSize = 32;
380  } else {
381  elementType = kThumbnail32BitData;
382  iconSize = 128;
383  }
384 
385  // Get icon into an IconFamily
386  IconFamilyHandle hIconFamily = 0;
387  IconRefToIconFamily(iconRef, kSelectorAllAvailableData, &hIconFamily);
388 
389  // Extract data
390  Handle hRawBitmapData = NewHandle(iconSize * iconSize * 4);
391  result = GetIconFamilyData( hIconFamily, elementType, hRawBitmapData );
392  if (result != noErr) {
393  DisposeHandle(hRawBitmapData);
394  return QPixmap();
395  }
396 
397  // Convert data to QImage
398  QImage image(iconSize, iconSize, QImage::Format_ARGB32);
399  HLock(hRawBitmapData);
400  unsigned long* data = (unsigned long*) *hRawBitmapData;
401  for (int posy=0; posy<iconSize; ++posy, data+=iconSize) {
402 #ifdef __BIG_ENDIAN__
403  uchar* line = image.scanLine(posy);
404  memcpy(line, data, iconSize * 4);
405 #else
406  uchar* src = (uchar*) data;
407  uchar* dst = image.scanLine(posy);
408  for (int posx=0; posx<iconSize; src+=4, dst+=4, ++posx) {
409  dst[0] = src[3];
410  dst[1] = src[2];
411  dst[2] = src[1];
412  dst[3] = src[0];
413  }
414 #endif
415  }
416  HUnlock(hRawBitmapData);
417  DisposeHandle( hRawBitmapData );
418 
419  // Scale to wanted size
420  image = image.scaled(size, size, Qt::KeepAspectRatio, Qt::SmoothTransformation);
421  return QPixmap::fromImage(image);
422 }

437 KEYBT* createKeyBt(const UCKeyboardLayout* keyLayout, int vkk)
438 {
439  UInt32 deadKeyState = 0L;
440  UInt32 kbdType = kKeyboardISO;
441 
442  UniCharCount cnt1, cnt2;
443  UniChar unicodeSimple[100], unicodeShifted[100], unicodeCapsed[100];
444 
445  UCKeyTranslate(keyLayout, vkk, kUCKeyActionDisplay, 0, kbdType, kUCKeyTranslateNoDeadKeysBit, &deadKeyState, 100, &cnt1, unicodeSimple);
446  UCKeyTranslate(keyLayout, vkk, kUCKeyActionDisplay, (shiftKey >> 8) & 0xff, kbdType, kUCKeyTranslateNoDeadKeysBit, &deadKeyState, 100, &cnt2, unicodeShifted);
447  UCKeyTranslate(keyLayout, vkk, kUCKeyActionDisplay, (alphaLock >> 8) & 0xff, kbdType, kUCKeyTranslateNoDeadKeysBit, &deadKeyState, 100, &cnt2, unicodeCapsed);
448 
449  // workaround for wrong character returned from UCKeyTranslate for swiss-french keyboard.
450  if (unicodeSimple[0] == 94 && unicodeShifted[0] == 80)
451  {
452  unicodeShifted[0] = 168;
453  unicodeCapsed[0] = unicodeSimple[0];
454  }
455 
456  if (unicodeSimple[0] == 94 && unicodeShifted[0] == 63)
457  {
458  unicodeShifted[0] = 96;
459  unicodeCapsed[0] = unicodeSimple[0];
460  }
461  return new KEYBT(unicodeSimple[0], unicodeShifted[0], unicodeCapsed[0], 0,0,0, KEYCODE(0, vkk, 0), KEYCODE(0, vkk, 1), KEYCODE(0, vkk, 2));
462 }
