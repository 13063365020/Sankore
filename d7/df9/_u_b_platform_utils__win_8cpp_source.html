91 void UBPlatformUtils::runInstaller(const QString &installerFilePath)
92 {
93  QProcess process;
94 
95  QString escaped = QString("\"") + installerFilePath + QString("\" /SILENT");
96 
97  bool success = process.startDetached(escaped);
98  if (!success)
99  {
100  qWarning() << "Running '" << installerFilePath << "' failed (error=" << process.error() << ")";
101  QString verb = "runas";
102  ::ShellExecute(NULL, verb.utf16(), installerFilePath.utf16(), NULL, NULL, SW_HIDE);
103  }
104 }
