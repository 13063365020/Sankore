162 QImage* XPDFRenderer::createPDFImage(int pageNumber, qreal xscale, qreal yscale, const QRectF &bounds)
163 {
164  if (isValid())

172  GBool useMediaBox = gFalse;

175  mSliceX = 0.;
176  mSliceY = 0.;
177

181  rotation, useMediaBox, crop, printing);
182  }
183  else
184  {
185  mSliceX = bounds.x() * xscale;
186  mSliceY = bounds.y() * yscale;
187  qreal sliceW = bounds.width() * xscale;
188  qreal sliceH = bounds.height() * yscale;
189 
190  mDocument->displayPageSlice(mSplash, pageNumber, this->dpiForRendering * xscale, this->dpiForRendering * yscale,
191  rotation, useMediaBox, crop, printing, mSliceX, mSliceY, sliceW, sliceH);
192  }
193 
194  mpSplashBitmap = mSplash->getBitmap();
195  }
196  return new QImage(mpSplashBitmap->getDataPtr(), mpSplashBitmap->getWidth(), mpSplashBitmap->getHeight(), mpSplashBitmap->getWidth() * 3, QImage::Format_RGB888);
