70 void UBGraphicsGroupContainerItem::addToGroup(QGraphicsItem *item,bool removeAction)
71 {
72  if (!item) {
73  qWarning("UBGraphicsGroupContainerItem::addToGroup: cannot add null item");
74  return;
75  }
76  if (item == this) {
77  qWarning("UBGraphicsGroupContainerItem::addToGroup: cannot add a group to itself");
78  return;
79  }
80 
81  //TODO claudio
82  UBGraphicsItem* ubGraphics = dynamic_cast<UBGraphicsItem*>(item);
83  if(ubGraphics && ubGraphics->Delegate() && removeAction)
84  ubGraphics->Delegate()->setAction(0);
85 
86 
87  //Check if group is allready rotatable or flippable
88  if (childItems().count()) {
89  if (UBGraphicsItem::isFlippable(this) && !UBGraphicsItem::isFlippable(item)) {
90  Delegate()->setFlippable(false);
91  }
92  if (UBGraphicsItem::isRotatable(this) && !UBGraphicsItem::isRotatable(item)) {
93  Delegate()->setRotatable(false);
94  }
95  if (!UBGraphicsItem::isLocked(this) && UBGraphicsItem::isLocked(item)) {
96  Delegate()->setLocked(true);
97  }
98  } else {
99  Delegate()->setFlippable(UBGraphicsItem::isFlippable(item));
100  Delegate()->setRotatable(UBGraphicsItem::isRotatable(item));
101  Delegate()->setLocked(UBGraphicsItem::isLocked(item));
102  }
103 
104  // COMBINE
105  bool ok;
106  QTransform itemTransform = item->itemTransform(this, &ok);
107 
108  if (!ok) {
109  qWarning("UBGraphicsGroupContainerItem::addToGroup: could not find a valid transformation from item to group coordinates");
110  return;
111  }
112 
113  //setting item flags to given item
114  item->setSelected(false);
115  item->setFlag(QGraphicsItem::ItemIsSelectable, false);
116  item->setFlag( QGraphicsItem::ItemIsMovable, false);
117  item->setFlag(QGraphicsItem::ItemIsFocusable, true);
118 
119  QTransform newItemTransform(itemTransform);
120  item->setPos(mapFromItem(item, 0, 0));
121 
122  if (item->scene()) {
123  item->scene()->removeItem(item);
124  }
125 
126  if (corescene())
127  corescene()->removeItemFromDeletion(item);
128  item->setParentItem(this);
129 
130  // removing position from translation component of the new transform
131  if (!item->pos().isNull())
132  newItemTransform *= QTransform::fromTranslate(-item->x(), -item->y());
133 
134  // removing additional transformations properties applied with itemTransform()
135  QPointF origin = item->transformOriginPoint();
136  QMatrix4x4 m;
137  QList<QGraphicsTransform*> transformList = item->transformations();
138  for (int i = 0; i < transformList.size(); ++i)
139  transformList.at(i)->applyTo(&m);
140  newItemTransform *= m.toTransform().inverted();
141  newItemTransform.translate(origin.x(), origin.y());
142  newItemTransform.rotate(-item->rotation());
143  newItemTransform.scale(1/item->scale(), 1/item->scale());
144  newItemTransform.translate(-origin.x(), -origin.y());
145 
146  // ### Expensive, we could maybe use dirtySceneTransform bit for optimization
147 
148  item->setTransform(newItemTransform);
149  // item->d_func()->setIsMemberOfGroup(true);
150  prepareGeometryChange();
151  itemsBoundingRect |= itemTransform.mapRect(item->boundingRect() | item->childrenBoundingRect());
152  update();
153 }

341 void UBGraphicsGroupContainerItem::pRemoveFromGroup(QGraphicsItem *item)
342 {
343  if (!item) {
344  qWarning("QGraphicsItemGroup::removeFromGroup: cannot remove null item");
345  return;
346  }
347 
348  QGraphicsItem *newParent = parentItem();
349 
350  if (childItems().count()) {
351  if (!UBGraphicsItem::isFlippable(item) || !UBGraphicsItem::isRotatable(item)) {
352  bool flippableNow = true;
353  bool rotatableNow = true;
354  bool lockedNow = false;
355 
356  foreach (QGraphicsItem *item, childItems()) {
357  if (!UBGraphicsItem::isFlippable(item)) {
358  flippableNow = false;
359  }
360  if (!UBGraphicsItem::isRotatable(item)) {
361  rotatableNow = false;
362  }
363  if(UBGraphicsItem::isLocked(item))
364  lockedNow = true;
365 
366  if (!rotatableNow && !flippableNow && lockedNow) {
367  break;
368  }
369  }
370  Delegate()->setFlippable(flippableNow);
371  Delegate()->setRotatable(rotatableNow);
372  Delegate()->setLocked(lockedNow);
373  }
374  }
375 
376  // COMBINE
377  bool ok;
378  QTransform itemTransform;
379  if (newParent)
380  itemTransform = item->itemTransform(newParent, &ok);
381  else
382  itemTransform = item->sceneTransform();
383 
384  QPointF oldPos = item->mapToItem(newParent, 0, 0);
385  item->setParentItem(newParent);
386  item->setPos(oldPos);
387 
388  UBGraphicsScene *Scene = dynamic_cast<UBGraphicsScene *>(item->scene());
389  if (Scene)
390  {
391  Scene->addItem(item);
392  }
393 
394  // removing position from translation component of the new transform
395  if (!item->pos().isNull())
396  itemTransform *= QTransform::fromTranslate(-item->x(), -item->y());
397 
398  // removing additional transformations properties applied
399  // with itemTransform() or sceneTransform()
400  QPointF origin = item->transformOriginPoint();
401  QMatrix4x4 m;
402  QList<QGraphicsTransform*> transformList = item->transformations();
403  for (int i = 0; i < transformList.size(); ++i)
404  transformList.at(i)->applyTo(&m);
405  itemTransform *= m.toTransform().inverted();
406  itemTransform.translate(origin.x(), origin.y());
407  itemTransform.rotate(-item->rotation());
408  itemTransform.scale(1 / item->scale(), 1 / item->scale());
409  itemTransform.translate(-origin.x(), -origin.y());
410 
411  // ### Expensive, we could maybe use dirtySceneTransform bit for optimization
412 
413  item->setTransform(itemTransform);
414 // item->d_func()->setIsMemberOfGroup(item->group() != 0);
415 
416  // ### Quite expensive. But removeFromGroup() isn't called very often.
417  prepareGeometryChange();
418  itemsBoundingRect = childrenBoundingRect();
419 }
