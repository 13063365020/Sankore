198 static void _updateAnnotParentPage(Object *annotation,Object *newParentPage)
199 {
200  if( annotation )
201  {
202  std::string strP = "/P";
203  std::string &annotContent = annotation->getObjectContent();
204 
205  size_t startOfP = Parser::findTokenName(annotContent,strP);
206  if((int) startOfP == -1 )
207  {
208  return;
209  }
210  size_t endOfP = Parser::findEndOfElementContent(annotContent,startOfP + strP.size());
211 
212  // lets find object with reference to parent
213  std::vector<Object *> children = annotation->getChildrenByBounds(startOfP,endOfP);
214  if( children.size() == 0 )
215  {
216  return;
217  }
218 
219  Object *childWithP = children[0];
220  if( childWithP )
221  {
222  Object::ReferencePositionsInContent pagePosition = annotation->removeChild(childWithP);
223  annotation->eraseContent(startOfP,endOfP-startOfP);
224  std::stringstream strout;
225  strout <<"/P "<<newParentPage->getObjectNumber()<<" "<<newParentPage->getgenerationNumber()<<" R\n";
226  // to compensate posible deviation
227  for(size_t i =strout.str().size();i<endOfP-startOfP;i++)
228  {
229  strout<<" ";
230  }
231  annotation->insertToContent(startOfP,strout.str());
232  annotation->addChild(newParentPage,pagePosition);
233  }
234  }

237 // function performs adjusting of some color parameters of annotation
238 // to avoid interference with overlay content
239 static void _updateAnnotFormColor(Object *annotation )
240 {
241  std::string &objectContent = annotation->getObjectContent();
242  if((int) objectContent.find("/Widget") == -1 )
243  {
244  return;
245  }
246  size_t startOfAP = Parser::findTokenName(objectContent,"/AP");
247  if((int) startOfAP == -1 )
248  {
249  return;
250  }
251  size_t endOfAP = objectContent.find(">>", startOfAP);
252  std::vector<Object *> aps = annotation->getChildrenByBounds(startOfAP, endOfAP);
253  for(size_t i = 0; i < aps.size(); ++i)
254  {
255  Object * childWithAP = aps[i];
256  if( !childWithAP->hasStream() )
257  {
258  continue;
259  }
260  // first lets obtain and decode stream of Annotation appearrence stream
261  std::string & content = childWithAP->getObjectContent();
262 
263  Filter filter(childWithAP);
264  std::string decodedStream;
265  filter.getDecodedStream(decodedStream);
266 
267  // lets iterate over stream and find operator f and remove it!
268  size_t beg = 0;
269  size_t found = 0;
270  std::string token;
271  while(Parser::getNextWord(token,decodedStream,beg,&found))
272  {
273  if( token == "f" || token == "F" )
274  {
275  if((int) found != -1 )
276  {
277  decodedStream[found] = ' ';
278  }
279  break;
280  }
281  }
282  // Then we need to update Filter section (if any)
283  std::string filterStr = "/Filter";
284  size_t startOfFlate = Parser::findTokenName(content,filterStr);
285  if((int) startOfFlate != -1 )
286  {
287  size_t endOfFlate = Parser::findEndOfElementContent(content,startOfFlate+filterStr.size());
288  childWithAP->eraseContent(startOfFlate,endOfFlate-startOfFlate);
289  //encode and put new stream to object content
290  childWithAP->insertToContent(startOfFlate,"/Filter /FlateDecode ");
291  FlateDecode flate;
292  flate.encode(decodedStream);
293  }
294 
295  // update the length field
296  std::string lengthStr = "/Length";
297  size_t startOfLength = Parser::findTokenName(content,lengthStr,0);
298  if((int) startOfLength != -1 )
299  {
300  size_t endOfLength = Parser::findEndOfElementContent(content,startOfLength + lengthStr.size());
301  childWithAP->eraseContent(startOfLength,endOfLength-startOfLength);
302  std::stringstream ostr;
303  ostr<<"/Length "<< decodedStream.size()<<"\n";
304  childWithAP->insertToContent(startOfLength,ostr.str());
305 
306  // update the stream of object with new content
307  std::string stream("stream");
308  size_t leftBoundOfContentStream = content.find(stream);
309  if((int) leftBoundOfContentStream != -1 )
310  {
311  size_t rightBoundOfContentStream = content.find("endstream", leftBoundOfContentStream);
312  if((int) rightBoundOfContentStream == -1 )
313  {
314  rightBoundOfContentStream = content.size() - 1;
315  }
316  childWithAP->eraseContent(leftBoundOfContentStream, rightBoundOfContentStream - leftBoundOfContentStream);
317  decodedStream.insert(0,"\nstream\n");
318  childWithAP->insertToContent(leftBoundOfContentStream,decodedStream);
319  childWithAP->appendContent("endstream\n");
320  childWithAP->forgetStreamInFile();
321  }
322  }
323  }

325 // sometimes page object does not have resources,
326 // they are inherited from parent object
327 // this method processes such cases and insert resources from parent to page
328 // for correct X-Object transformation
329 static void processBasePageResources(Object *basePage)
330 {
331  if( basePage == NULL )
332  {
333  return;
334  }
335  std::string resourceToken = "/Resources";
336  if((int) Parser::findTokenName(basePage->getObjectContent(),resourceToken) == -1 )
337  {
338  // it seems base page does not have resources, they can be located in parent!
339  Object *resource = basePage->findPatternInObjOrParents(resourceToken);
340  if( resource )
341  {
342  std::string &resContStr = resource->getObjectContent();
343  size_t startOfRes = Parser::findTokenName(resContStr,resourceToken);
344  if((int) startOfRes == -1 )
345  {
346  // no resources at all
347  return;
348  }
349  size_t endOfRes = Parser::findEndOfElementContent(resContStr,startOfRes + resourceToken.size());
350  if((int) endOfRes == -1 )
351  {
352  return; // broken resources
353  }
354  std::string resourceContent = resContStr.substr(startOfRes,endOfRes-startOfRes);
355 
356  size_t positionToInsert = basePage->getObjectContent().find("<<");
357  if((int) positionToInsert == -1 )
358  {
359  positionToInsert = 0;
360  resourceContent.insert(0,"<<");
361  resourceContent.append(">>");
362  }
363  else
364  {
365  positionToInsert += strlen("<<");
366  }
367  // insert obtained resources to base page
368  basePage->insertToContent(positionToInsert,resourceContent);
369 
370  // if resource contains childs, then we need to add reference to them to current object
371  std::vector<Object *> resChilds = resource->getChildrenByBounds(startOfRes, endOfRes);
372  std::vector<Object *>::const_iterator objectIt(resChilds.begin());
373 
374  const Object::Children & children = resource->getChildren();
375 
376  for(; objectIt != resChilds.end(); objectIt++ )
377  {
378  Object::Children::const_iterator childrenIt = children.find( (*objectIt)->getObjectNumber());
379  if( childrenIt != children.end() )
380  {
381  Object::ReferencePositionsInContent refPositionInCont = (*childrenIt).second.second;
382 
383  Object::ReferencePositionsInContent::iterator positionIt(refPositionInCont.begin());
384 
385  Object::ReferencePositionsInContent newPositions;
386  for( ;positionIt != refPositionInCont.end(); positionIt++ )
387  {
388  newPositions.push_back( (*positionIt) - startOfRes + positionToInsert );
389  }
390 
391  basePage->addChild( (*objectIt),newPositions );
392  }
393  }
394  }
395  }

466 void Page::merge(Page * sourcePage, Document * parentDocument, MergePageDescription & description, bool isPageDuplicated)
467 {
468  if( sourcePage == NULL )
469  {
470  description.skipBasePage = true;
471  }
472  if(!description.skipOverlayPage)
473  {
474  // Lets recalculate final transformation of overlay page
475  // before it will be places into XObject
476  Rectangle mediaBox("/MediaBox",_root->getObjectContent());
477  description.overlayPageTransformation.recalculateTranslation(mediaBox.getWidth(),mediaBox.getHeight());
478 
479  std::vector<Object *> fake;
480  _pageToXObject(_root, fake);
481  }
482 
483  std::vector<Object *> toAllObjects;
484  std::vector<Object *> annotations;
485  Object * sourcePageToXObject = 0;
486  if(!description.skipBasePage)
487  {
488  RotationHandler rotationHandler(sourcePage->_root, "/Rotate", *this);
489  rotationHandler.processObjectContent();
490  description.basePageTransformation.addRotation(_rotation);
491 
492  if((int) sourcePage->_root->getObjectContent().find("/Annots") != -1 )
493  {
494  Object *crop = sourcePage->_root->findPatternInObjOrParents("/CropBox");
495  if( crop )
496  {
497  // we need to calculate special compensational shifting
498  // for annotations if cropbox is starting not from 0,0
499  Rectangle mediaBox("/CropBox", crop->getObjectContent());
500  if( !Utils::doubleEquals(mediaBox.x1,0) || !Utils::doubleEquals(mediaBox.y1,0) )
501  {
502  double shiftX = Utils::doubleEquals(mediaBox.x1,0)?0:-mediaBox.x1;
503  double shiftY = Utils::doubleEquals(mediaBox.y1,0)?0:-mediaBox.y1;
504  Translation compensation(shiftX,shiftY);
505  description.basePageTransformation.addAnnotsTransformation(compensation);
506  }
507  }
508  }
509  processBasePageResources(sourcePage->_root);
510  sourcePageToXObject = sourcePage->pageToXObject(toAllObjects, annotations, isPageDuplicated);
511  Rectangle mediaBox("/BBox", sourcePageToXObject->getObjectContent());
512  description.basePageTransformation.recalculateTranslation(mediaBox.getWidth(),mediaBox.getHeight());
513  }
514 
515  Object * catalog = 0;
516  unsigned int fake;
517  if(!parentDocument->getDocumentObject()->findObject(std::string("/Kids"), catalog, fake))
518  {
519  std::string error("Wrong document ");
520  error.append("There is no object with Kids field");
521  throw Exception(error);
522  }
523  Object::ReferencePositionsInContent pagePosition = catalog->removeChild(_root);
524  //create merged Page
525  unsigned int contentPosition, parentPosition, originalPage1Position, originalPage2Position;
526  std::pair<unsigned int, unsigned int> originalPageNumbers(_root->getObjectNumber(), 0);
527  if(!description.skipBasePage)
528  originalPageNumbers.second = sourcePageToXObject->getObjectNumber();
529  std::vector <Object::ChildAndItPositionInContent> annotsAndItPositions;
530  std::string mergedPageContent = _getMergedPageContent(contentPosition,
531  parentPosition,
532  originalPage1Position,
533  originalPage2Position,
534  originalPageNumbers,
535  description,
536  sourcePageToXObject,
537  annotations,
538  annotsAndItPositions
539  );
540  Object * mergedPage = new Object(_root->getObjectNumber(), _root->getgenerationNumber(), mergedPageContent);
541  toAllObjects.push_back(mergedPage);
542  std::vector < unsigned int > contentPositionVec, parentPositionVec, originalPage1PositionVec, originalPage2PositionVec;
543  contentPositionVec.push_back(contentPosition);
544  parentPositionVec.push_back(parentPosition);
545  originalPage1PositionVec.push_back(originalPage1Position);
546  originalPage2PositionVec.push_back(originalPage2Position);
547  Object * contentOfMergedPage = new Object(1, 0, _getContentOfContentObject(description));
548  toAllObjects.push_back(contentOfMergedPage);
549  parentDocument->addToAllObjects(toAllObjects);
550  mergedPage->addChild(contentOfMergedPage, contentPositionVec);
551  mergedPage->addChild(catalog, parentPositionVec);
552  if(!description.skipOverlayPage)
553  mergedPage->addChild(_root, originalPage1PositionVec);
554  if(!description.skipBasePage)
555  mergedPage->addChild(sourcePageToXObject, originalPage2PositionVec);
556  // Annotation parent page should be changed, since we moved old page
557  // to Xobject
558  if( !description.skipBasePage )
559  {
560  for(size_t i = 0; i< annotations.size();i++)
561  {
562  _updateAnnotParentPage(annotations[i],mergedPage);
563  }
564  }
565  for(size_t i = 0; i < annotsAndItPositions.size(); ++i)
566  {
567  mergedPage->addChild(annotsAndItPositions[i].first, annotsAndItPositions[i].second);
568  }
569  catalog->addChild(mergedPage, pagePosition);
570  _root = mergedPage;
571 }
