145 void UBMagnifier::createMask()
146 {
147  if(gView == NULL || mView == NULL) return;
148 
149  // calculate object size
150  QSize sceneSize = mView->size();
151  qreal isize = params.sizePercentFromScene * sceneSize.width() / 100;
152 
153  QImage mask_img(width(), height(), QImage::Format_Mono);
154  mask_img.fill(0xff);

157

161  mask_ptr.drawRoundedRect(QRect(sClosePixmap->width(), sClosePixmap->width(), size().width() - 2*sClosePixmap->width(), size().height() - 2*sClosePixmap->width()), sClosePixmap->width()/2, sClosePixmap->width()/2);

357 void UBMagnifier::grabPoint()
358 {
359  QMatrix transM = UBApplication::boardController->controlView()->matrix();
360  QPointF itemPos = gView->mapFromGlobal(updPointGrab);
361 
362  qreal zWidth = width() / (params.zoom * transM.m11());
363  qreal zWidthHalf = zWidth / 2;
364  qreal zHeight = height() / (params.zoom * transM.m22());
365  qreal zHeightHalf = zHeight / 2;
366 
367 
368  QPointF pfScLtF(UBApplication::boardController->controlView()->mapToScene(QPoint(itemPos.x(), itemPos.y())));
369 
370  float x = pfScLtF.x() - zWidthHalf;
371  float y = pfScLtF.y() - zHeightHalf;
372 
373  QPointF leftTop(x,y);
374  QPointF rightBottom(x + zWidth, y + zHeight);
375  QRectF srcRect(leftTop, rightBottom);
376 
377  QPixmap newPixMap(QSize(width(), height()));
378  QPainter painter(&newPixMap);
379 
380  UBApplication::boardController->activeScene()->render(&painter, QRectF(0,0,width(),height()), srcRect);
381  painter.end();
382 
383  pMap.fill(Qt::transparent);
384  pMap = newPixMap.scaled(QSize(width(), height()));
385  pMap.setMask(bmpMask);
386 
387  update();
388 }
