117 void UBGraphicsPolygonItem::setColor(const QColor& pColor)
118 {
119  QGraphicsPolygonItem::setBrush(QBrush(pColor));
120 
121  if (pColor.alphaF() >= 1.0)
122  {
123  mHasAlpha = false;
124  setPen(Qt::NoPen);
125  }
126  else
127  {
128  mHasAlpha = true;
129 
130  QColor penColor = pColor;
131 
132  // trick QT antialiasing
133  // TODO UB 4.x see if we can do better ... it does not behave well with 16 bit color depth
134  qreal trickAlpha = pColor.alphaF();
135 
136  if (trickAlpha >= 0.2 && trickAlpha < 0.6)
137  {
138  trickAlpha /= 12;
139  }
140  else if (trickAlpha < 0.8)
141  {
142  trickAlpha /= 5;
143  }
144  else if (trickAlpha < 1.0)
145  {
146  trickAlpha /= 2;
147  }
148 
149  penColor.setAlphaF(trickAlpha);
150  QGraphicsPolygonItem::setPen(QPen(penColor));
151  }
152 }
