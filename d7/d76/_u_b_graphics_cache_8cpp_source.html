158 void UBGraphicsCache::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
159 {

163

183  else if(eMaskShap_Rectangle == mMaskShape)

185  path.addRect(mHolePos.x() - mHoleSize.width()/2, mHolePos.y() - mHoleSize.height()/2, mHoleSize.width(), mHoleSize.height());

187  path.setFillRule(Qt::OddEvenFill);
188  }
189 
190  painter->drawPath(path);

297 QRectF UBGraphicsCache::updateRect(QPointF currentPoint)
298 {
299  QRectF r;
300  int x;
301  int y;
302 
303  x = qMin(currentPoint.x() - mHoleSize.width()/2, mOldShapePos.x() - mOldShapeSize.width()/2);
304  y = qMin(currentPoint.y() - mHoleSize.height()/2, mOldShapePos.y() - mOldShapeSize.height()/2);
305  r = QRect( x,
306  y,
307  qAbs(currentPoint.x() - mOldShapePos.x()) + mHoleSize.width(),
308  qAbs(currentPoint.y() - mOldShapePos.y()) + mHoleSize.height());
309  return r;
