603 bool UBApplication::eventFilter(QObject *obj, QEvent *event)
604 {
605  bool result = QObject::eventFilter(obj, event);
606 
607  if (event->type() == QEvent::FileOpen)
608  {
609  QFileOpenEvent *fileToOpenEvent = static_cast<QFileOpenEvent *>(event);
610 
611 #if defined(Q_WS_MACX)
612  ProcessSerialNumber psn;
613  GetCurrentProcess(&psn);
614  SetFrontProcess(&psn);
615 #endif
616 
617  applicationController->importFile(fileToOpenEvent->file());
618  }
619 
620  if (event->type() == QEvent::TabletLeaveProximity)
621  {
622  if (boardController && boardController->controlView())
623  boardController->controlView()->forcedTabletRelease();
624  }
625 
626 
627  if (event->type() == QEvent::ApplicationActivate)
628  {
629  boardController->controlView()->setMultiselection(false);
630  }
631 
632 #ifdef Q_WS_MAC
633  if (bIsMinimized && event->type() == QEvent::ApplicationActivate){
634  if (mainWindow->isHidden()) mainWindow->show();
635  bIsMinimized = false;
636  }
637 #endif
638  return result;
639 }
