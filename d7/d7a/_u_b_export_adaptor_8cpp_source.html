54 QString UBExportAdaptor::askForFileName(UBDocumentProxy* pDocument, const QString& pDialogTitle)
55 {
56  QString defaultName;
57 
58  defaultName += pDocument->metaData(UBSettings::documentName).toString() + exportExtention();
59 
60  defaultName = UBFileSystemUtils::cleanName(defaultName);
61  QString defaultPath = UBSettings::settings()->lastExportFilePath->get().toString() + "/" + defaultName;
62 
63  QString filename = QFileDialog::getSaveFileName(UBApplication::mainWindow, pDialogTitle, defaultPath);
64 
65  if (filename.size() == 0)
66  {
67  return filename;
68  }
69 
70  // add extension if needed
71  QFileInfo fileNameInfo(filename);
72  if (fileNameInfo.suffix() != exportExtention().mid(1, exportExtention().length() - 1))
73  {
74  filename += exportExtention();
75  }
76  UBSettings::settings()->lastExportFilePath->set(QVariant(fileNameInfo.absolutePath()));
77  QApplication::processEvents();
78 
79  return filename;
80 }
