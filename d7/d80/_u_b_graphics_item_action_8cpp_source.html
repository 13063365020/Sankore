48 UBGraphicsItemPlayAudioAction::UBGraphicsItemPlayAudioAction(QString audioFile, bool isNewAction, QObject *parent) :
49  UBGraphicsItemAction(eLinkToAudio,parent)
50  , mMediaObject(0)
51  , mIsLoading(true)
52 {
53  Q_ASSERT(audioFile.length() > 0);
54  if(isNewAction){
55  QString extension = QFileInfo(audioFile).completeSuffix();
56  QString destDir = UBApplication::boardController->selectedDocument()->persistencePath() + "/" + UBPersistenceManager::audioDirectory;
57  QString destFile = destDir + "/" + QUuid::createUuid().toString() + "." + extension;
58  if(!QDir(destDir).exists())
59  QDir(UBApplication::boardController->selectedDocument()->persistencePath()).mkdir(destDir);
60  //explanation : the audioFile could be relative. The method copy will return false so a second try is done adding
61  // the document file path
62  if(!QFile(audioFile).copy(destFile))
63  QFile(UBApplication::boardController->selectedDocument()->persistencePath() + "/" + audioFile).copy(destFile);
64  mAudioPath = destFile;
65  mFullPath = destFile;
66  }
67  else{
68  //another hack
69  if(UBApplication::documentController && UBApplication::documentController->selectedDocument()){
70  mAudioPath = UBApplication::documentController->selectedDocument()->persistencePath() + "/" + audioFile;
71  mFullPath = mAudioPath;
72  }
73  else
74  return;
75  }
76  mAudioOutput = new Phonon::AudioOutput(Phonon::MusicCategory, this);
77  mMediaObject = new Phonon::MediaObject(this);
78  Phonon::createPath(mMediaObject, mAudioOutput);
79  mMediaObject->setCurrentSource(Phonon::MediaSource(mAudioPath));
80 }

157 void UBGraphicsItemMoveToPageAction::play()
158 {
159  UBBoardController* boardController = UBApplication::boardController;
160 
161  switch (mActionType) {
162  case eMoveToFirstPage:
163  boardController->firstScene();
164  break;
165  case eMoveToLastPage:
166  boardController->lastScene();
167  break;
168  case eMoveToPreviousPage:
169  boardController->previousScene();
170  break;
171  case eMoveToNextPage:
172  boardController->nextScene();
173  break;
174  case eMoveToPage:
175  if(mPage > 0 && mPage < boardController->pageCount())
176  boardController->setActiveDocumentScene(mPage);
177  else
178  qWarning() << "scene number " << mPage << "ins't accessible anymore";
179  break;
180  default:
181  break;
182  }
183 }
