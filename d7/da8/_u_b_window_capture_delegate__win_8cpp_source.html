51 int UBWindowCaptureDelegate::execute()
52 {
53  mIsCapturing = true;
54  mCancel = false;
55  qApp->installEventFilter(this);
56 
57  while (mIsCapturing && !mCancel)
58  {
59  QPoint curPos = QCursor::pos();
60  if (mLastPoint != curPos)
61  {
62  processPos(curPos);
63  mLastPoint = curPos;
64  }
65  drawSelectionRect();
66  qApp->processEvents();
67  }
68 
69  if (!mCancel)
70  {
71  BringWindowToTop(mCurrentWindow);
72  RedrawWindow(mCurrentWindow, NULL, NULL, RDW_INVALIDATE | RDW_UPDATENOW | RDW_ALLCHILDREN);
73  QWaitCondition sleep;
74  QMutex mutex;
75  mutex.lock();
76  sleep.wait(&mutex, 200);
77  mutex.unlock();
78  mCapturedPixmap = QPixmap::grabWindow(mCurrentWindow);
79  return QDialog::Accepted;
80  }
81  else
82  {
83  RedrawWindow(NULL, NULL, NULL, RDW_INVALIDATE | RDW_UPDATENOW | RDW_ALLCHILDREN);
84  return QDialog::Rejected;
85  }
86 }

89 void UBWindowCaptureDelegate::processPos(QPoint pPoint)
90 {
91  POINT wPoint;
92 
93  wPoint.x = pPoint.x();
94  wPoint.y = pPoint.y();
95  HWND desktop = GetDesktopWindow();
96  HWND newWindow = ChildWindowFromPointEx(desktop, wPoint, CWP_SKIPINVISIBLE | CWP_SKIPTRANSPARENT);
97 
98  if (mCurrentWindow != newWindow && newWindow != desktop)
99  {
100  RedrawWindow(NULL, NULL, NULL, RDW_INVALIDATE | RDW_UPDATENOW | RDW_ALLCHILDREN);
101  mCurrentWindow = newWindow;
102  }
103 }

106 void UBWindowCaptureDelegate::drawSelectionRect()
107 {
108  RECT windowRect;
109  if (GetWindowRect(mCurrentWindow, &windowRect))
110  {
111  HDC hDC;
112  HPEN hOldPen;
113  HBRUSH hOldBrush;
114 
115  hDC = GetDC(NULL);
116 
117  hOldPen = (HPEN)SelectObject(hDC, CreatePen(PS_SOLID, 5, RGB(255, 0, 0)));
118  hOldBrush = (HBRUSH)SelectObject(hDC, GetStockObject(NULL_BRUSH));
119 
120  Rectangle(hDC, windowRect.left + 1, windowRect.top + 1, windowRect.right - 1, windowRect.bottom - 1);
121 
122  SelectObject(hDC, hOldBrush);
123  DeleteObject(SelectObject(hDC, hOldPen));
124 
125  ReleaseDC(NULL, hDC);
126  }
127 }

130 bool UBWindowCaptureDelegate::eventFilter(QObject *target, QEvent *event)
131 {
132  Q_UNUSED(target);
133 
134  // once the application is deactivated, it means that the user clicked somewhere else and we shall
135  // take the capture
136  if (event->type() == QEvent::MouseButtonPress || event->type() == QEvent::WindowDeactivate)
137  {
138  mIsCapturing = false;
139  }
140  else if (event->type() == QEvent::KeyPress)
141  {
142  mCancel = true;
143  }
144  return true;
145 }
