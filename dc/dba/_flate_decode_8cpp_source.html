73 bool FlateDecode::encode(std::string &decoded)
74 {
75  z_stream stream;
76 
77  stream.zalloc = (alloc_func)0;
78  stream.zfree = (free_func)0;
79  stream.opaque = (voidpf)0;
80 
81  size_t out_len = 0;
82  unsigned char *out_p = NULL;
83 
84  stream.next_out = out_p;
85  stream.avail_out = (uInt)out_len;
86 
87  stream.next_in = (unsigned char*)decoded.c_str();
88  stream.avail_in = (uInt)decoded.size();
89 
90  int err = deflateInit(&stream, Z_DEFAULT_COMPRESSION);
91  ZLIB_CHECK_ERR(err, "deflateInit");
92  if ( err != Z_OK )
93  {
94  return false;
95  }
96  bool toContinue = false;
97  int flush = Z_NO_FLUSH;
98  do
99  {
100  toContinue = false;
101  flush = (stream.avail_in == 0)?Z_FINISH:Z_NO_FLUSH;
102 
103  if ( !stream.avail_out )
104  {
105  // increase the space
106  out_p = (unsigned char*)realloc(out_p,out_len + ZLIB_MEM_DELTA);
107 
108  // init new memory
109  unsigned char *new_out_start = out_p + out_len;
110  memset(new_out_start,0,ZLIB_MEM_DELTA);
111 
112  // Point next_out to the next unused byte
113  stream.next_out = new_out_start;
114 
115  // Update the size of the buffer
116  stream.avail_out = (uInt)ZLIB_MEM_DELTA;
117  out_len += ZLIB_MEM_DELTA;
118  }
119 
120  err = deflate(&stream,flush);
121  if ( err == Z_OK && stream.avail_out == 0 )
122  {
123  toContinue = true;
124  }
125 
126  }
127  while ( toContinue || flush == Z_NO_FLUSH );
128 
129  err = deflateEnd(&stream);
130  ZLIB_CHECK_ERR(err, "deflateEnd");
131  if( err != Z_OK )
132  {
133  free(out_p);
134  return false;
135  }
136 
137  decoded = std::string((char*)out_p,stream.total_out);
138  free(out_p);
139  return true;
140 }

146 bool FlateDecode::decode(std::string & encoded)
147 {
148  z_stream stream;
149 
150  //some initialization of ZLIB stuff
151  stream.zalloc = (alloc_func)0;
152  stream.zfree = (free_func)0;
153  stream.opaque = (voidpf)0;
154 
155  //trace_hex((char*)encoded.c_str(),encoded.size());
156 
157  stream.next_in = (unsigned char*)encoded.c_str();
158  stream.avail_in = (uInt)encoded.size();
159 
160  int err = inflateInit(&stream);
161  ZLIB_CHECK_ERR(err,"InflateInit");
162  if ( err != Z_OK )
163  {
164  return false;
165  }
166  unsigned char *out_p = NULL;
167  int out_len = 0;
168 
169  stream.next_out = out_p;
170  stream.avail_out = out_len;
171 
172  for (;;)
173  {
174  if ( !stream.avail_out)
175  {
176  // there is no more space for deallocation - increase the space
177  out_p = (unsigned char*)realloc(out_p,out_len + ZLIB_MEM_DELTA);
178 
179  // init new memory
180  unsigned char *new_out_start = out_p + out_len;
181  memset(new_out_start,0,ZLIB_MEM_DELTA);
182 
183  // Point next_out to the next unused byte
184  stream.next_out = new_out_start;
185 
186  // Update the size of the uncompressed buffer
187  stream.avail_out = (uInt)ZLIB_MEM_DELTA;
188  out_len += ZLIB_MEM_DELTA;
189  }
190  err = inflate(&stream,Z_NO_FLUSH);
191 
192  if ( err == Z_STREAM_END)
193  {
194  break;
195  }
196  ZLIB_CHECK_ERR(err,"Deflate");
197  if ( err != Z_OK )
198  {
199  if( out_p )
200  {
201  free(out_p);
202  }
203  return false;
204  }
205  }
206  err = inflateEnd(&stream);
207  ZLIB_CHECK_ERR(err,"InflateEnd");
208  if( err != Z_OK )
209  {
210  if( out_p )
211  {
212  free(out_p);
213  }
214  return false;
215  }
216  encoded = std::string((char*)out_p,stream.total_out);
217  free(out_p);
218  // trace_hex((char*)encoded.c_str(),encoded.size());
219  // if predictor exists for that object, then lets decode it
220  if( _predict )
221  {
222  _predict->decode(encoded);
223  }
224 
225  return true;
