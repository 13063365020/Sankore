88 bool UBFileSystemUtils::copyFile(const QString &source, const QString &destination, bool overwrite)
89 {
90  if (!QFile::exists(source)) {
91  qDebug() << "file" << source << "does not present in fs";
92  return false;
93  }
94 
95  QString normalizedDestination = destination;
96  if (QFile::exists(normalizedDestination)) {
97  if (QFileInfo(normalizedDestination).isFile() && overwrite) {
98  QFile::remove(normalizedDestination);
99  }
100  } else {
101  normalizedDestination = normalizedDestination.replace(QString("\\"), QString("/"));
102  int pos = normalizedDestination.lastIndexOf("/");
103  if (pos != -1) {
104  QString newpath = normalizedDestination.left(pos);
105  if (!QDir().mkpath(newpath)) {
106  qDebug() << "can't create a new path at " << newpath;
107  }
108  }
109  }
110  return QFile::copy(source, normalizedDestination);
111 }

275 bool UBFileSystemUtils::copyDir(const QString& pSourceDirPath, const QString& pTargetDirPath)
276 {
277  if (pSourceDirPath == "" || pSourceDirPath == "." || pSourceDirPath == "..")
278  return false;
279 
280  QDir dirSource(pSourceDirPath);
281  QDir dirTarget(pTargetDirPath);
282 
283  if (!dirTarget.mkpath(pTargetDirPath))
284  return false;
285 
286  bool successSoFar = true;
287 
288  foreach(QFileInfo dirContent, dirSource.entryInfoList(QDir::Files | QDir::Dirs
289  | QDir::NoDotAndDotDot | QDir::Hidden , QDir::Name))
290  {
291  if (successSoFar)
292  {
293  if (dirContent.isDir())
294  {
295  successSoFar = copyDir(pSourceDirPath + "/" + dirContent.fileName(), pTargetDirPath + "/" + dirContent.fileName());
296  }
297  else
298  {
299  QFile f(pSourceDirPath + "/" + dirContent.fileName());
300  successSoFar = f.copy(pTargetDirPath + "/" + dirContent.fileName());
301  }
302  }
303  else
304  {
305  break;
306  }
307  }
308 
309  return successSoFar;
310 }

573  type = UBMimeType::VectorImage;

577  type = UBMimeType::AppleWidget;
578  }
579  else if (typeString == "application/widget")

582  }
583  else if (typeString.startsWith("video/"))
584  {
585  type = UBMimeType::Video;
586  }
587  else if (typeString.startsWith("audio/"))
588  {

592  {
593  type = UBMimeType::Flash;

596  {
597  type = UBMimeType::PDF;
598  }

600  {

602  }
603  else if (typeString.startsWith("internal/bookmark"))

739 bool UBFileSystemUtils::expandZipToDir(const QFile& pZipFile, const QDir& pTargetDir)
740 {
741  QuaZip zip(pZipFile.fileName());
742 
743  if(!zip.open(QuaZip::mdUnzip))
744  {
745  qWarning() << "ZIP expand failed. Cause zip.open(): " << zip.getZipError();
746  return false;
747  }
748 
749  zip.setFileNameCodec("UTF-8");
750  QuaZipFileInfo info;
751  QuaZipFile file(&zip);
752 
753  QString documentRootFolder = pTargetDir.absolutePath();
754 
755  if(!pTargetDir.exists())
756  pTargetDir.mkpath(documentRootFolder);
757 
758  QFile out;
759  char c;
760  for(bool more = zip.goToFirstFile(); more; more = zip.goToNextFile())
761  {
762  if(!zip.getCurrentFileInfo(&info))
763  {
764  //TOD UB 4.3 O display error to user or use crash reporter
765  qWarning() << "ZIP expand failed. Cause: getCurrentFileInfo(): " << zip.getZipError();
766  return false;
767  }
768 
769  if(!file.open(QIODevice::ReadOnly))
770  {
771  qWarning() << "ZIP expand failed. Cause: file.open(): " << zip.getZipError();
772  return false;
773  }
774 
775  if(file.getZipError()!= UNZ_OK)
776  {
777  qWarning() << "ZIP expand failed. Cause: file.getFileName(): " << zip.getZipError();
778  return false;
779  }
780 
781  QString newFileName = documentRootFolder + "/" + file.getActualFileName();
782  QFileInfo newFileInfo(newFileName);
783  QDir root(documentRootFolder);
784  root.mkpath(newFileInfo.absolutePath());
785 
786  out.setFileName(newFileName);
787  out.open(QIODevice::WriteOnly);
788 
789  // Slow like hell (on GNU/Linux at least), but it is not my fault.
790  // Not ZIP/UNZIP package's fault either.
791  // The slowest thing here is out.putChar(c).
792  QByteArray outFileContent = file.readAll();
793  if (out.write(outFileContent) == -1)
794  {
795  // qWarning() << "ZIP expand failed. Cause: Unable to write file";
796  // this may happen if we are decompressing a directory
797  }
798 
799  while(file.getChar(&c))
800  out.putChar(c);
801 
802  out.close();
803 
804  if(file.getZipError()!= UNZ_OK)
805  {
806  qWarning() << "ZIP expand failed. Cause: " << zip.getZipError();
807  return false;
808  }
809 
810  if(!file.atEnd())
811  {
812  qWarning() << "ZIP expand failed. Cause: read all but not EOF";
813  return false;
814  }
815 
816  file.close();
817 
818  if(file.getZipError()!= UNZ_OK)
819  {
820  qWarning() << "ZIP expand failed. Cause: file.close(): " << file.getZipError();
821  return false;
822  }
823 
824  }
825 
826  zip.close();
827 
828  if(zip.getZipError()!= UNZ_OK)
829  {
830  qWarning() << "ZIP expand failed. Cause: zip.close(): " << zip.getZipError();
831  return false;
832  }
833 
834  return true;
