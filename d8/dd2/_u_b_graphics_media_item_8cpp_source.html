162 QVariant UBGraphicsMediaItem::itemChange(GraphicsItemChange change, const QVariant &value)
163 {
164  if ((change == QGraphicsItem::ItemEnabledChange)
165  || (change == QGraphicsItem::ItemSceneChange)
166  || (change == QGraphicsItem::ItemVisibleChange))
167  {
168  if (mMediaObject && (!isEnabled() || !isVisible() || !scene()))
169  {
170  mMediaObject->pause();
171  }
172  }
173  else if (change == QGraphicsItem::ItemSceneHasChanged)
174  {
175  if (!scene())
176  {
177  mMediaObject->stop();
178  }
179  else
180  {
181  QString absoluteMediaFilename;
182 
183  if(mMediaFileUrl.toLocalFile().startsWith("audios/") || mMediaFileUrl.toLocalFile().startsWith("videos/")){
184  absoluteMediaFilename = scene()->document()->persistencePath() + "/" + mMediaFileUrl.toLocalFile();
185  }
186  else{
187  absoluteMediaFilename = mMediaFileUrl.toLocalFile();
188  }
189 
190  if (absoluteMediaFilename.length() > 0)
191  mMediaObject->setCurrentSource(Phonon::MediaSource(absoluteMediaFilename));
192 
193  }
194  }
195 
196  return UBGraphicsProxyWidget::itemChange(change, value);
197 }

353 void UBGraphicsMediaItem::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
354 {
355  if(mShouldMove && (event->buttons() & Qt::LeftButton))
356  {
357  QPointF offset = event->scenePos() - mMousePressPos;
358 
359  if (offset.toPoint().manhattanLength() > QApplication::startDragDistance())
360  {
361  QPointF mouseMovePos = mapFromScene(mMouseMovePos);
362  QPointF eventPos = mapFromScene( event->scenePos());
363 
364  QPointF translation = eventPos - mouseMovePos;
365  translate(translation.x(), translation.y());
366  }
367 
368  mMouseMovePos = event->scenePos();
369  }
370 
371  event->accept();
372 
373 }
