224 void UBBoardController::setBoxing(QRect displayRect)
225 {
226  if (displayRect.isNull())
227  {
228  mControlLayout->setContentsMargins(0, 0, 0, 0);
229  return;
230  }
231 
232  qreal controlWidth = (qreal)mMainWindow->centralWidget()->width();
233  qreal controlHeight = (qreal)mMainWindow->centralWidget()->height();
234  qreal displayWidth = (qreal)displayRect.width();
235  qreal displayHeight = (qreal)displayRect.height();
236 
237  qreal displayRatio = displayWidth / displayHeight;
238  qreal controlRatio = controlWidth / controlHeight;
239 
240  if (displayRatio < controlRatio)
241  {
242  // Pillarboxing
243  int boxWidth = (controlWidth - (displayWidth * (controlHeight / displayHeight))) / 2;
244  mControlLayout->setContentsMargins(boxWidth, 0, boxWidth, 0);
245  }
246  else if (displayRatio > controlRatio)
247  {
248  // Letterboxing
249  int boxHeight = (controlHeight - (displayHeight * (controlWidth / displayWidth))) / 2;
250  mControlLayout->setContentsMargins(0, boxHeight, 0, boxHeight);
251  }
252  else
253  {
254  // No boxing
255  mControlLayout->setContentsMargins(0, 0, 0, 0);
256  }
257 }

557 UBGraphicsItem *UBBoardController::duplicateItem(UBItem *item, bool bAsync, eItemActionType actionType)
558 {
559  if (!item)
560  return NULL;
561 
562  UBGraphicsItem *retItem = NULL;
563 
564  mLastCreatedItem = NULL;
565 
566  QUrl sourceUrl;
567  QByteArray pData;
568 
569  //common parameters for any item
570  QPointF itemPos;
571  QSizeF itemSize;
572 
573  QGraphicsItem *commonItem = dynamic_cast<QGraphicsItem*>(item);
574  if (commonItem)
575  {
576  qreal shifting = UBSettings::settings()->objectFrameWidth;
577  itemPos = commonItem->pos() + QPointF(shifting,shifting);
578  itemSize = commonItem->boundingRect().size();
579  commonItem->setSelected(false);
580  }
581 
582  UBMimeType::Enum itemMimeType;
583 
584  QString srcFile = item->sourceUrl().toLocalFile();
585  if (srcFile.isEmpty())
586  srcFile = item->sourceUrl().toString();
587 
588  QString contentTypeHeader;
589  if (!srcFile.isEmpty())
590  contentTypeHeader = UBFileSystemUtils::mimeTypeFromFileName(srcFile);
591 
592  if(NULL != qgraphicsitem_cast<UBGraphicsGroupContainerItem*>(commonItem))
593  itemMimeType = UBMimeType::Group;
594  else
595  itemMimeType = UBFileSystemUtils::mimeTypeFromString(contentTypeHeader);
596 
597  switch(static_cast<int>(itemMimeType))
598  {
599  case UBMimeType::AppleWidget:
600  case UBMimeType::W3CWidget:
601  {
602  UBGraphicsWidgetItem *witem = dynamic_cast<UBGraphicsWidgetItem*>(item);
603  if (witem)
604  {
605  sourceUrl = witem->getOwnFolder();
606  }
607  }break;
608 
609  case UBMimeType::Video:
610  case UBMimeType::Audio:
611  {
612  UBGraphicsMediaItem *mitem = dynamic_cast<UBGraphicsMediaItem*>(item);
613  if (mitem)
614  {
615  sourceUrl = mitem->mediaFileUrl();
616  if (bAsync)
617  {
618  downloadURL(sourceUrl, srcFile, itemPos, QSize(itemSize.width(), itemSize.height()), false, false);
619  return NULL; // async operation
620  }
621  }
622  }break;
623 
624  case UBMimeType::VectorImage:
625  {
626  UBGraphicsSvgItem *viitem = dynamic_cast<UBGraphicsSvgItem*>(item);
627  if (viitem)
628  {
629  pData = viitem->fileData();
630  sourceUrl = item->sourceUrl();
631  }
632  }break;
633 
634  case UBMimeType::RasterImage:
635  {
636  UBGraphicsPixmapItem *pixitem = dynamic_cast<UBGraphicsPixmapItem*>(item);
637  if (pixitem)
638  {
639  QBuffer buffer(&pData);
640  buffer.open(QIODevice::WriteOnly);
641  QString format = UBFileSystemUtils::extension(item->sourceUrl().toLocalFile());
642  pixitem->pixmap().save(&buffer, format.toLatin1());
643  }
644  }break;
645 
646  case UBMimeType::Group:
647  {
648  UBGraphicsGroupContainerItem* groupItem = dynamic_cast<UBGraphicsGroupContainerItem*>(item);
649  UBGraphicsGroupContainerItem* duplicatedGroup = NULL;
650 
651  QList<QGraphicsItem*> duplicatedItems;
652  QList<QGraphicsItem*> children = groupItem->childItems();
653 
654  mActiveScene->setURStackEnable(false);
655  foreach(QGraphicsItem* pIt, children){
