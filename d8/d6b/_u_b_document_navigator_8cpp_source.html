88 void UBDocumentNavigator::generateThumbnails(UBDocumentContainer* source)
89 {
90  mThumbsWithLabels.clear();
91  foreach(QGraphicsItem* it, mScene->items())
92  {
93  mScene->removeItem(it);
94  delete it;
95  it = NULL;
96  }
97 
98  for(int i = 0; i < source->pageCount(); i++)
99  {
100  const QPixmap* pix = source->pageAt(i);
101  Q_ASSERT(!pix->isNull());
102  int pageIndex = UBDocumentContainer::pageFromSceneIndex(i);
103 
104  UBSceneThumbnailNavigPixmap* pixmapItem = new UBSceneThumbnailNavigPixmap(*pix, source->selectedDocument(), i);
105 
106  QString label = pageIndex == 0 ? tr("Title page") : tr("Page %0").arg(pageIndex);
107  UBThumbnailTextItem *labelItem = new UBThumbnailTextItem(label);
108 
109  pixmapItem->setLabel(labelItem);
110 
111  UBImgTextThumbnailElement thumbWithText(pixmapItem, labelItem);
112  thumbWithText.setBorder(border());
113  mThumbsWithLabels.append(thumbWithText);
114 
115  mScene->addItem(pixmapItem);
116  mScene->addItem(labelItem);
117  }
118 
119  // Draw the items
120  refreshScene();
121 }

168 void UBDocumentNavigator::refreshScene()
169 {
170  qreal thumbnailHeight = mThumbnailWidth / UBSettings::minScreenRatio;
171 
172  for(int i = 0; i < mThumbsWithLabels.size(); i++)
173  {
174  // Get the item
175  UBImgTextThumbnailElement& item = mThumbsWithLabels[i];
176  int columnIndex = i % mNbColumns;
177  int rowIndex = i / mNbColumns;
178  item.Place(rowIndex, columnIndex, mThumbnailWidth, thumbnailHeight);
179  item.getCaption()->highlight(false);
180  }
181  scene()->setSceneRect(scene()->itemsBoundingRect());
182 
183  if (mThumbsWithLabels.count() > UBApplication::boardController->activeSceneIndex())
184  {
185  mThumbsWithLabels.at(UBApplication::boardController->activeSceneIndex()).getThumbnail()->setSelected(true);
186  mThumbsWithLabels.at(UBApplication::boardController->activeSceneIndex()).getCaption()->highlight(true);
187  }
188 }

254 void UBDocumentNavigator::mousePressEvent(QMouseEvent *event)
255 {
256  QGraphicsItem* pClickedItem = itemAt(event->pos());
257  if(NULL != pClickedItem)
258  {
259 
260  // First, select the clicked item
261  UBSceneThumbnailNavigPixmap* pCrntItem = dynamic_cast<UBSceneThumbnailNavigPixmap*>(pClickedItem);
262 
263  if(NULL == pCrntItem)
264  {
265  // If we fall here we may have clicked on the label instead of the thumbnail
266  UBThumbnailTextItem* pTextItem = dynamic_cast<UBThumbnailTextItem*>(pClickedItem);
267  if(NULL != pTextItem)
268  {
269  for(int i = 0; i < mThumbsWithLabels.size(); i++)
270  {
271  const UBImgTextThumbnailElement& el = mThumbsWithLabels.at(i);
272  if(el.getCaption() == pTextItem)
273  {
274  pCrntItem = el.getThumbnail();
275  break;
276  }
277  }
278  }
279  }
280 
281  int index = 0;
282  for(int i = 0; i < mThumbsWithLabels.size(); i++)
283  {
284  if (mThumbsWithLabels.at(i).getThumbnail() == pCrntItem)
285  {
286  index = i;
287  break;
288  }
289  }
290  UBApplication::boardController->setActiveDocumentScene(index);
291  }
292  QGraphicsView::mousePressEvent(event);
293 }
