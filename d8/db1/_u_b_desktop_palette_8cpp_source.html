41 UBDesktopPalette::UBDesktopPalette(QWidget *parent, UBRightPalette* _rightPalette)

167 void UBDesktopPalette::maximizeMe()
168 {
169  QList<QAction*> actions;
170  clearLayout();
171 
172  actions << mActionUniboard;
173  actions << UBApplication::mainWindow->actionPen;
174  actions << UBApplication::mainWindow->actionEraser;
175  actions << UBApplication::mainWindow->actionMarker;
176  actions << UBApplication::mainWindow->actionSelector;
177  actions << UBApplication::mainWindow->actionPointer;
178  if (UBPlatformUtils::hasVirtualKeyboard())
179  actions << UBApplication::mainWindow->actionVirtualKeyboard;
180 
181  actions << mActionCustomSelect;
182  actions << mDisplaySelectAction;
183  actions << mShowHideAction;
184 
185  setActions(actions);
186 
187  adjustSizeAndPosition();
