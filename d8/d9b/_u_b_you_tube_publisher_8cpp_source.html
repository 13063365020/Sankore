121 void UBYouTubePublisher::postClientLoginResponse(bool success, const QByteArray& pPayload)
122 {
123  Q_UNUSED(success);
124  mAuthToken = "";
125 
126  QString auth = QString::fromUtf8(pPayload);
127 
128  if (auth.contains("Auth="))
129  {
130  QStringList lines = auth.split("\n");
131 
132  foreach(QString line, lines)
133  {
134  if(line.startsWith("Auth="))
135  {
136  mAuthToken = line.replace("Auth=", "");
137  break;
138  }
139  }
140  }
141 
142  mAuthRequest->deleteLater();
143  mAuthRequest = 0;
144 
145  if(mAuthToken.length() == 0)
146  {
147  UBApplication::showMessage(tr("YouTube authentication failed."));
148  // success = false;
149  deleteLater();
150  }
151  else
152  {
153  qDebug() << "Retreived youtube auth token" << mAuthToken;
154  postVideoUploadRequest();
155  }

159 void UBYouTubePublisher::postVideoUploadRequest()
160 {
161  /*
162  * POST /feeds/api/users/default/uploads HTTP/1.1
163  Host: gdata.youtube.com
164  Authorization: GoogleLogin auth=<authentication_token>
165  GData-Version: 2
166  X-GData-Client: <client_id>
167  X-GData-Key: key=<developer_key>
168  Slug: <video_filename>
169  Content-Type: multipart/related; boundary="<boundary_string>"
170  Content-Length: <content_length>
171  Connection: close
172 
173  --<boundary_string>
174  Content-Type: application/atom+xml; charset=UTF-8
175 
176  API_XML_request
177  --<boundary_string>
178  Content-Type: <video_content_type>
179  Content-Transfer-Encoding: binary
180 
181  <Binary File Data>
182  --<boundary_string>--
183  */
184 
185  QFile videoFile(mVideoFilePath);
186 
187  if(!videoFile.open(QIODevice::ReadOnly))
188  {
189  qWarning() << "Cannot open file" << mVideoFilePath << "for upload to youtube";
190  return;
191  }
192 
193  QUrl url("http://uploads.gdata.youtube.com/feeds/api/users/default/uploads");
194 
195  QString boundary = "---------------------------f93dcbA3";
196  QString contentType = QString("multipart/related; boundary=\"%1\"").arg(boundary);
197 
198  mUploadRequest = new UBServerXMLHttpRequest(UBNetworkAccessManager::defaultAccessManager()
199  , contentType); // destroyed in postVideoUploadResponse
200 
201  mUploadRequest->setVerbose(true);
202  connect(mUploadRequest, SIGNAL(progress(qint64, qint64)), this, SLOT(progress(qint64, qint64)));
203 
204  connect(mUploadRequest, SIGNAL(finished(bool, const QByteArray&)), this, SLOT(postVideoUploadResponse(bool, const QByteArray&)));
205 
206  mUploadRequest->addHeader("X-GData-Key", "key=" + sYouTubeDeveloperKey);
207  mUploadRequest->addHeader("Authorization", QString("GoogleLogin auth=%1").arg(mAuthToken));
208  mUploadRequest->addHeader("GData-Version", "2");
209 
210  QFileInfo fi(mVideoFilePath);
211  mUploadRequest->addHeader("Slug", fi.fileName());
212  mUploadRequest->addHeader("Connection", "close"); // do we really ned that ?
213 
214  QByteArray payload;
215 
216  payload.append(QString("\n--" + boundary + "\n").toUtf8())
217  .append(QString("Content-Type: application/atom+xml; charset=UTF-8\n\n").toUtf8())
218  .append(youtubeMetadata().toUtf8());
219 
220  payload.append(QString("\n--" + boundary + "\n").toUtf8());
221 
222  QString videoMimeType = UBFileSystemUtils::mimeTypeFromFileName(mVideoFilePath);
223 
224  payload.append((QString("Content-Type: %1\n").arg(videoMimeType)).toUtf8())
225  .append(QString("Content-Transfer-Encoding: binary\n\n").toUtf8());
226 
227  payload.append(videoFile.readAll());
228 
229  payload.append(QString("\n--" + boundary + "--\n").toUtf8());
230 
231  mUploadRequest->post(url, payload);
232 }

254 QString UBYouTubePublisher::youtubeMetadata()
255 {
256  QString workingTitle;
257 
258  if (mTitle.length() > 0)
259  {
260  workingTitle = mTitle;
261  }
262  else
263  {
264  workingTitle = QFileInfo(mVideoFilePath).completeBaseName();
265  }
266 
267  QString metadata;
268 
269  metadata += "<?xml version=\"1.0\"?>\n";
270  metadata += "<entry xmlns=\"http://www.w3.org/2005/Atom\" xmlns:media=\"http://search.yahoo.com/mrss/\" xmlns:yt=\"http://gdata.youtube.com/schemas/2007\">\n";
271  metadata += " <media:group>\n";
272  metadata += QString(" <media:title type=\"plain\">%1</media:title>\n").arg(workingTitle);
273 
274  QString workingDescription = mDescription;
275 
276  if (workingDescription.length() > 4900)
277  {
278  workingDescription = workingDescription.left(4900) + "...";
279  }
280 
281  workingDescription += "\n\nhttp://www.open-sankore.org";
282 
283  if(workingDescription.length() == 0)
284  {
285  workingDescription = workingTitle;
286  }
287 
288  metadata += QString(" <media:description type=\"plain\">%1</media:description>\n").arg(workingDescription);
289 
290  foreach(QString cat, mCategories)
291  {
292  metadata += QString(" <media:category scheme=\"http://gdata.youtube.com/schemas/2007/categories.cat\">%1</media:category>\n").arg(cat);
293  }
294 
295  if (mKeywords.length() > 0)
296  {
297  metadata += QString(" <media:keywords>%1</media:keywords>\n").arg(mKeywords);
298  }
299 
300  metadata += " </media:group>\n";
301  metadata += "</entry>";
302 
303  return metadata;
304 
305 }
