90 static bool cp_rf(const QString &what, const QString &where)
91 {
92  QFileInfo whatFi(what);
93  QFileInfo whereFi = QFileInfo(where);
94 
95  if (!whatFi.exists()) {
96  qDebug() << what << "does not exist" << Q_FUNC_INFO;
97  return false;
98  } else if (whatFi.isFile()) {
99  QString whereDir = where.section(QDir::separator(), 0, -2, QString::SectionSkipEmpty | QString::SectionIncludeLeadingSep);
100  QString newFilePath = where;
101  if (!whereFi.exists()) {
102  QDir().mkpath(whereDir);
103  } else if (whereFi.isDir()) {
104  newFilePath = whereDir + "/" + whatFi.fileName();
105  }
106  if (QFile::exists(newFilePath)) {
107  QFile::remove(newFilePath);
108  }
109  if (!QFile::copy(what, newFilePath)) {
110  qDebug() << "can't copy" << what << "to" << where << Q_FUNC_INFO;
111  return false;
112  }
113  return true;
114  } else if (whatFi.isDir()) {
115 
116  if (whereFi.isFile() && whereFi.fileName().toLower() == whatFi.fileName().toLower()) {
117  qDebug() << "can't copy dir" << what << "to file" << where << Q_FUNC_INFO;
118  return false;
119  } else if (whereFi.isDir()) {
120  rm_r(where);
121  }
122 
123  QDir().mkpath(where);
124 
125  QFileInfoList fList = QDir(what).entryInfoList(QDir::AllEntries | QDir::NoDotAndDotDot);
126  foreach (QFileInfo sub, fList) {
127  if (!cp_rf(sub.absoluteFilePath(), where + "/" + sub.fileName()))
128  return false;
129  }
130  return true;
131  }
132  return true;
133 }

398  void cureFromText(const QString &tagName, QDomElement element)
399  {
400  if (tagName == tVideo
401  || tagName == tAudio
402  || tagName == tImage) {
403  QString newRelative = cureNCopy(element.attribute(aHref));
404  element.setAttribute(aHref, newRelative);
405  if (element.hasAttribute(aActionMedia)) {
406  QString newActionPath = cureNCopy(element.attribute(aActionMedia));
407  element.setAttribute(aActionMedia, newActionPath);
408  }
409  } else if (tagName == tForeignObject) {
410  QString type = element.attribute(aType);
411  if (type == vText) { // We don't have to care of the text object
412  if (element.hasAttribute(aActionMedia)) {
413  QString newRelative = cureNCopy(element.attribute(aActionMedia));
414  element.setAttribute(aActionMedia, newRelative);
415  }
416  return;
417  }
418  QString newRelative = cureNCopy(element.attribute(aSrc));
419  element.setAttribute(aSrc, newRelative);
420  } else if (tagName == tTeacherGuide) {
421  QDomElement nMediaElement = element.firstChildElement(tMedia);
422  while (!nMediaElement.isNull()) {
423  QString newRelative = cureNCopy(nMediaElement.attribute(aRelativePath));
424  nMediaElement.setAttribute(aRelativePath, newRelative);
425  nMediaElement = nMediaElement.nextSiblingElement(tMedia);
426  }
