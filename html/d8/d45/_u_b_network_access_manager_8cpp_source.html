155 void UBNetworkAccessManager::sslErrors(QNetworkReply *reply, const QList<QSslError> &error)
156 {
157  // check if SSL certificate has been trusted already
158  QString replyHost = reply->url().host() + ":" + reply->url().port();
159  if(!sslTrustedHostList.contains(replyHost))
160  {
161  QWidget *mainWindow = QApplication::activeWindow();
162 
163  QStringList errorStrings;
164  for (int i = 0; i < error.count(); ++i)
165  errorStrings += error.at(i).errorString();
166 
167  QString errors = errorStrings.join(QLatin1String("\n"));
168 
169  QMessageBox messageBox;
170  messageBox.setParent(mainWindow);
171  messageBox.setWindowFlags(Qt::Dialog);
172  messageBox.setWindowTitle(QCoreApplication::applicationName());
173  messageBox.setText(tr("SSL Errors:\n\n%1\n\n%2\n\n"
174  "Do you want to ignore these errors for this host?").arg(reply->url().toString()).arg(errors));
175  QPushButton* yesButton = messageBox.addButton(tr("Yes"),QMessageBox::YesRole);
176  messageBox.addButton(tr("No"),QMessageBox::NoRole);
177  messageBox.setIcon(QMessageBox::Question);
178  messageBox.exec();
179 
180  if(messageBox.clickedButton() == yesButton) {
181  reply->ignoreSslErrors();
182  sslTrustedHostList.append(replyHost);
183  }
184  }
