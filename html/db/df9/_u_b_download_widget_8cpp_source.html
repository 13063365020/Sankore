164 void UBDownloadWidget::onDownloadUpdated(int id, qint64 crnt, qint64 total)
165 {
166  if(NULL != mpTree)
167  {
168  QAbstractItemModel* model = mpTree->model();
169  if(NULL != model)
170  {
171  for(int i=0; i< model->rowCount(); i++)
172  {
173  QModelIndex currentIndex = model->index(i, eItemColumn_Desc);
174  if(id == currentIndex.data(Qt::UserRole + 2))
175  {
176  // We found the right item, now we update the progress bar
177  model->setData(currentIndex, crnt, Qt::UserRole);
178  model->setData(currentIndex, total, Qt::UserRole + 1);
179  break;
180  }
181  }
182  }
183  }
184 }
