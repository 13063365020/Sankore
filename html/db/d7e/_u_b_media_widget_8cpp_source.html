153 void UBMediaWidget::createMediaPlayer()
154 {
155  mpMediaContainer = new QWidget();
156  mpMediaContainer->setObjectName("UBMediaVideoContainer");
157  mMediaLayout = new QHBoxLayout();
158  mpMediaContainer->setLayout(mMediaLayout);
159 
160  if(eMediaType_Video == mType){

162  if(isVisible()){
163  mpVideoWidget = new Phonon::VideoWidget(this);
164  mMediaLayout->addStretch(1);
165  mMediaLayout->addWidget(mpVideoWidget);
166  mMediaLayout->addStretch(1);
167  Phonon::createPath(mpMediaObject, mpVideoWidget);
168  adaptSizeToVideo();
169  }
170  mpAudioOutput = new Phonon::AudioOutput(Phonon::VideoCategory, this);
171  Phonon::createPath(mpMediaObject, mpAudioOutput);
172  }else if(eMediaType_Audio == mType){
173  mMediaLayout->setContentsMargins(10, 10, 10, 10);
174  mpCover = new QLabel(mpMediaContainer);
175  //mpMediaContainer->setStyleSheet(QString("background: none;"));
176  setAudioCover(":images/libpalette/soundIcon.svg");
177  mpCover->setScaledContents(true);
178  mMediaLayout->addStretch(1);
179  mMediaLayout->addWidget(mpCover);
180  mMediaLayout->addStretch(1);
181  mpAudioOutput = new Phonon::AudioOutput(Phonon::MusicCategory, this);
182  Phonon::createPath(mpMediaObject, mpAudioOutput);
183  }
184  mpLayout->addWidget(mpMediaContainer, 1);
185  mpLayout->addLayout(mpSeekerLayout, 0);
186  setActionsParent(mpMediaContainer);

192 void UBMediaWidget::adaptSizeToVideo()
193 {
194  if(NULL != mpMediaContainer){
195  int origW = mpMediaContainer->width();
196  int origH = mpMediaContainer->height();
197  int newW = width();
198  float scaleFactor = (float)origW/(float)newW;
199  int newH = origH/scaleFactor;
200  resize(newW, height() + newH);
201  }
202 }

209 void UBMediaWidget::onStateChanged(Phonon::State newState, Phonon::State oldState)
210 {
211  if(!mGeneratingThumbnail){
212  if(Phonon::LoadingState == oldState && Phonon::StoppedState == newState){
213  if(eMediaType_Video == mType){

216  mpMediaObject->play();
217  mpMediaObject->pause();

219  }
220  }else if(Phonon::PlayingState == oldState && Phonon::PausedState == newState){
221  mpPlayStopButton->setPixmap(QPixmap(":images/play.svg"));
222  mpPauseButton->setEnabled(false);
223  }else if((Phonon::PausedState == oldState && Phonon::PlayingState == newState) ||
224  (Phonon::StoppedState == oldState && Phonon::PlayingState == newState)){
225  mpPlayStopButton->setPixmap(QPixmap(":images/stop.svg"));
226  mpPauseButton->setEnabled(true);
227  }else if(Phonon::PlayingState == oldState && Phonon::StoppedState == newState){
228  mpPlayStopButton->setPixmap(QPixmap(":images/play.svg"));
229  mpPauseButton->setEnabled(false);
230  mpSlider->setValue(0);

232 
233  }
