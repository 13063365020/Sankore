253 void UBFeaturesActionBar::dropEvent(QDropEvent *event)
254 {
255  const UBFeaturesMimeData *fMimeData = qobject_cast<const UBFeaturesMimeData*>(event->mimeData());
256 
257  if (!fMimeData) {
258  qWarning() << "data came from not supported widget";
259  event->ignore();
260  return;
261  }
262 
263  QWidget *dest = childAt(event->pos());
264  if (dest == mpDeleteBtn) {
265  QList<UBFeature> featuresList = fMimeData->features();
266  foreach (UBFeature curFeature, featuresList) {
267  if (!curFeature.isDeletable()) {
268  qWarning() << "Undeletable feature found, stopping deleting process";
269  event->ignore();
270  return;
271  }
272  }
273 
274  } else if (dest == mpFavoriteBtn) {
275  event->setDropAction( Qt::CopyAction);
276  event->accept();
277 
278  emit addToFavorite(fMimeData);
279 
280  } else if (dest == mpRemoveFavoriteBtn) {
281  event->setDropAction(Qt::MoveAction);
282  event->accept();
283 
284  emit removeFromFavorite(fMimeData);
285  }
286 }

288 bool UBFeaturesActionBar::eventFilter(QObject *obj, QEvent *event)
289 {
290  if (event->type() == QEvent::DragEnter) {
291  QDragEnterEvent *enterEvent = static_cast<QDragEnterEvent*>(event);
292  const UBFeaturesMimeData *fMimeData = qobject_cast<const UBFeaturesMimeData*>(enterEvent->mimeData());
293  if (fMimeData) {
294 
295  if (obj == mpDeleteBtn) {
296  foreach (UBFeature curFeature, fMimeData->features()) {
297  if (!curFeature.isDeletable()) {
298  event->ignore();
299  return true;
300  }
301  }
302  event->accept();
303 
304  } else if (obj == mpFavoriteBtn) {
305  foreach (UBFeature curFeature, fMimeData->features()) {
306  if (curFeature.getType() == FEATURE_FOLDER) {
307  event->ignore();
308  return true;
309  }
310  }
311  event->accept();
312 
313  } else if (obj == mpRemoveFavoriteBtn) {
314  event->accept();
315  return true;
316  }
317  }
318  } else if (event->type() == QEvent::Drop) {
319  QDropEvent *dropEvent = static_cast<QDropEvent*>(event);
320  const UBFeaturesMimeData *fMimeData = qobject_cast<const UBFeaturesMimeData*>(dropEvent->mimeData());
321  if (fMimeData) {
322 
323  if (obj == mpDeleteBtn) {
324  dropEvent->setDropAction(Qt::MoveAction);
325  event->accept();
326  emit deleteElements(fMimeData);
327 
328  } else if (obj == mpFavoriteBtn) {
329  dropEvent->setDropAction(Qt::CopyAction);
330  event->accept();
331  emit addToFavorite(fMimeData);
332 
333  } else if (obj == mpRemoveFavoriteBtn) {
334  dropEvent->setDropAction(Qt::MoveAction);
335  event->accept();
336 
337  emit removeFromFavorite(fMimeData);
338  }
339  }
340  }
341  return QWidget::eventFilter(obj, event);
342 }
