74 /*
75  ToolbarSearch is a very basic search widget that also contains a small history.
76  Searches are turned into urls that use Google to perform search
77  */
78 WBToolbarSearch::WBToolbarSearch(QWidget *parent)
79  : WBSearchLineEdit(parent)
80  , mAutosaver(new UBAutoSaver(this))
81  , mMaxSavedSearches(10)
82  , mStringListModel(new QStringListModel(this))
83 {
84  QMenu *m = menu();
85  connect(m, SIGNAL(aboutToShow()), this, SLOT(aboutToShowMenu()));
86  connect(m, SIGNAL(triggered(QAction*)), this, SLOT(triggeredMenuAction(QAction*)));
87 
88  QCompleter *completer = new QCompleter(mStringListModel, this);
89  completer->setCompletionMode(QCompleter::InlineCompletion);
90  lineEdit()->setCompleter(completer);
91 
92  connect(lineEdit(), SIGNAL(returnPressed()), SLOT(searchNow()));
93  setInactiveText(tr("Search"));
94  load();
95 }

111 void WBToolbarSearch::load()
112 {
113  QSettings settings;
114  settings.beginGroup(QLatin1String("toolbarsearch"));
115  QStringList list = settings.value(QLatin1String("recentSearches")).toStringList();
116  mMaxSavedSearches = settings.value(QLatin1String("maximumSaved"), mMaxSavedSearches).toInt();
117  mStringListModel->setStringList(list);
118  settings.endGroup();
119 }

121 void WBToolbarSearch::searchNow()
122 {
123  QString searchText = lineEdit()->text();
124  QStringList newList = mStringListModel->stringList();
125 
126  if (newList.contains(searchText))
127  newList.removeAt(newList.indexOf(searchText));
128  newList.prepend(searchText);
129 
130  if (newList.size() >= mMaxSavedSearches)
131  newList.removeLast();
132 
133  QWebSettings *globalSettings = QWebSettings::globalSettings();
134  if (!globalSettings->testAttribute(QWebSettings::PrivateBrowsingEnabled))
135  {
136  mStringListModel->setStringList(newList);
137  mAutosaver->changeOccurred();
138  }
139 
140  QUrl url(QLatin1String("http://www.google.com/search"));
141  url.addQueryItem(QLatin1String("q"), searchText);
142  url.addQueryItem(QLatin1String("ie"), QLatin1String("UTF-8"));
143  url.addQueryItem(QLatin1String("oe"), QLatin1String("UTF-8"));
144  url.addQueryItem(QLatin1String("client"), QLatin1String("uniboard-browser"));
145  emit search(url);
146 }

148 void WBToolbarSearch::aboutToShowMenu()
149 {
150  lineEdit()->selectAll();
151  QMenu *m = menu();
152  m->clear();
153  QStringList list = mStringListModel->stringList();
154  if (list.isEmpty())
155  {
156  m->addAction(tr("No Recent Searches"));
157  return;
158  }
159 
160  QAction *recent = m->addAction(tr("Recent Searches"));
161  recent->setEnabled(false);
162  for (int i = 0; i < list.count(); ++i)
163  {
164  QString text = list.at(i);
165  m->addAction(text)->setData(text);
166  }
167  m->addSeparator();
168  m->addAction(tr("Clear Recent Searches"), this, SLOT(clear()));
169 }
