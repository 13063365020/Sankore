62 void x11SendKey(Display *display, int keyCode, int modifiers)
63 {
64 
65 // Get the root window for the current display.
66  Window winRoot = XDefaultRootWindow(display);
67  if (winRoot==0)
68  return;
69 
70 // Find the window which has the current keyboard focus.
71  Window winFocus;
72  int revert;
73  XGetInputFocus(display, &winFocus, &revert);
74 
75 // Send a fake key press event to the window.
76  XKeyEvent event = createKeyEvent(display, winFocus, winRoot, true, keyCode, modifiers);
77  XSendEvent(event.display, event.window, True, KeyPressMask, (XEvent *)&event);
78 
79 // Send a fake key release event to the window.
80  event = createKeyEvent(display, winFocus, winRoot, false, keyCode, modifiers);
81  XSendEvent(event.display, event.window, True, KeyPressMask, (XEvent *)&event);

179 void UBKeyboardPalette::onActivated(bool activated)
180 {
181  if (activated)
182  {
183  if (storage)
184  {
185  qDebug() << "Keybard already activated....";
186  return;
187  }
188 
189  Display *display = XOpenDisplay(0);
190  if(display == NULL)
191  return;
192 
193  XDisplayKeycodes(display, &this->min_keycodes, &this->max_keycodes);
194  KeySym* keySyms = XGetKeyboardMapping(display, min_keycodes,
195  max_keycodes - min_keycodes, &byte_per_code);
196 
197  storage = keySyms;
198 
199  XCloseDisplay(display);
200 
201  onLocaleChanged(locales[nCurrentLocale]);
202 
203  }
204  else
205  {
206  Display *display = XOpenDisplay(0);
207  if(display == NULL)
208  {
209  qDebug() << "Keybard not activated....";
210  return;
211  }
212 
213  KeySym* keySyms = (KeySym*)storage;
214  if (keySyms!=NULL)
215  {
216  qDebug() << "Default key table restored.....";
217 
218  XChangeKeyboardMapping(display, min_keycodes, byte_per_code,
219  keySyms, max_keycodes - min_keycodes);
220 
221  XFree(keySyms);
222  storage = NULL;
223  }
224 
225  XCloseDisplay(display);
226  }

229 void UBKeyboardPalette::onLocaleChanged(UBKeyboardLocale* locale)
230 {
231  const int maxMapOffset = 3; //Suppose to have at least 2 keysym groups due to X11 xlib specification
232 
233  Display *display = XOpenDisplay(0);
234  if(display == NULL)
235  return;
236 
237  int byte_per_code;
238  KeySym* keySyms = XGetKeyboardMapping(display, min_keycodes,
239  max_keycodes - min_keycodes, &byte_per_code);
240 
241  for(int i=0; i<SYMBOL_KEYS_COUNT; i++)
242  {
243  // loop by keybt
244  for(int j=0; j<8; j++)
245  {
246  KEYCODE& kc = (*locale)[i]->codes[j];
247  if (!kc.empty())
248  {
249  if (kc.modifier <= maxMapOffset)
250  keySyms[kc.code * byte_per_code + kc.modifier] = kc.symbol;
251 
252  }
253  }
254  }
255 
256  //Now look for modifiers > 5 and reassign them to free places
257  for(int i=0; i<SYMBOL_KEYS_COUNT; i++)
258  {
259  // loop by keybt
260  for(int j=0; j<8; j++)
261  {
262  KEYCODE& kc = (*locale)[i]->codes[j];
263  if (!kc.empty())
264  {
265  if (kc.modifier > maxMapOffset)
266  {
267  for(int i1=0; i1<SYMBOL_KEYS_COUNT; i1++)
268  for(int j1=0; j1<=maxMapOffset; j1++)
269  if (keySyms[i1 * byte_per_code + j1]==NoSymbol)
270  {
271  kc.code =i1;
272  kc.modifier =j1;
273  break;
274  }
275 
276  }
277  keySyms[kc.code * byte_per_code + kc.modifier] = kc.symbol;
278  }
279  }
280  }
281 
282 
283  XChangeKeyboardMapping(display, min_keycodes, byte_per_code, keySyms, max_keycodes - min_keycodes);
284  XFree(keySyms);
285 
286  XCloseDisplay(display);
287 }
