106 /* calculatePoints() is used to calculate polygon's apexes coordinates.
107  * This function handles orientation changes too.
108  * Items are repositionated and path is redeterminates.
109  */
110 void UBGraphicsAristo::calculatePoints(QRectF bounds)
111 {
112  switch (mOrientation) {
113  case Bottom:
114  C.setX(bounds.center().x());
115  C.setY(bounds.bottom());
116 
117  A.setX(bounds.left());
118  A.setY(bounds.bottom() - bounds.width() / 2);
119 
120  B.setX(bounds.right());
121  B.setY(bounds.bottom() - bounds.width() / 2);
122  break;
123  case Top:
124  C.setX(bounds.center().x());
125  C.setY(bounds.top());
126 
127  A.setX(bounds.left());
128  A.setY(bounds.top() + bounds.width() / 2);
129 
130  B.setX(bounds.right());
131  B.setY(bounds.top() + bounds.width() / 2);
132  break;
133  default:
134  break;
135  }
136 
137  setItemsPos();
138  setPath(determinePath());
139 }

155 /*
156  * determinePath() modify the shape according to apexes coordinates and control item positions.
157  * This is useful when orientation is modified.
158  * Returns the painter path corresponding to object parameters.
159  */
160 QPainterPath UBGraphicsAristo::determinePath()
161 {
162  QPainterPath path;
163 
164  QPolygonF polygon;
165  polygon << A << B << C;
166 
167  qreal rotationAngle = mOrientation == Bottom ? mMarkerAngle : mMarkerAngle - 360 * (int)(mMarkerAngle / 360);
168  QTransform t;
169  t.rotate(rotationAngle);
170  bool markerIsInPolygon = true;
171  if ((mOrientation == Bottom && mMarkerAngle < 90) || (mOrientation == Top && mMarkerAngle < 270))
172  markerIsInPolygon = polygon.containsPoint(t.map(markerButtonRect().topLeft()) + rotationCenter(), Qt::OddEvenFill);
173  else
174  markerIsInPolygon = polygon.containsPoint(t.map(markerButtonRect().bottomLeft()) + rotationCenter(), Qt::OddEvenFill);
175 
176  path.moveTo(A);
177 
178  QRectF mappedRect = t.mapRect(markerButtonRect());
179  path.lineTo(QPointF(mappedRect.translated(rotationCenter()).left(), A.y()));
180  if (!markerIsInPolygon) {
181  if (mOrientation == Top) {
182  path.lineTo(QPointF(mappedRect.translated(rotationCenter()).left(), mappedRect.translated(rotationCenter()).bottom()));
183  path.lineTo(QPointF(mappedRect.translated(rotationCenter()).right(), mappedRect.translated(rotationCenter()).bottom()));
184  }
185  else if (mOrientation == Bottom) {
186  path.lineTo(QPointF(mappedRect.translated(rotationCenter()).left(), mappedRect.translated(rotationCenter()).top()));
187  path.lineTo(QPointF(mappedRect.translated(rotationCenter()).right(), mappedRect.translated(rotationCenter()).top()));
188  }
189  }
190  path.lineTo(QPointF(mappedRect.translated(rotationCenter()).right(), A.y()));
191 
192  path.lineTo(QPointF(resizeButtonRect().translated(rotationCenter()).left(), A.y()));
193  if (mOrientation == Top) {
194  path.lineTo(QPointF(resizeButtonRect().translated(rotationCenter()).left(), resizeButtonRect().translated(rotationCenter()).bottom()));
195  path.lineTo(QPointF(resizeButtonRect().translated(rotationCenter()).right(), resizeButtonRect().translated(rotationCenter()).bottom()));
196  }
197  else if (mOrientation == Bottom) {
198  path.lineTo(QPointF(resizeButtonRect().translated(rotationCenter()).left(), resizeButtonRect().translated(rotationCenter()).top()));
199  path.lineTo(QPointF(resizeButtonRect().translated(rotationCenter()).right(), resizeButtonRect().translated(rotationCenter()).top()));
200  }
201  path.lineTo(QPointF(resizeButtonRect().translated(rotationCenter()).right(), A.y()));
202 
203  path.lineTo(B);
204  path.lineTo(C);
205  path.lineTo(A);
206 
207  return path;
208 }

266 void UBGraphicsAristo::paintRulerGraduations(QPainter *painter)
267 {
268  /* defining useful constants */
269  const int centimeterGraduationHeight = 15;
270  const int halfCentimeterGraduationHeight = 10;
271  const int millimeterGraduationHeight = 5;
272  const int millimetersPerCentimeter = 10;
273  const int millimetersPerHalfCentimeter = 5;
274 
275  painter->save();
276  painter->setFont(font());
277  QFontMetricsF fontMetrics(painter->font());
278 
279  /* Browsing milliters in half width of ruler side */
280  for (int millimeters = 0; millimeters < (boundingRect().width() / 2 - sLeftEdgeMargin - sRoundingRadius) / sPixelsPerMillimeter; millimeters++)
281  {
282  /* defining graduationHeight ; values are different to draw bigger lines if millimiter considered is a centimeter or a half centimeter */
283  int graduationHeight = (0 == millimeters % millimetersPerCentimeter) ?
284  centimeterGraduationHeight :
285  ((0 == millimeters % millimetersPerHalfCentimeter) ?
286  halfCentimeterGraduationHeight : millimeterGraduationHeight);
287 
288  /* correcting graduationHeight: draw the line in the other direction in case ruler is top-oriented, to stay inside the tool and inside the rect */
289  graduationHeight = mOrientation == Bottom ? graduationHeight : - graduationHeight;
290 
291  /* drawing graduation to the left and to the right of origin, which is the center of graduated side */
292  painter->drawLine(QLine(rotationCenter().x() + sPixelsPerMillimeter * millimeters, rotationCenter().y(), rotationCenter().x() + sPixelsPerMillimeter * millimeters, rotationCenter().y() + graduationHeight));
293  if (millimeters != 0)
294  painter->drawLine(QLine(rotationCenter().x() - sPixelsPerMillimeter * millimeters, rotationCenter().y(), rotationCenter().x() - sPixelsPerMillimeter * millimeters, rotationCenter().y() + graduationHeight));
295 
296  /* drawing associated value if considered graduation is a centimeter */
297  if (0 == millimeters % millimetersPerCentimeter)
298  {
299  /* defining graduation value */
300  QString text = QString("%1").arg((int)(millimeters / millimetersPerCentimeter));
301 
302  /* staying inside polygon */
303  if (rotationCenter().x() + sPixelsPerMillimeter * millimeters + fontMetrics.width(text) / 2 < boundingRect().right())
304  {
305  qreal textWidth = fontMetrics.width(text);
306  qreal textHeight = fontMetrics.tightBoundingRect(text).height() + 5;
307 
308  /* text y-coordinate is different according to tool's orientation */
309  qreal textY = mOrientation == Bottom ? A.y() + 5 + centimeterGraduationHeight : A.y() - 5 - centimeterGraduationHeight + graduationHeight;
310 
311  /* if text's rect is not out of polygon's bounds, drawing value below or above graduation */
312  QPointF intersectionPoint;
313 
314  bool paint = false;
315 
316  if (mOrientation == Bottom && QLineF(QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters - textWidth / 2, rotationCenter().y()), QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters - textWidth / 2, textY + textHeight)).intersect(QLineF(A, C), &intersectionPoint) != QLineF::BoundedIntersection && QLineF(QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters + textWidth / 2, rotationCenter().y()), QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters + textWidth / 2, textY + textHeight)).intersect(QLineF(A, C), &intersectionPoint) != QLineF::BoundedIntersection) {
317  paint = true;
318  }
319  else if (mOrientation == Top && QLineF(QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters - textWidth / 2, rotationCenter().y()), QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters - textWidth / 2, textY - textHeight)).intersect(QLineF(A, C), &intersectionPoint) != QLineF::BoundedIntersection && QLineF(QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters + textWidth / 2, rotationCenter().y()), QPointF(rotationCenter().x() - sPixelsPerMillimeter * millimeters + textWidth / 2, textY - textHeight)).intersect(QLineF(A, C), &intersectionPoint) != QLineF::BoundedIntersection) {
320  paint = true;
321  }
322 
323  if (paint) {
324  painter->drawText(
325  QRectF(rotationCenter().x() + sPixelsPerMillimeter * millimeters - textWidth / 2, textY, textWidth, textHeight),
326  Qt::AlignVCenter, text);
327  if (millimeters != 0)
328  painter->drawText(
329  QRectF(rotationCenter().x() - sPixelsPerMillimeter * millimeters - textWidth / 2, textY, textWidth, textHeight),
330  Qt::AlignVCenter, text);
331  }
332  }
333  }
334  }
335  painter->restore();
336 }

338 void UBGraphicsAristo::paintProtractorGraduations(QPainter* painter)
339 {
340  /* defining useful constants */
341  const int tenDegreeGraduationLength = 15;
342  const int fiveDegreeGraduationLength = 10;
343  const int oneDegreeGraduationLength = 5;
344 
345  painter->save();
346 
347  QFont font1 = painter->font();
348 #ifdef Q_WS_MAC
349  font1.setPointSizeF(font1.pointSizeF() - 3);
350 #endif
351  QFontMetricsF fm1(font1);
352 
353  //Font for internal arc
354  QFont font2 = painter->font();
355  font2.setPointSizeF(font1.pointSizeF()/1.5);
356  QFontMetricsF fm2(font2);
357 
358  /* defining virtual arc diameter */
359  qreal rad = radius();
360 
361  QPointF center = rotationCenter();
362 
363  /* browsing angles */
364  for (int angle = 1; angle < mSpan; angle++)
365  {
366  int graduationLength = (0 == angle % 10) ? tenDegreeGraduationLength : ((0 == angle % 5) ? fiveDegreeGraduationLength : oneDegreeGraduationLength);
367 
368  qreal co = cos(((qreal)angle + mStartAngle) * PI/180);
369  qreal si = sin(((qreal)angle + mStartAngle) * PI/180);
370 
371  /* inverse sinus according to the orientation, to draw graduations on the polygon */
372  si = mOrientation == Bottom ? -si : si;
373 
374  /* drawing the graduation around the virtual arc */
375  if (angle >= sArcAngleMargin && angle <= mSpan - sArcAngleMargin)
376  painter->drawLine(QLineF(QPointF(center.x()+ rad/2*co, center.y() - rad/2*si),
377  QPointF(center.x()+ (rad/2 + graduationLength)*co,
378  center.y() - (rad/2 + graduationLength)*si)));
379 
380 
381  QPointF intersectionPoint;
382  QLineF referenceLine;
383  if (angle < 90)
384  referenceLine.setP1(B);
385  else
386  referenceLine.setP1(A);
387  referenceLine.setP2(C);
388 
389  /* if angle is 10-multiple, drawing it's value, rotated to be easily red */
390  if (0 == angle % 10) {
391  QString grad = QString("%1").arg((int)(angle));
392  QString grad2 = QString("%1").arg((int)mSpan - angle);
393 
394  painter->setFont(font2);
395 
396  painter->save();
397  painter->translate(center.x() + (rad/2 + graduationLength*1.5)*co, center.y() - (rad/2 + graduationLength*1.5)*si);
398  int degrees = mOrientation == Bottom ? angle : -angle;
399  painter->rotate(-90 + degrees);
400  painter->drawText(- fm2.width(grad)/2, - fm2.height()/2, fm2.width(grad), fm2.height(), Qt::AlignCenter, grad);
401  painter->restore();
402 
403  painter->setFont(font1);
404 
405 
406  /* drawing the graduation near tool's side */
407  if (QLineF(QPointF(center.x()+ rad/2*co, center.y() - rad/2*si),
408  QPointF(center.x()+ (rad/2 + graduationLength)*co,
409  center.y() - (rad/2 + graduationLength)*si)).intersect(referenceLine, &intersectionPoint) == QLineF::UnboundedIntersection)
410 
411  painter->drawLine(QLineF(QPointF(center.x() + (rad/2 + graduationLength*1.5 + fm2.width(grad)/2)*co,
412  center.y() - (rad/2 + graduationLength*1.5 + fm2.height()/2)*si),
413  intersectionPoint));
414 
415  }
416 
417  /* drawing the graduation near tool's side */
418  else
419  if (QLineF(QPointF(center.x()+ rad/2*co, center.y() - rad/2*si),
420  QPointF(center.x()+ (rad/2 + graduationLength)*co,
421  center.y() - (rad/2 + graduationLength)*si)).intersect(referenceLine, &intersectionPoint) == QLineF::UnboundedIntersection)
422 
423  painter->drawLine(QLineF(QPointF(intersectionPoint.x() - (graduationLength*1.5)*co,
424  intersectionPoint.y() + (graduationLength*1.5)*si),
425  intersectionPoint));
426  }
427 
428  painter->restore();
429 }

431 /* paintMarker() adjust marker button according to the current angle, draw the line allowing user to set precisely the angle, and draw the current angle's value. */
432 void UBGraphicsAristo::paintMarker(QPainter *painter)
433 {
434  /* adjusting marker button */
435  mMarkerSvgItem->resetTransform();
436  mMarkerSvgItem->translate(-markerButtonRect().left(), -markerButtonRect().top());
437  mMarkerSvgItem->rotate(mMarkerAngle);
438  mMarkerSvgItem->translate(markerButtonRect().left(), markerButtonRect().top());
439 
440 
441  qreal co = cos((mMarkerAngle) * PI/180);
442  qreal si = sin((mMarkerAngle) * PI/180);
443 
444  /* Setting point composing the line (from point C) which intersects the line we want to draw. */
445  QPointF referencePoint;
446  if (mOrientation == Bottom) {
447  if ((int)mMarkerAngle % 360 < 90)
448  referencePoint = B;
449  else
450  referencePoint = A;
451  }
452  else if (mOrientation == Top) {
453  if ((int)mMarkerAngle % 360 < 270 && (int)mMarkerAngle % 360 > 0)
454  referencePoint = A;
455  else
456  referencePoint = B;
457  }
458 
459  /* getting intersection point to draw the wanted line */
460  QLineF intersectedLine(rotationCenter(), QPointF(rotationCenter().x()+co, rotationCenter().y()+si));
461  QPointF intersectionPoint;
462  if (intersectedLine.intersect(QLineF(referencePoint, C), &intersectionPoint))
463  painter->drawLine(QLineF(intersectionPoint, rotationCenter()));
464 
465  /* drawing angle value */
466  qreal rightAngle = mOrientation == Bottom ? mMarkerAngle : 360 - mMarkerAngle;
467 
468 
469  QString angleText = QString("%1°").arg(rightAngle, 0, 'f', 1);
470 
471  QFont font1 = painter->font();
472 #ifdef Q_WS_MAC
473  font1.setPointSizeF(font1.pointSizeF() - 3);
474 #endif
475  QFontMetricsF fm1(font1);
476 
477  if (mOrientation == Bottom)
478  painter->drawText(rotationCenter().x() - fm1.width(angleText)/2 - radius()/8, rotationCenter().y() + radius()/8 - fm1.height()/2, fm1.width(angleText), fm1.height(), Qt::AlignCenter, angleText);
479  else
480  painter->drawText(rotationCenter().x() - fm1.width(angleText)/2 - radius()/8, rotationCenter().y() - radius()/8 - fm1.height()/2, fm1.width(angleText), fm1.height(), Qt::AlignCenter, angleText);
481 }

777 /*
778  * toolfromPos() returns the item type corresponding to the given position.
779  * This method is used to reduce the amount of code in each event function and improve class' maintainability.
780  * pos: event's position ; a rotation is done to counter elements rotation, like the marker button.
781  */
782 UBGraphicsAristo::Tool UBGraphicsAristo::toolFromPos(QPointF pos)
783 {
784  pos = pos - rotationCenter();
785 
786  qreal rotationAngle = mOrientation == Bottom ? - mMarkerAngle : Top ? 360 * (int)(mMarkerAngle / 360 + 1) - mMarkerAngle : 0;
787 
788  QTransform t;
789  t.rotate(rotationAngle);
790  QPointF p2 = t.map(pos);
791 
792  if (resizeButtonRect().contains(pos))
793  return Resize;
794  else if (closeButtonRect().contains(pos))
795  return Close;
796  else if (rotateRect().contains(pos))
797  return Rotate;
798  else if (markerButtonRect().contains(p2))
799  return MoveMarker;
800  else if (hFlipRect().contains(pos))
801  return HorizontalFlip;
802  else if (shape().contains(pos))
803  return Move;
804  else
805  return None;
806 }
