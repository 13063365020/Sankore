172 QString WBDownloadItem::saveFileName(const QString &directory) const
173 {
174  // Move this function into QNetworkReply to also get file name sent from the server
175  QString path = m_url.path();
176  QFileInfo info(path);
177  QString baseName = info.completeBaseName();
178  QString endName = info.suffix();
179 
180  if (baseName.isEmpty())
181  {
182  baseName = QLatin1String("unnamed_download");
183  qDebug() << "DownloadManager:: downloading unknown file:" << m_url;
184  }
185  QString name = directory + baseName + QLatin1Char('.') + endName;
186  if (QFile::exists(name))
187  {
188  // already exists, don't overwrite
189  int i = 1;
190  do {
191  name = directory + baseName + QLatin1Char('-') + QString::number(i++) + QLatin1Char('.') + endName;
192  } while (QFile::exists(name));
193  }
194  return name;
195 }

303 void WBDownloadItem::updateInfoLabel()
304 {
305  if (m_reply->error() == QNetworkReply::NoError)
306  return;
307 
308  qint64 bytesTotal = progressBar->maximum();
309  bool running = !downloadedSuccessfully();
310 
311  // update info label
312  double speed = mBytesReceived * 1000.0 / mDownloadTime.elapsed();
313  double timeRemaining = ((double)(bytesTotal - mBytesReceived)) / speed;
314  QString timeRemainingString = tr("seconds");
315  if (timeRemaining > 60)
316  {
317  timeRemaining = timeRemaining / 60;
318  timeRemainingString = tr("minutes");
319  }
320  timeRemaining = floor(timeRemaining);
321 
322  // When downloading the eta should never be 0
323  if (timeRemaining == 0)
324  timeRemaining = 1;
325 
326  QString info;
327  if (running)
328  {
329  QString remaining;
330  if (bytesTotal != 0)
331  remaining = tr("- %4 %5 remaining")
332  .arg(timeRemaining)
333  .arg(timeRemainingString);
334  info = QString(tr("%1 of %2 (%3/sec) %4"))
335  .arg(dataString(mBytesReceived))
336  .arg(bytesTotal == 0 ? tr("?", "unknown file size") : dataString(bytesTotal))
337  .arg(dataString((int)speed))
338  .arg(remaining);
339  }
340  else
341  {
342  if (mBytesReceived == bytesTotal)
343  info = dataString(m_output.size());
344  else
345  info = tr("%1 of %2 - Stopped")
346  .arg(dataString(mBytesReceived))
347  .arg(dataString(bytesTotal));
348  }
349  downloadInfoLabel->setText(info);
350 }

565 void WBDownloadManager::save() const
566 {
567  QSettings settings;
568  settings.beginGroup(QLatin1String("downloadmanager"));
569  QMetaEnum removePolicyEnum = staticMetaObject.enumerator(staticMetaObject.indexOfEnumerator("RemovePolicy"));
570  settings.setValue(QLatin1String("removeDownloadsPolicy"), QLatin1String(removePolicyEnum.valueToKey(m_RemovePolicy)));
571 
572  if (m_RemovePolicy == Exit)
573  return;
574 
575  for (int i = 0; i < mDownloads.count(); ++i)
576  {
577  QString key = QString(QLatin1String("download_%1_")).arg(i);
578  settings.setValue(key + QLatin1String("url"), mDownloads[i]->m_url);
579  settings.setValue(key + QLatin1String("location"), QFileInfo(mDownloads[i]->m_output).filePath());
580  settings.setValue(key + QLatin1String("done"), mDownloads[i]->downloadedSuccessfully());
581  }
582 
583  int i = mDownloads.count();
584  QString key = QString(QLatin1String("download_%1_")).arg(i);
585 
586  while (settings.contains(key + QLatin1String("url")))
587  {
588  settings.remove(key + QLatin1String("url"));
589  settings.remove(key + QLatin1String("location"));
590  settings.remove(key + QLatin1String("done"));
591  key = QString(QLatin1String("download_%1_")).arg(++i);
592  }
593 }

596 void WBDownloadManager::load()
597 {
598  mIsLoading = true;
599 
600  QSettings settings;
601  settings.beginGroup(QLatin1String("downloadmanager"));
602 
603  QByteArray value = settings.value(QLatin1String("removeDownloadsPolicy"), QLatin1String("Never")).toByteArray();
604  QMetaEnum removePolicyEnum = staticMetaObject.enumerator(staticMetaObject.indexOfEnumerator("RemovePolicy"));
605  m_RemovePolicy = removePolicyEnum.keyToValue(value) == -1 ?
606  Never :
607  static_cast<RemovePolicy>(removePolicyEnum.keyToValue(value));
608 
609  int i = 0;
610  QString key = QString(QLatin1String("download_%1_")).arg(i);
611 
612  while (settings.contains(key + QLatin1String("url")))
613  {
614  QUrl url = settings.value(key + QLatin1String("url")).toUrl();
615  QString fileName = settings.value(key + QLatin1String("location")).toString();
616  bool done = settings.value(key + QLatin1String("done"), true).toBool();
617  if (!url.isEmpty() && !fileName.isEmpty())
618  {
619  WBDownloadItem *item = new WBDownloadItem(0, this);
620  item->m_output.setFileName(fileName);
621  item->fileNameLabel->setText(QFileInfo(item->m_output.fileName()).fileName());
622  item->m_url = url;
623  item->stopButton->setVisible(false);
624  item->stopButton->setEnabled(false);
625  item->tryAgainButton->setVisible(!done);
626  item->tryAgainButton->setEnabled(!done);
627  item->progressBar->setVisible(!done);
628  addItem(item);
629  }
630  key = QString(QLatin1String("download_%1_")).arg(++i);
631  }
632 
633  cleanupButton->setEnabled(mDownloads.count() - activeDownloads() > 0);
634 
635  mIsLoading = false;
636 }
