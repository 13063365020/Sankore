242 void UBDesktopAnnotationController::setAssociatedPalettePosition(UBActionPalette *palette, const QString &actionName)
243 {
244  QPoint desktopPalettePos = mDesktopPalette->geometry().topLeft();
245  QList<QAction*> actions = mDesktopPalette->actions();
246  int yPen = 0;
247 
248  foreach(QAction* act, actions)
249  {
250  if(act->objectName() == actionName)
251  {
252  int iAction = actions.indexOf(act);
253  yPen = iAction * (mDesktopPalette->buttonSize().height() + 2 * mDesktopPalette->border() +6); // This is the mysterious value (6)
254  break;
255  }
256  }
257 
258  // First determine if the palette must be shown on the left or on the right
259  if(desktopPalettePos.x() <= (mTransparentDrawingView->width() - (palette->width() + mDesktopPalette->width() + mRightPalette->width() + 20))) // we take a small margin of 20 pixels
260  {
261  // Display it on the right
262  desktopPalettePos += QPoint(mDesktopPalette->width(), yPen);
263  }
264  else
265  {
266  // Display it on the left
267  desktopPalettePos += QPoint(0 - palette->width(), yPen);
268  }
269 
270  palette->setCustomPosition(true);
271  palette->move(desktopPalettePos);
272 }

836 void UBDesktopAnnotationController::updateMask(bool bTransparent)
837 {
838  if(bTransparent)
839  {
840  // Here we have to generate a new mask This method is certainly resource
841  // consuming but for the moment this is the only solution that I found.
842  mMask = QPixmap(mTransparentDrawingView->width(), mTransparentDrawingView->height());
843  mMask.fill(Qt::transparent);
844 
845  QPainter p;
846 
847  p.begin(&mMask);
848 
849  p.setPen(Qt::red);
850  p.setBrush(QBrush(Qt::red));
851 
852  // Here we draw the widget mask
853  if(mDesktopPalette->isVisible())
854  {
855  p.drawRect(mDesktopPalette->geometry().x(), mDesktopPalette->geometry().y(), mDesktopPalette->width(), mDesktopPalette->height());
856  }
857  if(UBApplication::boardController->paletteManager()->mKeyboardPalette->isVisible())
858  {
859  p.drawRect(UBApplication::boardController->paletteManager()->mKeyboardPalette->geometry().x(), UBApplication::boardController->paletteManager()->mKeyboardPalette->geometry().y(),
860  UBApplication::boardController->paletteManager()->mKeyboardPalette->width(), UBApplication::boardController->paletteManager()->mKeyboardPalette->height());
861  }
862 
863  if(UBApplication::boardController->paletteManager()->leftPalette()->isVisible())
864  {
865  QRect leftPalette(UBApplication::boardController->paletteManager()->leftPalette()->geometry().x(),
866  UBApplication::boardController->paletteManager()->leftPalette()->geometry().y(),
867  UBApplication::boardController->paletteManager()->leftPalette()->width(),
868  UBApplication::boardController->paletteManager()->leftPalette()->height());
869 
870  QRect tabsPalette(UBApplication::boardController->paletteManager()->leftPalette()->getTabPaletteRect());
871 
872  p.drawRect(leftPalette);
873  p.drawRect(tabsPalette);
874  }
875 
876  if(UBApplication::boardController->paletteManager()->rightPalette()->isVisible())
877  {
878  QRect rightPalette(UBApplication::boardController->paletteManager()->rightPalette()->geometry().x(),
879  UBApplication::boardController->paletteManager()->rightPalette()->geometry().y(),
880  UBApplication::boardController->paletteManager()->rightPalette()->width(),
881  UBApplication::boardController->paletteManager()->rightPalette()->height());
882 
883  QRect tabsPalette(UBApplication::boardController->paletteManager()->rightPalette()->getTabPaletteRect());
884 
885  p.drawRect(rightPalette);
886  p.drawRect(tabsPalette);
887  }
888 
889 #ifdef Q_WS_X11
890  //Rquiered only for compiz wm
891  //TODO. Window manager detection screen
892 
893  if (UBApplication::boardController->paletteManager()->addItemPalette()->isVisible()) {
894  p.drawRect(UBApplication::boardController->paletteManager()->addItemPalette()->geometry());
895  }
896 
897 #endif
898 
899  p.end();
900 
901  // Then we add the annotations. We create another painter because we need to
902  // apply transformations on it for coordinates matching
903  QPainter annotationPainter;
904 
905  QTransform trans;
906  trans.translate(mTransparentDrawingView->width()/2, mTransparentDrawingView->height()/2);
907 
908  annotationPainter.begin(&mMask);
909  annotationPainter.setPen(Qt::red);
910  annotationPainter.setBrush(Qt::red);
911 
912  annotationPainter.setTransform(trans);
913 
914  QList<QGraphicsItem*> allItems = mTransparentDrawingScene->items();
915 
916  for(int i = 0; i < allItems.size(); i++)
917  {
918  QGraphicsItem* pCrntItem = allItems.at(i);
919 
920  if(pCrntItem->isVisible() && pCrntItem->type() == UBGraphicsPolygonItem::Type)
921  {
922  QPainterPath crntPath = pCrntItem->shape();
923  QRectF rect = crntPath.boundingRect();
924 
925  annotationPainter.drawRect(rect);
926  }
927  }
928 
929  annotationPainter.end();
930 
931  mTransparentDrawingView->setMask(mMask.mask());
932  }
933  else
934  {
935  // Remove the mask
936  QPixmap noMask(mTransparentDrawingView->width(), mTransparentDrawingView->height());
937  mTransparentDrawingView->setMask(noMask.mask());
938  }
939 }
