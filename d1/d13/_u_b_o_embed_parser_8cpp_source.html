56 void UBOEmbedParser::parse(const QString& html)
57 {
58  mContents.clear();
59  QString query = "<link([^>]*)>";
60  QRegExp exp(query);
61  QStringList results;
62  int count = 0;
63  int pos = 0;
64  while ((pos = exp.indexIn(html, pos)) != -1) {
65  ++count;
66  pos += exp.matchedLength();
67  QStringList res = exp.capturedTexts();
68  if("" != res.at(1)){
69  results << res.at(1);
70  }
71  }
72 
73  QVector<QString> oembedUrls;
74 
75  if(2 <= results.size()){
76  for(int i=1; i<results.size(); i++){
77  if("" != results.at(i)){
78  QString qsNode = QString("<link%0>").arg(results.at(i));
79  QDomDocument domDoc;
80  domDoc.setContent(qsNode);
81  QDomNode linkNode = domDoc.documentElement();
82 
83  // At this point, we have a node that is the <link> element. Now we have to parse its attributes
84  // in order to check if it is a oEmbed node or not
85  QDomAttr typeAttribute = linkNode.toElement().attributeNode("type");
86  if(typeAttribute.value().contains("oembed")){
87  // The node is an oembed one! We have to get the url and the type of oembed encoding
88  QDomAttr hrefAttribute = linkNode.toElement().attributeNode("href");
89  QString url = hrefAttribute.value();
90  oembedUrls.append(url);
91  }
92  }
93  }
94  }
95 
96  mPending = oembedUrls.size();
97 
98  if(0 == mPending){
99  emit oembedParsed(mContents);
100  }else{
101  // Here we start the parsing (finally...)!
102  for(int i=0; i<oembedUrls.size(); i++){
103  emit parseContent(oembedUrls.at(i));
104  }
105  }
106 }

166 sOEmbedContent UBOEmbedParser::getXMLInfos(const QString &xml)
167 {

169 
170  QDomDocument domDoc;
171  domDoc.setContent(xml);
172  QDomNode oembed = domDoc.documentElement();
173 
174  QDomNodeList children = oembed.toElement().childNodes();
175 
176  for(int i=0; i<children.size(); i++){
177  QDomNode node = children.at(i);
178  QString tag = node.nodeName();
179  QString value = node.toElement().text();
180  if("provider_url" == tag){
181  content.providerUrl = value;
182  }else if("title" == tag){
183  content.title = value;
184  }else if("html" == tag){
185  content.html = value;
186  }else if("author_name" == tag){
187  content.author = value;
188  }else if("height" == tag){
189  content.height = value.toInt();
190  }else if("thumbnail_width" == tag){
191  content.thumbWidth = value.toInt();
192  }else if("width" == tag){
193  content.width = value.toInt();
194  }else if("version" == tag){
195  content.version = value.toFloat();
196  }else if("author_url" == tag){
197  content.authorUrl = value;
198  }else if("provider_name" == tag){
199  content.providerName = value;
200  }else if("thumbnail_url" == tag){
201  content.thumbUrl = value;
202  }else if("type" == tag){
203  content.type = value;
204  }else if("thumbnail_height" == tag){

235 void UBOEmbedParser::onFinished(QNetworkReply *reply)
236 {
237  if(QNetworkReply::NoError == reply->error()){
238  QString receivedDatas = reply->readAll().constData();
239  sOEmbedContent crntContent;
240  // The received datas can be in two different formats: JSON or XML
241  if(receivedDatas.contains("<oembed>")){
242  // XML !
243  crntContent = getXMLInfos(receivedDatas);
244  }else if(receivedDatas.contains("{\"provider_url")){
245  // JSON !
246  crntContent = getJSONInfos(receivedDatas);
247  }
248 
249  // As we don't want duplicates, we have to check if the content title has already
250  // been parsed.
251  if("" != crntContent.title && !mParsedTitles.contains(crntContent.title)){
252  mParsedTitles << crntContent.title;
253  mContents << crntContent;
254  }
255 
256  }else{
257  // We decided to not handle the error case here. If there is a problem with
258  // getting the oembed content information, we just don't handle it: the content
259  // will not be available for importation.
260  }
261 
262  // Decrement the number of content to analyze
263  mPending--;
264  if(0 == mPending){
265  // All the oembed contents have been parsed. We notify it!
266  emit oembedParsed(mContents);
267  }
268 }
