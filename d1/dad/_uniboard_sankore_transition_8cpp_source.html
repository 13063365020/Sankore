153 bool UniboardSankoreTransition::checkWidget(QString& sankoreWidgetIndexPath)
154 {
155  QFile file(sankoreWidgetIndexPath);
156  if (!file.open(QIODevice::ReadOnly | QIODevice::Text))
157  return false;
158 
159  QByteArray documentByteArray = file.readAll();
160  file.close();
161 
162  QString documentString(documentByteArray);
163 
164  QRegExp swfOriginFilePathRegExp("<param name=\"movie\" value=\"(.*)\">");
165  swfOriginFilePathRegExp.setMinimal(true);
166  swfOriginFilePathRegExp.indexIn(documentString);
167  QString origin = swfOriginFilePathRegExp.cap(1);
168  if(origin.contains("http://")){
169  // an url is the source of the swf. The source is kept as is.
170  return true;
171  }
172 
173  //changing the path
174  QRegExp swfDataPathRegExp("<object(.*)data=\"(.*)interactive content/Web/(.*)\"(.*)>");
175  swfDataPathRegExp.setMinimal(true);
176  documentString.replace(swfDataPathRegExp,"<object\\1data=\"\\3\">");
177 
178  QRegExp swfMoviePathRegExp("<param name=\"movie\" value=\"(.*)interactive content/Web/(.*)\">");
179  swfMoviePathRegExp.setMinimal(true);
180  documentString.replace(swfMoviePathRegExp,"<param name=\"movie\" value=\"\\2\">");
181 
182  //copy the swf on the right place
183  QRegExp swfFileNameRegExp("<param name=\"movie\" value=\"(.*)\">");
184  swfFileNameRegExp.setMinimal(true);
185  swfFileNameRegExp.indexIn(documentString);
186  QString swfFileName = swfFileNameRegExp.cap(1);
187  int lastDirectoryLevel = sankoreWidgetIndexPath.lastIndexOf("/");
188  if (lastDirectoryLevel == -1)
189  lastDirectoryLevel = sankoreWidgetIndexPath.lastIndexOf("\\");
190 
191 
192  QString destination(sankoreWidgetIndexPath.left(lastDirectoryLevel) + "/" + swfFileName);
193  QFile(origin).copy(destination);
194 
195  if (!file.open(QIODevice::WriteOnly | QIODevice::Text))
196  return false;
197 
198  file.write(documentString.toAscii());
199  file.close();
200 
201  return true;
202 }

243 void UniboardSankoreTransition::executeTransition()
244 {
245  bool result = false;
246  QString backupDestinationPath = mTransitionDlg->backupPath() + "/OldSankoreAndUniboardVersionsBackup";
247  result = UBFileSystemUtils::copyDir(mUniboardSourceDirectory + "/document", backupDestinationPath);
248  result &= UBFileSystemUtils::copyDir(mOldSankoreDirectory + "/document", backupDestinationPath);
249 
250  QString uniboardDocumentDirectory = mUniboardSourceDirectory + "/document";
251  QFileInfoList fileInfoList = UBFileSystemUtils::allElementsInDirectory(uniboardDocumentDirectory);
252  fileInfoList.append(UBFileSystemUtils::allElementsInDirectory(mOldSankoreDirectory + "/document"));
253 
254  QFileInfoList::iterator fileInfo;
255  QString sankoreDocumentDirectory = UBSettings::userDocumentDirectory();
256 
257  for (fileInfo = fileInfoList.begin(); fileInfo != fileInfoList.end() && result; fileInfo += 1) {
258  if (fileInfo->isDir() && (fileInfo->fileName().startsWith("Uniboard Document ") || fileInfo->fileName().startsWith("Sankore Document "))){
259  QString sankoreDocumentName = fileInfo->fileName();
260  emit transitioningFile(sankoreDocumentName);
261  sankoreDocumentName.replace("Uniboard","Sankore");
262  QString sankoreDocumentPath = sankoreDocumentDirectory + "/" + sankoreDocumentName;
263  result = UBFileSystemUtils::copyDir(fileInfo->filePath(),sankoreDocumentPath);
264  result &= updateSankoreHRef(sankoreDocumentPath);
265  QString sankoreWidgetPath = sankoreDocumentDirectory + "/" + sankoreDocumentName + "/widgets";
266  result &= updateIndexWidget(sankoreWidgetPath);
267  }
268  }
269 
270  if (!result){
271  qWarning() << "The transition has failed";
272  rollbackDocumentsTransition(fileInfoList);
273  UBFileSystemUtils::deleteDir(backupDestinationPath);
274  }
275  else{
276  UBFileSystemUtils::deleteDir(mOldSankoreDirectory);
277  UBFileSystemUtils::deleteDir(mUniboardSourceDirectory);
278  }
279 
280  emit transitionFinished(result);
281 }
