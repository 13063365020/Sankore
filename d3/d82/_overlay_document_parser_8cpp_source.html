54 
55  for(objAndSIter = objectsAndPositions.begin(); objAndSIter != objectsAndPositions.end(); ++objAndSIter)

57  unsigned int nextPosition = fileSize;
58  for(objAndPIter = objectsAndPositions.begin(); objAndPIter != objectsAndPositions.end(); ++objAndPIter)
59  {
60  if((objAndPIter->second > objAndSIter->second) && (objAndPIter->second < nextPosition))
61  nextPosition = objAndPIter->second;
62  }
63  objectsAndSizes[objAndSIter->first] = nextPosition - objAndSIter->second;
64  }
65 
66  bool notEndOfFile = true;
67  do
68  {
69  unsigned long partStart = fileSize;
70  std::map<unsigned int, unsigned long>::iterator objIter;
71  for(objIter = objectsAndPositions.begin(); objIter != objectsAndPositions.end(); ++objIter)
72  {
73  if(objIter->second < partStart)
74  partStart = objIter->second;
75  }
76  unsigned long nextPartStart = partStart + partSize;
77 
78  if((nextPartStart) < fileSize)
79  _getPartOfFileContent(partStart, partSize);
80  else
81  {
82  _getPartOfFileContent(partStart, fileSize - partStart);
83  nextPartStart = fileSize;
84  notEndOfFile = false;
85  }
86 
87  for(objIter = objectsAndPositions.begin(); objIter != objectsAndPositions.end(); )
88  {
89  if((objectsAndSizes[objIter->first] + objIter->second <= nextPartStart) && (objIter->second >= partStart) && ((objIter->second < nextPartStart)))
90  {
91  std::pair<unsigned int, unsigned int> streamBounds;
92  unsigned int objectNumber;
93  unsigned int generationNumber;
94  bool hasObjectStream;
95  const std::string content = _getObjectContent(objIter->second - partStart, objectNumber, generationNumber, streamBounds, hasObjectStream);
96  streamBounds.first += partStart;
97  streamBounds.second += partStart;
98  Object * newObject = new Object(objectNumber, generationNumber, content, _document->_documentName ,streamBounds, hasObjectStream);
99  _objects[objectNumber] = newObject;
100  std::map<unsigned int, unsigned long>::iterator temp = objIter;
101  ++objIter;
102  objectsAndPositions.erase(temp);
103  continue;
104  }
105  ++objIter;
106  }
107  partStart = nextPartStart;
108  }
109  while(notEndOfFile);
110 }

117 void OverlayDocumentParser::_getPartOfFileContent(long startOfPart, unsigned int length)
118 {
119  ifstream pdfFile;
120  pdfFile.open (_fileName.c_str(), ios::binary );
121  if (pdfFile.fail())
122  {
123  stringstream errorMessage("File ");

125  throw Exception(errorMessage);
126  }
127  ios_base::seekdir dir;
128  if(startOfPart >= 0)
129  dir = ios_base::beg;
130  else
131  dir = ios_base::end;

133  _fileContent.resize(length);

138 void OverlayDocumentParser::_readXref(std::map<unsigned int, unsigned long> & objectsAndSizes)
139 {
140  _getPartOfFileContent(- DOC_PART_WITH_START_OF_XREF, DOC_PART_WITH_START_OF_XREF);
141  unsigned int startOfStartxref = _fileContent.find("startxref");
142  unsigned int startOfNumber = _fileContent.find_first_of(Parser::NUMBERS, startOfStartxref);
143  unsigned int endOfNumber = _fileContent.find_first_not_of(Parser::NUMBERS, startOfNumber + 1);
144  std::string startXref = _fileContent.substr(startOfNumber, endOfNumber - startOfNumber);
145  unsigned int strtXref = Utils::stringToInt(startXref);
146 
147  unsigned int sizeOfXref = Utils::getFileSize(_fileName.c_str()) - strtXref;
148  _getPartOfFileContent(strtXref, sizeOfXref);
149  unsigned int leftBoundOfObjectNumber = _fileContent.find("0 ") + strlen("0 ");
150  unsigned int rightBoundOfObjectNumber = _fileContent.find_first_not_of(Parser::NUMBERS, leftBoundOfObjectNumber);
151  std::string objectNuberStr = _fileContent.substr(leftBoundOfObjectNumber, rightBoundOfObjectNumber - leftBoundOfObjectNumber);
152  unsigned long objectNumber = Utils::stringToInt(objectNuberStr);
153  unsigned int startOfObjectPosition = _fileContent.find("0000000000 65535 f ") + strlen("0000000000 65535 f ");
154  for(unsigned long i = 1; i < objectNumber; ++i)
155  {
156  startOfObjectPosition = _fileContent.find_first_of(Parser::NUMBERS, startOfObjectPosition);
157  unsigned int endOfObjectPostion = _fileContent.find(" 00000 n", startOfObjectPosition);
158  std::string objectPostionStr = _fileContent.substr(startOfObjectPosition, endOfObjectPostion - startOfObjectPosition);
159  objectsAndSizes[i] = Utils::stringToInt(objectPostionStr);
160  startOfObjectPosition = endOfObjectPostion + strlen(" 00000 n");
161  }
