121 void UBFeaturesWidget::currentSelected(const QModelIndex &current)
122 {
123  if (!current.isValid()) {
124  qWarning() << "SLOT:currentSelected, invalid index catched";
125  return;
126  }
127 
128  QString objName = sender()->objectName();
129 
130  if (objName.isEmpty()) {
131  qWarning() << "incorrect sender";
132  } else if (objName == objNamePathList) {
133  //Calling to reset the model for listView. Maybe separate function needed
134  controller->searchStarted("", centralWidget->listView());
135  }
136 
137  UBFeature feature = controller->getFeature(current, objName);
138 
139  if ( feature.isFolder() ) {
140  QString newPath = feature.getFullVirtualPath();
141 
142  controller->setCurrentElement(feature);
143  controller->siftElements(newPath);
144 
145  centralWidget->switchTo(UBFeaturesCentralWidget::MainList);
146 
147  if ( feature.getType() == FEATURE_FAVORITE ) {
148  mActionBar->setCurrentState( IN_FAVORITE );
149 
150  } else if ( feature.getType() == FEATURE_CATEGORY && feature.getName() == "root" ) {
151  mActionBar->setCurrentState( IN_ROOT );
152 
153  } else if (feature.getType() == FEATURE_TRASH) {
154  mActionBar->setCurrentState(IN_TRASH);
155 
156  } else if (feature.getType() == FEATURE_SEARCH) {
157  //The search feature behavior is not standard. If features list clicked - show empty element
158  //else show existing saved features search QWebView
159  if (sender()->objectName() == objNameFeatureList) {
160  centralWidget->showElement(feature, UBFeaturesCentralWidget::FeaturesWebView);
161  } else if (sender()->objectName() == objNamePathList) {
162  centralWidget->switchTo(UBFeaturesCentralWidget::FeaturesWebView);
163  }
164 
165  } else {
166  mActionBar->setCurrentState(IN_FOLDER);
167  }
168 
169  } else {
170  if(feature.getType() == FEATURE_BOOKMARK){
171  QString url;
172  QFile bookmarkFile(feature.getFullPath().toLocalFile());
173  if(bookmarkFile.open(QIODevice::ReadOnly|QIODevice::Text)){
174  url = QString::fromUtf8(bookmarkFile.readAll());
175  bookmarkFile.close();
176  UBApplication::webController->loadUrl(QUrl(url));
177  }
178  else
179  qWarning() << "failed to read file named " << feature.getFullPath().toLocalFile();
180  }
181  else{
182  centralWidget->showElement(feature, UBFeaturesCentralWidget::FeaturePropertiesList);
183  mActionBar->setCurrentState( IN_PROPERTIES );
184  }
185  }
186 
187  mActionBar->cleanText();
188  emit allowNewFolderButton(controller->newFolderAllowed());
189 
190 }

301 void UBFeaturesWidget::onDisplayMetadata( QMap<QString,QString> metadata )
302 {
303  QString previewImageUrl = ":images/libpalette/notFound.png";
304 
305  QString widgetsUrl = QUrl::fromEncoded(metadata["Url"].toAscii()).toString();
306  QString widgetsThumbsUrl = QUrl::fromEncoded(metadata["thumbnailUrl"].toAscii()).toString();

309  metadata.insert("Title",metadata["Title"] + "." + QFileInfo(widgetsUrl).suffix());
310

312  UBMimeType::Enum thumbType = UBFileSystemUtils::mimeTypeFromString(strType);
313 
314  switch (static_cast<int>(thumbType)) {
315  case UBMimeType::Audio:
316  previewImageUrl = ":images/libpalette/soundIcon.svg";
317  break;
318 
319  case UBMimeType::Video:
320  previewImageUrl = ":images/libpalette/movieIcon.svg";
321  break;
322 
323  case UBMimeType::Flash:
324  previewImageUrl = ":images/libpalette/FlashIcon.svg";
325  break;
326 
327  case UBMimeType::RasterImage:
328  case UBMimeType::VectorImage:
329  previewImageUrl = widgetsUrl;
330  break;
331  }
332 
333  if (!widgetsThumbsUrl.isNull()) {
334  previewImageUrl = ":/images/libpalette/loading.png";
335  if (!imageGatherer)
336  imageGatherer = new UBDownloadHttpFile(0, this);
337 
338  connect(imageGatherer, SIGNAL(downloadFinished(int, bool, QUrl, QUrl, QString, QByteArray, QPointF, QSize, bool)), this, SLOT(onPreviewLoaded(int, bool, QUrl, QUrl, QString, QByteArray, QPointF, QSize, bool)));
339 
340  // We send here the request and store its reply in order to be able to cancel it if needed
341  imageGatherer->get(QUrl(widgetsThumbsUrl), QPoint(0,0), QSize(), false);
342  }
343 
344  UBFeature feature( "/root", QImage(previewImageUrl), QString(), widgetsUrl, FEATURE_ITEM );
345  feature.setMetadata( metadata );

489 void UBFeaturesListView::dragMoveEvent( QDragMoveEvent *event )
490 {
491  const UBFeaturesMimeData *fMimeData = qobject_cast<const UBFeaturesMimeData*>(event->mimeData());
492  QModelIndex index = indexAt(event->pos());
493  UBFeature onFeature = model()->data(index, Qt::UserRole + 1).value<UBFeature>();
494  if (fMimeData) {
495  if (!index.isValid()
496  || !onFeature.isFolder()
497  || !onFeature.testPermissions(UBFeature::WRITE_P)) {
498  event->ignore();
499  return;
500  }
501  foreach (UBFeature curFeature, fMimeData->features()) {
502  if (curFeature == onFeature) {
503  event->ignore();
504  return;
505  }
506  }
507  }
508 
509  if ( event->mimeData()->hasUrls() || event->mimeData()->hasImage() ) {
510  event->acceptProposedAction();
511  } else {
512  event->ignore();
513  }
