158 UBDocumentProxy* UBDocumentManager::importFile(const QFile& pFile, const QString& pGroup)
159 {
160  QFileInfo fileInfo(pFile);
161 
162  foreach (UBImportAdaptor *adaptor, mImportAdaptors)

164  if (adaptor->supportedExtentions().lastIndexOf(fileInfo.suffix().toLower()) != -1)

182  document = UBPersistenceManager::persistenceManager()->createDocument(pGroup

185  , QString()
186  , 0

188

191  if (importAdaptor->folderToCopy() != "")

229 int UBDocumentManager::addFilesToDocument(UBDocumentProxy* document, QStringList fileNames)
230 {
231  int nImportedDocuments = 0;
232  foreach(const QString& fileName, fileNames)
233  {
234  UBApplication::showMessage(tr("Importing file %1").arg(fileName));
235 
236  QFile file(fileName);
237  QFileInfo fileInfo(file);
238 
239  foreach (UBImportAdaptor *adaptor, mImportAdaptors)
240  {
241  if (adaptor->supportedExtentions().lastIndexOf(fileInfo.suffix().toLower()) != -1)
242  {
243  UBApplication::setDisabled(true);
244 
245  if (adaptor->isDocumentBased())
246  {
247  UBDocumentBasedImportAdaptor* importAdaptor = (UBDocumentBasedImportAdaptor*)adaptor;
248 
249  if (importAdaptor->addFileToDocument(document, file))
250  nImportedDocuments++;
251  }
252  else
253  {
254  UBPageBasedImportAdaptor* importAdaptor = (UBPageBasedImportAdaptor*)adaptor;
255 
256  QUuid uuid = QUuid::createUuid();
257  QString filepath = file.fileName();
258  if (importAdaptor->folderToCopy() != "")
259  {
260  bool b = UBPersistenceManager::persistenceManager()->addFileToDocument(document, file.fileName(), importAdaptor->folderToCopy() , uuid, filepath);
261  if (!b)
262  {
263  continue;
264  }
265  }
266 
267  QList<UBGraphicsItem*> pages = importAdaptor->import(uuid, filepath);
268  int nPage = 0;
269  foreach(UBGraphicsItem* page, pages)
270  {
271  UBApplication::showMessage(tr("Inserting page %1 of %2").arg(++nPage).arg(pages.size()), true);
272  int pageIndex = document->pageCount();
273  UBGraphicsScene* scene = UBPersistenceManager::persistenceManager()->createDocumentSceneAt(document, pageIndex);
274  importAdaptor->placeImportedItemToScene(scene, page);
275  UBPersistenceManager::persistenceManager()->persistDocumentScene(document, scene, pageIndex);
276  UBApplication::boardController->addEmptyThumbPage();
277  }
278 
279  UBPersistenceManager::persistenceManager()->persistDocumentMetadata(document);
280  UBApplication::showMessage(tr("Import of file %1 successful.").arg(file.fileName()));
281  nImportedDocuments++;
282  }
283 
284  UBApplication::setDisabled(false);
285  }
286  }
287  }
288  return nImportedDocuments;
289 }
