191 void UBTeacherGuideEditionWidget::load(QDomDocument doc)
192 {
193  qDebug() << "LOAD UBApplication::boardController->currentPage() " << UBApplication::boardController->currentPage();
194  cleanData();
195  for (QDomElement element = doc.documentElement().firstChildElement();
196  !element.isNull(); element = element.nextSiblingElement()) {
197  QString tagName = element.tagName();
198  if (tagName == "title")
199  mpPageTitle->setInitialText(element.attribute("value"));
200  else if (tagName == "comment")
201  mpComment->setInitialText(element.attribute("value"));
202  else if (tagName == "media")
203  onAddItemClicked(mpAddAMediaItem, 0, &element);
204  else if (tagName == "link")
205  onAddItemClicked(mpAddALinkItem, 0, &element);
206  else if (tagName == "action")
207  onAddItemClicked(mpAddAnActionItem, 0, &element);
208  }
209 }

326 void UBTeacherGuideEditionWidget::onAddItemClicked(QTreeWidgetItem* widget, int column, QDomElement *element)
327 {
328  int addSubItemWidgetType = widget->data(column, Qt::UserRole).toInt();
329  if (addSubItemWidgetType != eUBTGAddSubItemWidgetType_None) {
330  QTreeWidgetItem* newWidgetItem = new QTreeWidgetItem(widget);
331  newWidgetItem->setData(column, Qt::UserRole, eUBTGAddSubItemWidgetType_None);
332  newWidgetItem->setData(1, Qt::UserRole, eUBTGAddSubItemWidgetType_None);
333  newWidgetItem->setIcon(1, QIcon(":images/close.svg"));
334 
335  switch (addSubItemWidgetType) {
336  case eUBTGAddSubItemWidgetType_Action: {
337  UBTGActionWidget* actionWidget = new UBTGActionWidget(widget);
338  if (element)
339  actionWidget->initializeWithDom(*element);
340  mpTreeWidget->setItemWidget(newWidgetItem, 0, actionWidget);
341  break;
342  }
343  case eUBTGAddSubItemWidgetType_Media: {
344  UBTGMediaWidget* mediaWidget = new UBTGMediaWidget(widget);
345  if (element)
346  mediaWidget->initializeWithDom(*element);
347  mpTreeWidget->setItemWidget(newWidgetItem,0, mediaWidget);
348  break;
349  }
350  case eUBTGAddSubItemWidgetType_Url: {
351  UBTGUrlWidget* urlWidget = new UBTGUrlWidget();
352  if (element)
353  urlWidget->initializeWithDom(*element);
354  mpTreeWidget->setItemWidget(newWidgetItem, 0, urlWidget);
355  break;
356  }
357  default:
358  delete newWidgetItem;
359  qCritical() << "onAddItemClicked no action set";
360  return;
361  }
362 
363  if (addSubItemWidgetType != eUBTGAddSubItemWidgetType_None && !widget->isExpanded())
364  widget->setExpanded(true);
365  else {
366  //to update the tree and subtrees
367  widget->setExpanded(false);
368  widget->setExpanded(true);
369  }
370  }
371  else if (column == 1 && addSubItemWidgetType == eUBTGAddSubItemWidgetType_None) {
372  UBTGMediaWidget* media = dynamic_cast<UBTGMediaWidget*>(mpTreeWidget->itemWidget(widget, 0));
373  if (media)
374  media->removeSource();
375  int index = mpTreeWidget->currentIndex().row();
376  QTreeWidgetItem* toBeDeletedWidgetItem = widget->parent()->takeChild(index);
377  delete toBeDeletedWidgetItem;
378  }
379 }
