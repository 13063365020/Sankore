54 void UBHttpFileDownloader::downloadNext()
55 {
56  if (mUrlsToDownload.length() > 0 && mFilesToSave.length() > 0)
57  {
58  QUrl url = mUrlsToDownload.takeFirst();
59  mCurrentFile = mFilesToSave.takeFirst();
60 
61  if (!mCurrentFile->open(QIODevice::WriteOnly | QIODevice::Truncate))
62  {
63  qCritical() << "cannot open " << mCurrentFile << " for writing ...";
64  return;
65  }
66 
67  UBNetworkAccessManager * nam = UBNetworkAccessManager::defaultAccessManager();
68  QNetworkRequest request(url);
69  mReply = nam->get(request); // UB 4.2 TODO who owns and delete the reply ?
70 
71  connect(mReply, SIGNAL(finished ()), this, SLOT(finished ()));
72  connect(mReply, SIGNAL(error ( QNetworkReply::NetworkError )),
73  this, SLOT(error ( QNetworkReply::NetworkError)));
74  connect(mReply, SIGNAL(downloadProgress ( qint64 , qint64 )),
75  this, SLOT(downloadProgress(qint64 , qint64)));
76 
77  connect(mReply, SIGNAL(readyRead ()), this, SLOT( readyRead()));
78 
79  }
80  else
81  {
82  emit finished(mSuccess);
83  }
84 
85 }
