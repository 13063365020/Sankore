152 // work around for handling tablet events on MAC OS with Qt 4.8.0 and above
153 #if defined(Q_WS_MACX)
154 bool UBMainWindow::event(QEvent *event)
155 {
156  bool bRes = QMainWindow::event(event);
157 
158  if (NULL != UBApplication::boardController)
159  {
160  UBBoardView *controlV = UBApplication::boardController->controlView();
161  if (controlV && controlV->isVisible())
162  {
163  switch (event->type())
164  {
165  case QEvent::TabletEnterProximity:
166  case QEvent::TabletLeaveProximity:
167  case QEvent::TabletMove:
168  case QEvent::TabletPress:
169  case QEvent::TabletRelease:
170  {
171  return controlV->directTabletEvent(event);
172  }
173  }
174  }
175  }
176  return bRes;
177 }

196 bool UBMainWindow::yesNoQuestion(QString windowTitle, QString text)
197 {
198  QMessageBox messageBox;
199  messageBox.setParent(this);
200  messageBox.setWindowTitle(windowTitle);
201  messageBox.setText(text);
202  QPushButton* yesButton = messageBox.addButton(tr("Yes"),QMessageBox::YesRole);
203  messageBox.addButton(tr("No"),QMessageBox::NoRole);
204  messageBox.setIcon(QMessageBox::Question);
205 
206 #ifdef Q_WS_X11
207  // to avoid to be handled by x11. This allows us to keep to the back all the windows manager stuff like palette, toolbar ...
208  messageBox.setWindowFlags(Qt::Dialog | Qt::X11BypassWindowManagerHint);
209 
210  //To calculate the correct size
211  messageBox.show();
212  QSize messageBoxSize=messageBox.size();
213 
214  //to center on the screen because it's no more handled by X11
215  QRect controlScreenRect = UBApplication::applicationController->displayManager()->controlGeometry();
216  messageBox.move((controlScreenRect.width()/2) - (messageBoxSize.width()*0.5), (controlScreenRect.height()/2) - (messageBoxSize.height()*0.5));
217 #else
218  messageBox.setWindowFlags(Qt::Dialog);
219 #endif
220 
221  messageBox.exec();
222  return messageBox.clickedButton() == yesButton;
223 }
