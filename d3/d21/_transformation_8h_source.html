55  void setParameters(double a, double b, double c, double d, double e, double f)
56  {
57  _a = Utils::normalizeValue(a);
58  _b = Utils::normalizeValue(b);
59  _c = Utils::normalizeValue(c);
60  _d = Utils::normalizeValue(d);
61  _e = Utils::normalizeValue(e);
62  _f = Utils::normalizeValue(f);

64  void add(const TransformationMatrix & tm)
65  {
66  double newA = _a*tm._a + _b*tm._c;
67  double newB = _a*tm._b + _b*tm._d;
68  double newC = _c*tm._a + _d*tm._c;
69  double newD = _c*tm._b + _d*tm._d;
70  double newE = _e*tm._a + _f*tm._c + tm._e;
71  double newF = _e*tm._b + _f*tm._d + tm._f;
72 
73  // we need to round the values to avoid not-needed transformation
74  // since 1.e-17 is not 0 from PDF point of view, while such double
75  // value really means 0.
76  _a = Utils::normalizeValue(newA);
77  _b = Utils::normalizeValue(newB);
78  _c = Utils::normalizeValue(newC);
79  _d = Utils::normalizeValue(newD);
80  _e = Utils::normalizeValue(newE);
81  _f = Utils::normalizeValue(newF);
82  }

90  }

265  // method recalculates the final translation in order to put
266  // object into needed x,y coordinates.
267  // Page is located from position 0,0
268  void recalculateTranslation(double width, double height)
269  {
270  double dx1 = 0;
271  double dy1 = 0;
272 
273  double scaling = ( Utils::doubleEquals(_scale,0))?1:_scale;
274 
275  switch(_angel)
276  {
277  case 0:
278  dx1 = _x/scaling;
279  dy1 = _y/scaling;
280  break;
281  case -270:
282  case 90:
283  dx1 = _y/scaling ;
284  dy1 = - _x /scaling - height;
285  break;
286  case 180:
287  case -180:
288  dx1 = - _x /scaling - width;
289  dy1 = - _y /scaling - height;
290  break;
291  case 270:
292  case -90:
293  dx1 = - _y/scaling - width;
294  dy1 = _x/scaling;
295  break;
296  default:
297  std::cerr<<"Unsupported rotation parameter"<<_angel<<std::endl;
298  break;
299  }
300 
301  //std::cerr<< "dx1 = "<<dx1<<"dy1 = "<<dy1<<std::endl;
302  if( ! (Utils::doubleEquals(dx1,0) && Utils::doubleEquals(dy1,0)) )
303  {
304  // This translation is needed to put transformed content into
305  // desired coordinates
306  _transforms.push_back(new Translation(dx1,dy1));
307  }
308  }
