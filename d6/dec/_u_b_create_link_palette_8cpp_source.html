68 void UBCreateLinkLabel::dropEvent(QDropEvent *event)
69 {
70  setBackgroundRole(QPalette::NoRole);
71  const QMimeData *mimeData = event->mimeData();
72 
73  QString path;
74 
75  if (mimeData->hasImage())
76  setText(tr("Images are not accepted"));
77  else if (mimeData->hasHtml())
78  path = mimeData->html();
79  else if (mimeData->hasText())
80  path = mimeData->text();
81  else if (mimeData->hasUrls())
82  path = mimeData->urls().at(0).toLocalFile();
83  else
84  setText(tr("Cannot display data"));
85 
86  if(UBFileSystemUtils::mimeTypeFromFileName(path).contains("audio")){
87  if(!path.startsWith("file:"))
88  setText(QFileInfo(path).completeBaseName());
89  else
90  setText(QFileInfo(QUrl(path).toLocalFile()).completeBaseName());
91  }
92  else
93  setText(tr("Dropped file isn't reconized to be an audio file"));
94 
95  emit droppedFile(path);
96  event->acceptProposedAction();
97 }
