71 void UBExportFullPDF::saveOverlayPdf(UBDocumentProxy* pDocumentProxy, const QString& filename)
72 {
73  if (!pDocumentProxy || filename.length() == 0 || pDocumentProxy->pageCount() == 0)
74  return;
75 
76  //PDF
77  qDebug() << "exporting document to PDF Merger" << filename;
78  QPrinter pdfPrinter;
79 
80  pdfPrinter.setOutputFormat(QPrinter::PdfFormat);
81  pdfPrinter.setResolution(UBSettings::settings()->pdfResolution->get().toInt());
82  pdfPrinter.setOutputFileName(filename);
83  pdfPrinter.setFullPage(true);
84 
85  QPainter* pdfPainter = 0;
86 
87  for(int pageIndex = 0 ; pageIndex < pDocumentProxy->pageCount(); pageIndex++)
88  {
89  UBGraphicsScene* scene = UBPersistenceManager::persistenceManager()->loadDocumentScene(pDocumentProxy, pageIndex);
90  // set background to white, no grid for PDF output
91  bool isDark = scene->isDarkBackground();
92  bool isCrossed = scene->isCrossedBackground();
93  scene->setBackground(false, false);
94 
95  // set high res rendering
96  scene->setRenderingQuality(UBItem::RenderingQualityHigh);
97  scene->setRenderingContext(UBGraphicsScene::PdfExport);
98 
99  QSize pageSize = scene->nominalSize();
100 
101  UBGraphicsPDFItem *pdfItem = qgraphicsitem_cast<UBGraphicsPDFItem*>(scene->backgroundObject());
102 
103  if (pdfItem) mHasPDFBackgrounds = true;
104 
105  pdfPrinter.setPaperSize(QSizeF(pageSize.width()*mScaleFactor, pageSize.height()*mScaleFactor), QPrinter::Point);
106 
107  if (!pdfPainter) pdfPainter = new QPainter(&pdfPrinter);
108 
109  if (pageIndex != 0) pdfPrinter.newPage();
110 
111  //render to PDF
112  scene->setDrawingMode(true);
113  scene->render(pdfPainter, QRectF(), scene->normalizedSceneRect());
114 
115  //restore screen rendering quality
116  scene->setRenderingContext(UBGraphicsScene::Screen);
117  scene->setRenderingQuality(UBItem::RenderingQualityNormal);
118 
119  //restore background state
120  scene->setDrawingMode(false);
121  scene->setBackground(isDark, isCrossed);
122  }
123 
124  if (pdfPainter) delete pdfPainter;
125 }

150 void UBExportFullPDF::persistsDocument(UBDocumentProxy* pDocumentProxy, const QString& filename)
151 {
152  QFile file(filename);
153  if (file.exists()) file.remove();
154 
155  QString overlayName = filename;
156  overlayName.replace(".pdf", "_overlay.pdf");
157 
158  QFile previousOverlay(overlayName);
159  if (previousOverlay.exists())
160  previousOverlay.remove();
161 
162  mHasPDFBackgrounds = false;
163 
164  saveOverlayPdf(pDocumentProxy, overlayName);
165 
166  if (!mHasPDFBackgrounds)
167  {
168  QFile f(overlayName);
169  f.rename(filename);
170  }
171  else
172  {
173  Merger merger;
174  try
175  {
176  merger.addOverlayDocument(QFile::encodeName(overlayName).constData());
177 
178  MergeDescription mergeInfo;
179 
180  int existingPageCount = pDocumentProxy->pageCount();
181 
182  for(int pageIndex = 0 ; pageIndex < existingPageCount; pageIndex++)
183  {
184  UBGraphicsScene* scene = UBPersistenceManager::persistenceManager()->loadDocumentScene(pDocumentProxy, pageIndex);
185  UBGraphicsPDFItem *pdfItem = qgraphicsitem_cast<UBGraphicsPDFItem*>(scene->backgroundObject());
186 
187  QSize pageSize = scene->nominalSize();
188 
189  if (pdfItem)
190  {
191  QString pdfName = UBPersistenceManager::objectDirectory + "/" + pdfItem->fileUuid().toString() + ".pdf";
192  QString backgroundPath = pDocumentProxy->persistencePath() + "/" + pdfName;
193  QRectF annotationsRect = scene->itemsBoundingRect();
194 
195  // Original datas
196  double xAnnotation = qRound(annotationsRect.x());
197  double yAnnotation = qRound(annotationsRect.y());
198  double xPdf = qRound(pdfItem->sceneBoundingRect().x());
199  double yPdf = qRound(pdfItem->sceneBoundingRect().y());
200  double hPdf = qRound(pdfItem->sceneBoundingRect().height());
201 
202  // Exportation-transformed datas
203  double hScaleFactor = pageSize.width()/scene->itemsBoundingRect().width();
204  double vScaleFactor = pageSize.height()/scene->itemsBoundingRect().height();
205  double scaleFactor = qMin(hScaleFactor, vScaleFactor);
206 
207  double xAnnotationsOffset = 0;
208  double yAnnotationsOffset = 0;
209  double hPdfTransformed = qRound(hPdf * scaleFactor);
210 
211  // Here, we force the PDF page to be on the topleft corner of the page
212  double xPdfOffset = 0;
213  double yPdfOffset = (hPdf - hPdfTransformed) * mScaleFactor;
214 
215  // Now we align the items
216  xPdfOffset += (xPdf - xAnnotation) * scaleFactor * mScaleFactor;
217  yPdfOffset -= (yPdf - yAnnotation) * scaleFactor * mScaleFactor;
218 
219  TransformationDescription pdfTransform(xPdfOffset, yPdfOffset, scaleFactor, 0);
220  TransformationDescription annotationTransform(xAnnotationsOffset, yAnnotationsOffset, 1, 0);
221 
222  MergePageDescription pageDescription(pageSize.width() * mScaleFactor,
223  pageSize.height() * mScaleFactor,
224  pdfItem->pageNumber(),
225  QFile::encodeName(backgroundPath).constData(),
226  pdfTransform,
227  pageIndex + 1,
228  annotationTransform,
229  false, false);
230 
231  mergeInfo.push_back(pageDescription);
232 
233  merger.addBaseDocument(QFile::encodeName(backgroundPath).constData());
234  }
235  else
236  {
237  MergePageDescription pageDescription(pageSize.width() * mScaleFactor,
238  pageSize.height() * mScaleFactor,
239  0,
240  "",
241  TransformationDescription(),
242  pageIndex + 1,
243  TransformationDescription(),
244  false, true);
245 
246  mergeInfo.push_back(pageDescription);
247  }
248  }
249 
250  merger.merge(QFile::encodeName(overlayName).constData(), mergeInfo);
251 
252  merger.saveMergedDocumentsAs(QFile::encodeName(filename).constData());
253 
254  }
255  catch(Exception e)
256  {
257  qDebug() << "PdfMerger failed to merge documents to " << filename << " - Exception : " << e.what();
258 
259  // default to raster export
260  UBExportPDF::persistsDocument(pDocumentProxy, filename);
261  }
262 
263  if (!UBApplication::app()->isVerbose())
264  {
265  QFile::remove(overlayName);
266  }
267  }
268 }
