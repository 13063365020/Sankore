138 static void hex_dump_one_line(int numberChars_n,
139  const char *input_p,
140  char *output_p)
141 {
142  int i;
143  char* to_hex_p;
144  char* to_char_p;
145  char char_c;
146 
147  /* Set pointer to the begining of hexadecimal area */
148  to_hex_p=output_p;
149 
150  /* Set pointer to the begining of textual area */
151  to_char_p=output_p+56;
152 
153  /* Write spaces between hexadecimal and textual areas */
154  memset(output_p+50,' ',6);
155 
156  /* some kind of delimeter */
157  *(output_p+53) = '#';
158 
159  /* Print out the hex area */
160  for (i = 0 ; i < 16 ; i++)
161  {
162  /* Two spaces beetwen "four columns" */
163  if (!(i&3))
164  {
165  *to_hex_p++=' ';
166  }
167 
168  /* One space between columns */
169  *to_hex_p++=' ';
170 
171  if (i < numberChars_n)
172  {
173  /* Print out byte in hexadecimal form */
174  *to_hex_p++=TO_HEX_CHAR((input_p[i]>>4)&0xF);
175  *to_hex_p++=TO_HEX_CHAR(input_p[i]&0xF);
176 
177  /* Output the char */
178  char_c = input_p[i]&0xFF;
179 
180  if ( char_c<0x20 || char_c>0x7E )
181  {
182  char_c = '.';
183  }
184  *to_char_p=char_c;
185  to_char_p++;
186  }
187  else
188  {
189  *to_hex_p++=' ';
190  *to_hex_p++=' ';
191  *to_char_p++=' ';
192  }
193  } /* for */
194 }
