129 void UBWebPluginPDFWidget::paintEvent(QPaintEvent *event)
130 {
131  QPainter painter(this);
132  if (!mRenderer)
133  {
134  // PDF not yet downloaded
135  return;
136  }
137 
138  if (!mRenderer->isValid())
139  {
140  qWarning("UBWebPluginPDFWidget::paintEvent: Invalid renderer");
141  return;
142  }
143 
144  QSizeF pageSize = mRenderer->pageSizeF(mPageNumber);
145 
146  painter.translate((geometry().width() - (pageSize.width() * mScale)) / 2, 0);
147  painter.scale(mScale, mScale);
148  mRenderer->render(&painter, mPageNumber, event->rect());
149  painter.setPen(QPen(Qt::gray, 1));
150  painter.drawRect(0, 0, pageSize.width(), pageSize.height());
151 }
