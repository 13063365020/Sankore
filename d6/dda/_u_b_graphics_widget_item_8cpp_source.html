377 QString UBGraphicsWidgetItem::widgetName(const QUrl& widgetPath)
378 {
379  QString name;
380  QString version;
381  QFile w3CConfigFile(widgetPath.toLocalFile() + "/config.xml");
382  QFile appleConfigFile(widgetPath.toLocalFile() + "/Info.plist");
383 
384  if (w3CConfigFile.exists() && w3CConfigFile.open(QFile::ReadOnly)) {
385  QDomDocument doc;
386  doc.setContent(w3CConfigFile.readAll());
387  QDomElement root = doc.firstChildElement("widget");
388  if (!root.isNull()) {
389  QDomElement nameElement = root.firstChildElement("name");
390  if (!nameElement.isNull())
391  name = nameElement.text();
392  version = root.attribute("version", "");
393  }
394  w3CConfigFile.close();
395  }
396  else if (appleConfigFile.exists() && appleConfigFile.open(QFile::ReadOnly)) {
397  QDomDocument doc;
398  doc.setContent(appleConfigFile.readAll());
399  QDomElement root = doc.firstChildElement("plist");
400  if (!root.isNull()) {
401  QDomElement dictElement = root.firstChildElement("dict");
402  if (!dictElement.isNull()) {
403  QDomNodeList childNodes = dictElement.childNodes();
404 
405  /* looking for something like
406  * ..
407  * <key>CFBundleDisplayName</key>
408  * <string>brain scans</string>
409  * ..
410  */
411 
412  for(int i = 0; i < childNodes.count() - 1; i++) {
413  if (childNodes.at(i).isElement()) {
414  QDomElement elKey = childNodes.at(i).toElement();
415  if (elKey.text() == "CFBundleDisplayName") {
416  if (childNodes.at(i + 1).isElement()) {
417  QDomElement elValue = childNodes.at(i + 1).toElement();
418  name = elValue.text();
419  }
420  }
421  else if (elKey.text() == "CFBundleShortVersionString") {
422  if (childNodes.at(i + 1).isElement()) {
423  QDomElement elValue = childNodes.at(i + 1).toElement();
424  version = elValue.text();
425  }
426  }
427  }
428  }
429  }
430  }
431  appleConfigFile.close();
432  }
433  QString result;
434 
435  if (name.length() > 0) {
436  result = name;
437  if (version.length() > 0) {
438  result += " ";
439  result += version;
440  }
441  }
442  return result;
443 }

674  int lastSlashIndex = path.lastIndexOf("/");
675  if (lastSlashIndex > 0)
676  path = path.mid(0, lastSlashIndex + 1);
677  }
678 
679  QFile plistFile(path + "/Info.plist");
680  plistFile.open(QFile::ReadOnly);
681 
682  QByteArray plistBin = plistFile.readAll();
683  QString plist = QString::fromUtf8(plistBin);
684 
685  int mainHtmlIndex = plist.indexOf("MainHTML");
686  int mainHtmlIndexStart = plist.indexOf("<string>", mainHtmlIndex);
687  int mainHtmlIndexEnd = plist.indexOf("</string>", mainHtmlIndexStart);
688 
689  if (mainHtmlIndex > -1 && mainHtmlIndexStart > -1 && mainHtmlIndexEnd > -1)
690  mMainHtmlFileName = plist.mid(mainHtmlIndexStart + 8, mainHtmlIndexEnd - mainHtmlIndexStart - 8);
691 
692  mMainHtmlUrl = pWidgetUrl;
693  mMainHtmlUrl.setPath(pWidgetUrl.path() + "/" + mMainHtmlFileName);
694 
695  load(mMainHtmlUrl);
696 
697  QPixmap defaultPixmap(pWidgetUrl.toLocalFile() + "/Default.png");
698
