35 UBToolbarButtonGroup::UBToolbarButtonGroup(QToolBar *toolBar, const QList<QAction*> &actions)
36  : QWidget(toolBar)

40  , mActionGroup(0)
41 {
42  Q_ASSERT(actions.size() > 0);
43 
44  mToolButton = qobject_cast<QToolButton*>(toolBar->layout()->itemAt(0)->widget());
45  Q_ASSERT(mToolButton);

54  mActionGroup = new QActionGroup(this);

56 
57  QSize buttonSize;
58

60

62  {
63  mActionGroup->addAction(action);
64 
65  QToolButton *button = new QToolButton(this);
66  mButtons.append(button);
67  button->setDefaultAction(action);
68  button->setCheckable(true);
69 
70  if(i == 0)
71  {
72  button->setObjectName("ubButtonGroupLeft");
73  }
74  else if (i == actions.size() - 1)
75  {
76  button->setObjectName("ubButtonGroupRight");
77  }

79  {
80  button->setObjectName("ubButtonGroupCenter");
81  }
82

85  horizontalLayout->addWidget(button);

119 void UBToolbarButtonGroup::selected(QAction *action)
120 {
121  foreach(QWidget *widget, action->associatedWidgets())
122  {
123  QToolButton *button = qobject_cast<QToolButton*>(widget);
124  if (button)
125  {
126  int i = 0;
127  foreach(QAction *eachAction, mActions)
128  {
129  if (eachAction == action)
130  {
131  setCurrentIndex(i);
132  emit activated(i);
133  break;
134  }
135  i++;
136  }
137  }
138  }

193 void UBToolbarButtonGroup::colorPaletteChanged()
194 {
195  bool isDarkBackground = UBSettings::settings()->isDarkBackground();
196 
197  QList<QColor> colors;
198 
199  if (UBDrawingController::drawingController()->stylusTool() == UBStylusTool::Pen
200  || UBDrawingController::drawingController()->stylusTool() == UBStylusTool::Line)
201  {
202  colors = UBSettings::settings()->penColors(isDarkBackground);
203  }
204  else if (UBDrawingController::drawingController()->stylusTool() == UBStylusTool::Marker)
205  {
206  colors = UBSettings::settings()->markerColors(isDarkBackground);
207  }
208
