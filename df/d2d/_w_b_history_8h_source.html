73 class WBHistoryItem
74 {
75  public:
76  WBHistoryItem() {}
77  WBHistoryItem(const QString &u,
78  const QDateTime &d = QDateTime(), const QString &t = QString())
79  : title(t), url(u), dateTime(d) {}
80 
81  inline bool operator==(const WBHistoryItem &other) const
82  { return other.title == title
83  && other.url == url && other.dateTime == dateTime; }
84 
85  // history is sorted in reverse
86  inline bool operator <(const WBHistoryItem &other) const
87  { return dateTime > other.dateTime; }
88 
89  QString title;
90  QString url;
91  QDateTime dateTime;
92 };
