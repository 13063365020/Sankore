552 void UBKeyButton::onPress()
553 {
554  if (keybt!=NULL)
555  {
556 #if defined(Q_WS_MACX)
557  int codeIndex = keyboard->nSpecialModifierIndex + (shifted())?1:(capsed() ? 2 : 0);
558 #else
559  int codeIndex = keyboard->nSpecialModifierIndex * 2 + shifted();
560 #endif
561  if (keyboard->nSpecialModifierIndex)
562  {
563  if (keybt->codes[codeIndex].empty())
564  {
565  sendUnicodeSymbol(keyboard->specialModifier);
566  sendUnicodeSymbol(keybt->codes[shifted()]);
567  }
568  else
569  {
570  sendUnicodeSymbol(keybt->codes[codeIndex]);
571  }
572 
573  keyboard->nSpecialModifierIndex = 0;
574  }
575  else
576  {
577 
578 #if defined(Q_WS_MACX)
579  int nSpecialModifierIndex;
580 
581  if (shifted())
582  nSpecialModifierIndex = keybt->modifierShift;
583  else if(capsed())
584  nSpecialModifierIndex = keybt->modifierCaps;
585  else
586  nSpecialModifierIndex = keybt->modifierNo;
587 #else
588  int nSpecialModifierIndex = shifted()? keybt->modifierShift : keybt->modifierNo;
589 #endif
590 
591 
592  if (nSpecialModifierIndex)
593  {
594  keyboard->nSpecialModifierIndex = nSpecialModifierIndex;
595  keyboard->specialModifier = keybt->codes[codeIndex];
596  }
597  else
598  {
599  sendUnicodeSymbol(keybt->codes[codeIndex]);
600  }
601  }
602  }
603 
604  if (keyboard->shift)
605  {
606  keyboard->shift = false;
607  keyboard->update();
608  }
609 }
