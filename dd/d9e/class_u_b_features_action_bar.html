event->ignore();

QWidget *dest = childAt(event->pos());

foreach (UBFeature curFeature, featuresList) {

} else if (dest == mpFavoriteBtn) {

event->setDropAction(Qt::MoveAction);
event->accept();
emit removeFromFavorite(fMimeData);
