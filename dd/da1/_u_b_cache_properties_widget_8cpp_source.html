74 
75  QPainter painter(this);
76  painter.setPen(mMaskColor);

357 void UBCachePropertiesWidget::updateCurrentCache()
358 {
359  bool isBoardMode = false;
360  // this widget can work only on Board mode
361  if( UBApplication::applicationController != NULL )
362  {
363  // if app controller is available, and current mode is Board, and no show desktop, than all ok, just process
364  if( UBApplication::applicationController->displayMode() == UBApplicationController::Board &&
365  !UBApplication::applicationController->isShowingDesktop())
366  isBoardMode = true;
367  }
368  // if app controller == null, than we do not know what mode now, so just process
369  else
370  isBoardMode = true;
371 
372  if(isBoardMode)
373  {
374  // Get the current page cache
375  QList<QGraphicsItem*> items = UBApplication::boardController->activeScene()->items();
376  foreach(QGraphicsItem* it, items)
377  {
378  if("Cache" == it->data(Qt::UserRole).toString())
379  {
380  setEnabled(true);
381  emit showTab(this);
382  mpCurrentCache = dynamic_cast<UBGraphicsCache*>(it);
383  if((NULL != mpCurrentCache) && (!mCaches.contains(mpCurrentCache)))
384  mCaches.append(mpCurrentCache);
385  else
386  return
387 
388  // Update the values of the cache properties
389  mpWidthSlider->setValue(mpCurrentCache->holeWidth());
390  mpHeightSlider->setValue(mpCurrentCache->holeHeight());
391  syncCacheColor(mpCurrentCache->maskColor());
392  mpPreviewWidget->setHoleSize(QSize(mpWidthSlider->value(), mpHeightSlider->value()));
393  mpCurrentCache->setMode(UBSettings::settings()->cacheMode->get().toInt());
394 
395  mActualShape = mpCurrentCache->maskshape();
396  switch(mActualShape)
397  {
398  case eMaskShape_Circle:
399  mpCircleButton->setChecked(true);
400  mpSquareButton->setChecked(false);
401  break;
402  case eMaskShap_Rectangle:
403  mpCircleButton->setChecked(false);
404  mpSquareButton->setChecked(true);
405  break;
406  }
407 
408 
409  mpPreviewWidget->setShape(mActualShape);
410  return;
411 
412  }
413  }
414  }
415 
416  // If we fall here, that means:
417  // 1 - that this page has no cache
418  // 2 - we are not in Board mode
419  // 3 - we are in Board mode, but show desktop (as really - Desktop mode)
420  emit hideTab(this);
421  mpCurrentCache = NULL;
422  setDisabled(true);
423 }
