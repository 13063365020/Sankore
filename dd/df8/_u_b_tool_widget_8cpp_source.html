131 bool UBToolWidget::eventFilter(QObject *obj, QEvent *event)
132 {
133  if (mShouldMoveWidget && obj == mWebView && event->type() == QEvent::MouseMove)
134  {
135  QMouseEvent *mouseMoveEvent = static_cast<QMouseEvent*>(event);
136 
137  if (mouseMoveEvent->buttons() & Qt::LeftButton)
138  {
139  move(pos() - mMousePressPos + mWebView->mapTo(this, mouseMoveEvent->pos()));
140 
141  event->accept();
142  return true;
143  }
144  }
145 
146  // standard event processing
147  return QObject::eventFilter(obj, event);
148 }

170 void UBToolWidget::mouseReleaseEvent(QMouseEvent *event)
171 {
172  mShouldMoveWidget = false;
173 
174  if (event->pos().x() >= 0 && event->pos().x() < sClosePixmap->width() && event->pos().y() >= 0 && event->pos().y() < sClosePixmap->height()) {
175  UBApplication::boardController->removeTool(this);
176  event->accept();
177  }
178  else if (mToolWidget->canBeContent() && event->pos().x() >= mContentMargin && event->pos().x() < mContentMargin + sUnpinPixmap->width() && event->pos().y() >= 0 && event->pos().y() < sUnpinPixmap->height()) {
179  UBApplication::boardController->moveToolWidgetToScene(this);
180  event->accept();
181  }
182  else
183  QWidget::mouseReleaseEvent(event); /* don't propgate to parent, the widget is deleted in UBApplication */
184 
185 }
