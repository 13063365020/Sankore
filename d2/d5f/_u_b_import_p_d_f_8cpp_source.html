63 QList<UBGraphicsItem*> UBImportPDF::import(const QUuid& uuid, const QString& filePath)
64 {
65  QList<UBGraphicsItem*> result;
66 
67  PDFRenderer *pdfRenderer = PDFRenderer::rendererForUuid(uuid, filePath, true); // renderer is automatically deleted when not used anymore
68 
69  if (!pdfRenderer->isValid())
70  {
71  UBApplication::showMessage(tr("PDF import failed."));
72  return result;
73  }
74  pdfRenderer->setDPI(this->dpi);
75 
76  int pdfPageCount = pdfRenderer->pageCount();
77 
78  for(int pdfPageNumber = 1; pdfPageNumber <= pdfPageCount; pdfPageNumber++)
79  {
80  UBApplication::showMessage(tr("Importing page %1 of %2").arg(pdfPageNumber).arg(pdfPageCount), true);
81  result << new UBGraphicsPDFItem(pdfRenderer, pdfPageNumber); // deleted by the scene
82  }
83  return result;
84 }
