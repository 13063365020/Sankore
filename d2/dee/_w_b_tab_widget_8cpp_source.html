183 void WBTabBar::mouseMoveEvent(QMouseEvent *event)
184 {
185  if (event->buttons() == Qt::LeftButton) {
186  int diffX = event->pos().x() - mDragStartPos.x();
187  int diffY = event->pos().y() - mDragStartPos.y();
188  if ((event->pos() - mDragStartPos).manhattanLength() > QApplication::startDragDistance()
189  && diffX < 3 && diffX > -3
190  && diffY < -10) {
191  QDrag *drag = new QDrag(this);
192  QMimeData *mimeData = new QMimeData;
193  QList<QUrl> urls;
194  int index = tabAt(event->pos());
195  QUrl url = tabData(index).toUrl();
196  urls.append(url);
197  mimeData->setUrls(urls);
198  mimeData->setText(tabText(index));
199  mimeData->setData(QLatin1String("action"), "tab-reordering");
200  drag->setMimeData(mimeData);
201  drag->exec();
202  }
203  }
204  QTabBar::mouseMoveEvent(event);
205 }

539 {

541  index = currentIndex();
542  if (index < 0 || index >= count())
543  return;
544 
545  bool hasFocus = false;
546  if (WBWebView *tab = webView(index))
547  {
548  hasFocus = tab->hasFocus();
549 
550  mRecentlyClosedTabsAction->setEnabled(true);
551  mRecentlyClosedTabs.prepend(tab->url());
552  if (mRecentlyClosedTabs.size() >= WBTabWidget::sRecentlyClosedTabsSize)
553  mRecentlyClosedTabs.removeLast();
554  }
555 
556  QWidget *lineEdit = mLineEdits->widget(index);
557  mLineEdits->removeWidget(lineEdit);
558  lineEdit->deleteLater();
559  QWidget *webView = widget(index);
560  removeTab(index);
561  webView->deleteLater();
562 
563  if (hasFocus && count() > 0)
564  currentWebView()->setFocus();
565 
566  if (count() == 0){
567  newTab();
568  emit currentChanged(0);
569  }
570 }

741 {

744  QDataStream stream(&sd, QIODevice::ReadOnly);
745  if (stream.atEnd())
746  return false;

751  stream >> v;
