62 //parse object's content and fill out vector with
63 //necessary decoders
64 std::vector <Decoder * > Filter::_getDecoders()
65 {
66  std::string streamHeader;
67  static std::string whitespacesAndDelimeters(" \t\f\v\n\r<<>>]/");
68  _objectWithStream->getHeader(streamHeader);
69  unsigned int filterPosition = streamHeader.find("/Filter");
70  std::vector <Decoder * > result;
71  unsigned int startOfDecoder = filterPosition + 1;
72  while(1)
73  {
74  startOfDecoder = streamHeader.find("/", startOfDecoder);
75  if((int)startOfDecoder == -1)
76  break;
77  else
78  ++startOfDecoder;
79  unsigned int endOfDecoder = streamHeader.find_first_of(whitespacesAndDelimeters, startOfDecoder);
80  if((int)endOfDecoder == -1)
81  break;
82  std::map<std::string, Decoder *>::iterator foundDecoder =
83  _allDecoders.find(streamHeader.substr(startOfDecoder, endOfDecoder - startOfDecoder));
84  if(foundDecoder == _allDecoders.end())
85  break;
86  Decoder * decoder = foundDecoder->second;
87  decoder->initialize(_objectWithStream);
88  result.push_back(decoder);
89  }
90  return result;
