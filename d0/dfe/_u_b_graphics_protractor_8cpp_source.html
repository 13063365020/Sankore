181 void UBGraphicsProtractor::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
182 {
183  QPointF currentPoint = event->pos();
184  QLineF startLine(rect().center(), mPreviousMousePos);
185  QLineF currentLine(rect().center(), currentPoint);
186  qreal angle = startLine.angleTo(currentLine);
187  qreal scaleFactor = currentLine.length()/startLine.length();
188 
189  switch (mCurrentTool)
190  {
191  case Rotate :
192  prepareGeometryChange();
193  mStartAngle = mStartAngle + angle;
194  setStartAngle(mStartAngle * 16);
195  mPreviousMousePos = currentPoint;
196  break;
197 
198  case Resize :
199  prepareGeometryChange();
200  translate(rect().center().x(), rect().center().y());
201  scale(scaleFactor, scaleFactor);
202  translate(-rect().center().x(), -rect().center().y());
203  mScaleFactor *= scaleFactor;
204  break;
205 
206  case MoveMarker :
207 
208  mCurrentAngle += angle;
209  if ((int)mCurrentAngle % 360 > 270)
210  mCurrentAngle = 0;
211  else if ((int)mCurrentAngle % 360 >= 180)
212  mCurrentAngle = 180;
213 
214  mPreviousMousePos = currentPoint;
215  update();
216  break;
217 
218  case Move :
219  QGraphicsEllipseItem::mouseMoveEvent(event);
220  break;
221 
222  default :
223  break;
224  }
225 
226  if (mCurrentTool != Move)
227  event->accept();
228 
229 }

369 void UBGraphicsProtractor::paintGraduations(QPainter *painter)
370 {
371  painter->save();
372 
373  const int tenDegreeGraduationLength = 15;
374  const int fiveDegreeGraduationLength = 10;
375  const int oneDegreeGraduationLength = 5;
376 
377  QFont font1 = painter->font();
378 #ifdef Q_WS_MAC
379  font1.setPointSizeF(font1.pointSizeF() - 3);
380 #endif
381  QFontMetricsF fm1(font1);
382 
383  //Font for internal arc
384  QFont font2 = painter->font();
385  font2.setPointSizeF(font1.pointSizeF()/1.5);
386  QFontMetricsF fm2(font2);
387 
388  qreal rad = radius();
389 
390  QPointF center = rect().center();
391  painter->drawArc(QRectF(center.x() - rad/2, center.y() - rad/2, rad, rad), mStartAngle*16, mSpan*16);
392 
393  for (int angle = 1; angle < mSpan; angle++)
394  {
395  int graduationLength = (0 == angle % 10) ? tenDegreeGraduationLength : ((0 == angle % 5) ? fiveDegreeGraduationLength : oneDegreeGraduationLength);
396  qreal co = cos(((qreal)angle + mStartAngle) * PI/180);
397  qreal si = sin(((qreal)angle + mStartAngle) * PI/180);
398  if (0 == angle % 90)
399  painter->drawLine(QLineF(QPointF(center.x(), center.y()), QPointF(center.x() + co*tenDegreeGraduationLength, center.y() - si*tenDegreeGraduationLength)));
400 
401  //external arc
402  painter->drawLine(QLineF(QPointF(center.x()+ rad*co, center.y() - rad*si),
403  QPointF(center.x()+ (rad - graduationLength)*co, center.y() - (rad - graduationLength)*si)));
404  //internal arc
405  painter->drawLine(QLineF(QPointF(center.x()+ rad/2*co, center.y() - rad/2*si),
406  QPointF(center.x()+ (rad/2 + graduationLength)*co,
407  center.y() - (rad/2 + graduationLength)*si)));
408 
409  if (0 == angle % 10)
410  {
411  //external arc
412  painter->setFont(font1);
413  QString grad = QString("%1").arg((int)(angle));
414  QString grad2 = QString("%1").arg((int)(mSpan - angle));
415 
416  painter->drawText(QRectF(center.x() + (rad - graduationLength*1.5)*co - fm1.width(grad)/2,
417  center.y() - (rad - graduationLength*1.5)*si - fm1.height()/2,
418  fm1.width(grad), fm1.height()), Qt::AlignTop, grad);
419 
420  //internal arc
421  painter->setFont(font2);
422  painter->drawText(QRectF(center.x() + (rad/2 + graduationLength*1.5)*co - fm2.width(grad2)/2,
423  center.y() - (rad/2 + graduationLength*1.5)*si - fm2.height()/2,
424  fm2.width(grad2), fm2.height()), Qt::AlignTop, grad2);
425  painter->setFont(font1);
426 
427  }
428  }
429 
430  painter->restore();
431 }

541 UBGraphicsProtractor::Tool UBGraphicsProtractor::toolFromPos(QPointF pos)
542 {
543  pos = pos - rect().center();
544  QLineF line(QPointF(0,0), pos);
545  QTransform t;
546  t.rotate(mStartAngle);
547  QPointF p1 = t.map(pos);
548  t.rotate(mCurrentAngle);
549  QPointF p2 = t.map(pos);
550 
551  if (resizeButtonRect().contains(p1))
552  return Resize;
553  else if (closeButtonRect().contains(p1))
554  return Close;
555  else if (resetButtonRect().contains(p1))
556  return Reset;
557  else if (rotateButtonRect().contains(p1))
558  return Rotate;
559  else if (markerButtonRect().contains(p2))
560  return MoveMarker;
561  else if (line.length() <= radius())
562  return Move;
563  else
564  return None;
565 }
