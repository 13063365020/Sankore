110 bool WBWebPage::acceptNavigationRequest(QWebFrame *frame, const QNetworkRequest &request, NavigationType type)
111 {
112  // ctrl open in new tab
113  // ctrl-shift open in new tab and select
114  // ctrl-alt open in new window
115  if (type == QWebPage::NavigationTypeLinkClicked
116  && (mKeyboardModifiers & Qt::ControlModifier
117  || mPressedButtons == Qt::MidButton))
118  {
119  WBWebView *webView;
120 
121  bool selectNewTab = (mKeyboardModifiers & Qt::ShiftModifier);
122  webView = mainWindow()->tabWidget()->newTab(selectNewTab);
123 
124  webView->load(request);
125  mKeyboardModifiers = Qt::NoModifier;
126  mPressedButtons = Qt::NoButton;
127 
128  return false;
129  }
130 
131  if (frame == mainFrame())
132  {
133  mLoadingUrl = request.url();
134  emit loadingUrl(mLoadingUrl);
135  }
136 
137  return QWebPage::acceptNavigationRequest(frame, request, type);
138 }

162 void WBWebPage::handleUnsupportedContent(QNetworkReply *reply)
163 {
164  if(reply->url().scheme() == "mailto")
165  {
166  bool result = QDesktopServices::openUrl(reply->url());
167  if (result)
168  return;
169  }
170 
171  QString contentType = reply->header(QNetworkRequest::ContentTypeHeader).toString();
172  bool isPDF = (contentType == "application/pdf");
173 
174  // Delete this big "if (isPDF)" block to get the pdf directly inside the browser
175  if (isPDF)
176  {
177  QMessageBox messageBox(mainWindow());
178  messageBox.setText(tr("Download PDF Document: would you prefer to download the PDF file or add it to the current Sankore document?"));
179 
180  messageBox.addButton(tr("Download"), QMessageBox::AcceptRole);
181  QAbstractButton *addButton = messageBox.addButton(tr("Add to Current Document"), QMessageBox::AcceptRole);
182 
183  messageBox.exec();
184  if (messageBox.clickedButton() == addButton)
185  {
186  UBApplication::applicationController->showBoard();
187  UBApplication::boardController->downloadURL(reply->request().url());
188  return;
189  }
190  else
191  {
192  isPDF = false;
193  }
194  }
195 
196  if (!isPDF && reply->error() == QNetworkReply::NoError)
197  {
198  if(contentType == "application/widget")
199  WBBrowserWindow::downloadManager()->handleUnsupportedContent(reply,false, UBSettings::settings()->userGipLibraryDirectory());
200  else
201  WBBrowserWindow::downloadManager()->handleUnsupportedContent(reply);
202  return;
203  }
204 
205  QFile file;
206  file.setFileName(isPDF ? QLatin1String(":/webbrowser/object-wrapper.html") : QLatin1String(":/webbrowser/notfound.html"));
207 
208  bool isOpened = file.open(QIODevice::ReadOnly);
209  Q_ASSERT(isOpened);
210  QString html;
211  if (isPDF)
212  {
213  html = QString(QLatin1String(file.readAll()))
214  .arg(tr("PDF"))
215  .arg("application/x-ub-pdf")
216  .arg(reply->url().toString());
217  }
218  else
219  {
220  QString title = tr("Error loading page: %1").arg(reply->url().toString());
221  html = QString(QLatin1String(file.readAll()))
222  .arg(title)
223  .arg(reply->errorString())
224  .arg(reply->url().toString());
225  }
226 
227  QList<QWebFrame*> frames;
228  frames.append(mainFrame());
229  while (!frames.isEmpty())
230  {
231  QWebFrame *frame = frames.takeFirst();
232  if (frame->url() == reply->url())
233  {
234  frame->setHtml(html, reply->url());
235  return;
236  }
237  QList<QWebFrame *> children = frame->childFrames();
238  foreach(QWebFrame *frame, children)
239  frames.append(frame);
240  }
241 
242  if (mLoadingUrl == reply->url())
243  {
244  mainFrame()->setHtml(html, reply->url());
245  }
246 }
