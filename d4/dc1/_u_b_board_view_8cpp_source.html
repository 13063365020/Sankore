350 void UBBoardView::tabletEvent (QTabletEvent * event)
351 {
352  if (!mUseHighResTabletEvent) {
353  event->setAccepted (false);
354  return;
355  }
356 
357  UBDrawingController *dc = UBDrawingController::drawingController ();
358 
359  QPointF tabletPos = event->pos();
360  UBStylusTool::Enum currentTool = (UBStylusTool::Enum)dc->stylusTool ();
361 
362  if (event->type () == QEvent::TabletPress || event->type () == QEvent::TabletEnterProximity) {
363  if (event->pointerType () == QTabletEvent::Eraser) {
364  dc->setStylusTool (UBStylusTool::Eraser);
365  mUsingTabletEraser = true;
366  }
367  else {
368  if (mUsingTabletEraser && currentTool == UBStylusTool::Eraser)
369  dc->setStylusTool (dc->latestDrawingTool ());
370 
371  mUsingTabletEraser = false;
372  }
373  }
374 
375  // if event are not Pen events, we drop the tablet stuff and route everything through mouse event
376  if (currentTool != UBStylusTool::Pen && currentTool != UBStylusTool::Line && currentTool != UBStylusTool::Marker && !mMarkerPressureSensitive){
377  event->setAccepted (false);
378  return;
379  }
380 
381  QPointF scenePos = viewportTransform ().inverted ().map (tabletPos);
382 
383  qreal pressure = 1.0;
384  if (((currentTool == UBStylusTool::Pen || currentTool == UBStylusTool::Line) && mPenPressureSensitive) || (currentTool == UBStylusTool::Marker && mMarkerPressureSensitive))
385  pressure = event->pressure ();
386 
387 
388  bool acceptEvent = true;
389 
390 #ifdef Q_WS_MAC
391  //Work around #1388. After selecting annotation tool in desktop mode, annotation view appears on top when
392  //using Mac OS X. In this case tablet event should send mouse event so as to let user interact with
393  //stylus palette.
394  Q_ASSERT(UBApplication::applicationController->uninotesController());
395  if (UBApplication::applicationController->uninotesController()->drawingView() == this) {
396  if (UBApplication::applicationController->uninotesController()->desktopPalettePath().contains(event->pos())) {
397  acceptEvent = false;
398  }
399  }
400 #endif
401 
402  switch (event->type ()) {
403  case QEvent::TabletPress: {
404  mTabletStylusIsPressed = true;
405  scene()->inputDevicePress (scenePos, pressure);
406 
407  break;
408  }
409  case QEvent::TabletMove: {
410  if (mTabletStylusIsPressed)
411  scene ()->inputDeviceMove (scenePos, pressure);
412 
413  acceptEvent = false; // rerouted to mouse move
414 
415  break;
416 
417  }
418  case QEvent::TabletRelease: {
419  UBStylusTool::Enum currentTool = (UBStylusTool::Enum)dc->stylusTool ();
420  scene ()->setToolCursor (currentTool);
421  setToolCursor (currentTool);
422 
423  scene ()->inputDeviceRelease ();
424 
425  mPendingStylusReleaseEvent = false;
426 
427  mTabletStylusIsPressed = false;
428  mMouseButtonIsPressed = false;
429 
430  break;
431  }
432  default: {
433  //NOOP - avoid compiler warning
434  }
435  }
436 
437  // ignore mouse press and mouse move tablet event so that it is rerouted to mouse events,
438  // documented in QTabletEvent Class Reference:
439  /* The event handler QWidget::tabletEvent() receives all three types of tablet events.
440  Qt will first send a tabletEvent then, if it is not accepted, it will send a mouse event. */
441  //
442  // This is a workaround to the fact that tablet event are not delivered to child widget (like palettes)
443  //
444 
445  event->setAccepted (acceptEvent);
446 
447 }

488 void UBBoardView::handleItemsSelection(QGraphicsItem *item)
489 {
490 // we need to select new pressed itemOnBoard and deselect all other items.
491 // the trouble is in:
492 // some items can has parents (groupped items or strokes, or strokes in groups).
493 // some items is already selected and we don't need to reselect them
494 //
495 // item selection managed by QGraphicsView::mousePressEvent(). It should be called later.
496 
497  if (item)
498  {
499  // item has group as first parent - it is any item or UBGraphicsStrokesGroup.
500  if(item->parentItem() && UBGraphicsGroupContainerItem::Type == movingItem->parentItem()->type())
501  return;
502 
503  // delegate buttons shouldn't selected
504  if (DelegateButton::Type == item->type())
505  return;
506 
507  // click on svg items (images on Frame) shouldn't change selection.
508  if (QGraphicsSvgItem::Type == item->type())
509  return;
510 
511  // Delegate frame shouldn't selected
512  if (UBGraphicsDelegateFrame::Type == item->type())
513  return;
514 
515 
516  // if we need to uwe multiple selection - we shouldn't deselect other items.
517  if (!isMultipleSelectionEnabled())
518  {
519  // here we need to determine what item is pressed. We should work
520  // only with UB items.
521  if ((UBGraphicsItemType::UserTypesCount > item->type()) && (item->type() > QGraphicsItem::UserType))
522  {
523  // if Item can be selected at mouse press - then we need to deselect all other items.
524  foreach(QGraphicsItem *iter_item, scene()->selectedItems())
525  {
526  if (iter_item != item)
527  {
528  iter_item->setSelected(false);
529  }
530  }
531  }
532  }
533  }
534 }

536 bool UBBoardView::itemShouldReceiveMousePressEvent(QGraphicsItem *item)
537 {
538 /*
539 Some items should receive mouse press events averytime,
540 some items should receive that events when they are selected,
541 some items shouldn't receive mouse press events at mouse press, but should receive them at mouse release (suspended mouse press event)
542 
543 Here we determines cases when items should to get mouse press event at pressing on mouse.
544 */
545 
546  if (!item)
547  return true;
548 
549  // for now background objects is not interactable, but it can be deprecated for some items in the future.
550  if (item == scene()->backgroundObject())
551  return false;
552 
553  // some behavior depends on current tool.
554  UBStylusTool::Enum currentTool = (UBStylusTool::Enum)UBDrawingController::drawingController()->stylusTool();
555 
556  qDebug() << item->type();
557 
558  switch(item->type())
559  {
560  case UBGraphicsProtractor::Type:
561  case UBGraphicsRuler::Type:
562  case UBGraphicsTriangle::Type:
563  case UBGraphicsCompass::Type:
564  case UBGraphicsCache::Type:
565  case UBGraphicsAristo::Type:
566  return true;
567 
568  case UBGraphicsDelegateFrame::Type:
569  if (currentTool == UBStylusTool::Play)
570  return false;
571  return true;
572  case UBGraphicsPixmapItem::Type:
573  case UBGraphicsSvgItem::Type:
574  if (currentTool == UBStylusTool::Play)
575  return true;
576  if (item->isSelected())
577  return true;
578  else
579  return false;
580 
581  case DelegateButton::Type:
582  return true;
583 
584  case UBGraphicsMediaItem::Type:
585  return false;
586 
587  case UBGraphicsTextItem::Type:
588  if (currentTool == UBStylusTool::Play)
589  return true;
590  if ((currentTool == UBStylusTool::Selector) && item->isSelected())
591  return true;
592  if ((currentTool == UBStylusTool::Selector) && item->parentItem() && item->parentItem()->isSelected())
593  return true;
594  if (currentTool != UBStylusTool::Selector)
595  return false;
596  break;
597 
598  case UBGraphicsItemType::StrokeItemType:
599  if (currentTool == UBStylusTool::Play)
600  return true;
601  break;
602  // Groups shouldn't reacts on any presses and moves for Play tool.
603  case UBGraphicsGroupContainerItem::Type:
604  if(currentTool == UBStylusTool::Play)
605  {
606  movingItem = NULL;
607  return true;
608  }
609  return false;
610  break;
611 
612 
613  case QGraphicsWebView::Type:
614  return true;
615 
616  case UBGraphicsWidgetItem::Type:
617  if (currentTool == UBStylusTool::Selector && item->parentItem() && item->parentItem()->isSelected())
618  return true;
619  if (currentTool == UBStylusTool::Selector && item->isSelected())
620  return true;
621  if (currentTool == UBStylusTool::Play)
622  return true;
623  return false;
624  break;
625 
626  }
627 
628  return !isUBItem(item); // standard behavior of QGraphicsScene for not UB items. UB items should be managed upper.
629 }

743 QGraphicsItem* UBBoardView::determineItemToMove(QGraphicsItem *item)
744 {
745  if(item)
746  {
747  UBStylusTool::Enum currentTool = (UBStylusTool::Enum)UBDrawingController::drawingController()->stylusTool();
748 
749  //W3C widgets should take mouse move events from play tool.
750  if ((UBStylusTool::Play == currentTool) && (UBGraphicsWidgetItem::Type == item->type()))
751  return item;
752 
753  // if item is in group
754  if(item->parentItem() && UBGraphicsGroupContainerItem::Type == item->parentItem()->type())
755  {
756  // play tool should move groups by any element
757  if (UBStylusTool::Play == currentTool && item->parentItem()->isSelected())
758  return item->parentItem();
759 
760  // groups should should be moved instead of strokes groups
761  if (UBGraphicsStrokesGroup::Type == item->type())
762  return item->parentItem();
763 
764  // selected groups should be moved by moving any element
765  if (item->parentItem()->isSelected())
766  return item;
767 
768  if (item->isSelected())
769  return NULL;
770 
771  return item->parentItem();
772  }
773 
774  // items like polygons placed in two groups nested, so we need to recursive call.
775  if(item->parentItem() && UBGraphicsStrokesGroup::Type == item->parentItem()->type())
776  return determineItemToMove(item->parentItem());
777  }

823 void UBBoardView::handleItemMouseMove(QMouseEvent *event)
824 {
825  // determine item to move (maybee we need to move group of item or his parent.
826  movingItem = determineItemToMove(movingItem);
827 
828  // items should be moved not every mouse move.
829  if (movingItem && itemShouldBeMoved(movingItem) && (mMouseButtonIsPressed || mTabletStylusIsPressed))
830  {
831  QPointF scenePos = mapToScene(event->pos());
832  QPointF newPos = movingItem->pos() + scenePos - mLastPressedMousePos;
833  movingItem->setPos(newPos);
834  mLastPressedMousePos = scenePos;
835  mWidgetMoved = true;
836  event->accept();
837  }
838  else
839  {
840  QPointF posBeforeMove;
841  QPointF posAfterMove;
842 
843  if (movingItem)
844  posBeforeMove = movingItem->pos();
845 
846  QGraphicsView::mouseMoveEvent (event);
847 
848  if (movingItem)
849  posAfterMove = movingItem->pos();
850 
851  mWidgetMoved = ((posAfterMove-posBeforeMove).manhattanLength() != 0);
852 
853  // a cludge for terminate moving of w3c widgets.
854  // in some cases w3c widgets catches mouse move and doesn't sends that events to web page,
855  // at simple - in google map widget - mouse move events doesn't comes to web page from rectangle of wearch bar on bottom right corner of widget.
856  if (mWidgetMoved && UBGraphicsW3CWidgetItem::Type == movingItem->type())
857  movingItem->setPos(posBeforeMove);
858  }
