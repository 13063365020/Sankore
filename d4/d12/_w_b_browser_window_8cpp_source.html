297 QUrl WBBrowserWindow::guessUrlFromString(const QString &string)
298 {
299  QString urlStr = string.trimmed();
300  QRegExp test(QLatin1String("^[a-zA-Z]+\\:.*"));
301 
302  // Check if it looks like a qualified URL. Try parsing it and see.
303  bool hasSchema = test.exactMatch(urlStr);
304  if (hasSchema)
305  {
306  int dotCount = urlStr.count(".");
307 
308  if (dotCount == 0 && !urlStr.contains(".com"))
309  {
310  urlStr += ".com";
311  }
312 
313  QUrl url = QUrl::fromEncoded(urlStr.toUtf8(), QUrl::TolerantMode);
314  if (url.isValid())
315  {
316  return url;
317  }
318  }
319 
320  // Might be a file.
321  if (QFile::exists(urlStr))
322  {
323  QFileInfo info(urlStr);
324  return QUrl::fromLocalFile(info.absoluteFilePath());
325  }
326 
327  // Might be a shorturl - try to detect the schema.
328  if (!hasSchema)
329  {
330  QString schema = "http";
331 
332  QString guessed = schema + "://" + urlStr;
333 
334  int dotCount = guessed.count(".");
335 
336  if (dotCount == 0 && !urlStr.contains(".com"))
337  {
338  guessed += ".com";
339  }
340 
341  QUrl url = QUrl::fromEncoded(guessed.toUtf8(), QUrl::TolerantMode);
342 
343  if (url.isValid())
344  return url;
345  }
346 
347  // Fall back to QUrl's own tolerant parser.
348  QUrl url = QUrl::fromEncoded(string.toUtf8(), QUrl::TolerantMode);
349 
350  // finally for cases where the user just types in a hostname add http
351  if (url.scheme().isEmpty())
352  url = QUrl::fromEncoded("http://" + string.toUtf8(), QUrl::TolerantMode);
353 
354  return url;
355 }

570 void WBBrowserWindow::aboutToShowBackMenu()
571 {
572  mHistoryBackMenu->clear();
573  if (!currentTabWebView())
574  return;
575  QWebHistory *history = currentTabWebView()->history();
576 
577  int historyCount = history->count();
578  int historyLimit = history->backItems(historyCount).count() - UBSettings::settings()->historyLimit->get().toReal();
579  if (historyLimit < 0)
580  historyLimit = 0;
581 
582  for (int i = history->backItems(historyCount).count() - 1; i >= historyLimit; --i)
583  {
584  QWebHistoryItem item = history->backItems(historyCount).at(i);
585 
586  QAction *action = new QAction(this);
587  action->setData(-1*(historyCount-i-1));
588 
589  if (!QWebSettings::iconForUrl(item.originalUrl()).isNull())
590  action->setIcon(item.icon());
591  action->setText(item.title().isEmpty() ? item.url().toString() : item.title());
592  mHistoryBackMenu->addAction(action);
593  }
594 
595  mHistoryBackMenu->addSeparator();
596 
597  QAction *action = new QAction(this);
598  action->setData("clear");
599  action->setText("Clear history");
600  mHistoryBackMenu->addAction(action);
601 
602 }

637 void WBBrowserWindow::openActionUrl(QAction *action)
638 {
639  QWebHistory *history = currentTabWebView()->history();
640 
641  if (action->data() == "clear")
642  {
643  history->clear();
644  return;
645  }
646 
647  int offset = action->data().toInt();
648  if (offset < 0)
649  history->goToItem(history->backItems(-1*offset).first());
650  else if (offset > 0)
651  history->goToItem(history->forwardItems(history->count() - offset + 1).back());
652  }
