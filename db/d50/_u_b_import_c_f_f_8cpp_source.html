136 QString UBImportCFF::expandFileToDir(const QFile& pZipFile, const QString& pDir)
137 {
138  QuaZip zip(pZipFile.fileName());
139 
140  if(!zip.open(QuaZip::mdUnzip)) {
141  qWarning() << "Import failed. Cause zip.open(): " << zip.getZipError();
142  return "";
143  }
144 
145  zip.setFileNameCodec("UTF-8");
146  QuaZipFileInfo info;
147  QuaZipFile file(&zip);
148 
149  //create unique cff document root fodler
150  //use current date/time and temp number for folder name
151  QString documentRootFolder;
152  int tmpNumber = 0;
153  QDir rootDir;
154  while (true) {
155  QString tempPath = QString("%1/sank%2.%3")
156  .arg(pDir)
157  .arg(QDateTime::currentDateTime().toString("dd_MM_yyyy_HH-mm"))
158  .arg(tmpNumber);
159  if (!rootDir.exists(tempPath)) {
160  documentRootFolder = tempPath;
161  break;
162  }
163  tmpNumber++;
164  if (tmpNumber == 100000) {
165  qWarning() << "Import failed. Failed to create temporary directory for iwb file";
166  return "";
167  }
168  }
169  if (!rootDir.mkdir(documentRootFolder)) {
170  qWarning() << "Import failed. Couse: failed to create temp folder for cff package";
171  }
172 
173  QFile out;
174  char c;
175  for(bool more=zip.goToFirstFile(); more; more=zip.goToNextFile()) {
176  if(!zip.getCurrentFileInfo(&info)) {
177  //TOD UB 4.3 O display error to user or use crash reporter
178  qWarning() << "Import failed. Cause: getCurrentFileInfo(): " << zip.getZipError();
179  return "";
180  }
181 // if(!file.open(QIODevice::ReadOnly)) {
182 // qWarning() << "Import failed. Cause: file.open(): " << zip.getZipError();
183 // return "";
184 // }
185  file.open(QIODevice::ReadOnly);
186  if(file.getZipError()!= UNZ_OK) {
187  qWarning() << "Import failed. Cause: file.getFileName(): " << zip.getZipError();
188  return "";
189  }
190 
191  QString newFileName = documentRootFolder + "/" + file.getActualFileName();
192 
193  QFileInfo newFileInfo(newFileName);
194  rootDir.mkpath(newFileInfo.absolutePath());
195 
196  out.setFileName(newFileName);
197  out.open(QIODevice::WriteOnly);
198 
199  while(file.getChar(&c))
200  out.putChar(c);
201 
202  out.close();
203 
204  if(file.getZipError()!=UNZ_OK) {
205  qWarning() << "Import failed. Cause: " << zip.getZipError();
206  return "";
207  }
208  if(!file.atEnd()) {
209  qWarning() << "Import failed. Cause: read all but not EOF";
210  return "";
211  }
212 
213  file.close();
214 
215  if(file.getZipError()!=UNZ_OK) {
216  qWarning() << "Import failed. Cause: file.close(): " << file.getZipError();
217  return "";
218  }
219  }
220 
221  zip.close();
222 
223  if(zip.getZipError()!=UNZ_OK) {
224  qWarning() << "Import failed. Cause: zip.close(): " << zip.getZipError();
225  return "";
226  }
227 
228  return documentRootFolder;
229 }
