265 void WBUrlIconLabel::mouseMoveEvent(QMouseEvent *event)
266 {
267  if (event->buttons() == Qt::LeftButton
268  && (event->pos() - m_dragStartPos).manhattanLength() > QApplication::startDragDistance()
269  && m_webView) {
270  QDrag *drag = new QDrag(this);
271  QMimeData *mimeData = new QMimeData;
272  mimeData->setText(QString::fromUtf8(m_webView->url().toEncoded()));
273  QList<QUrl> urls;
274  urls.append(m_webView->url());
275  mimeData->setUrls(urls);
276  drag->setMimeData(mimeData);
277  drag->exec();
278  }
279 }
