101 void UBZoomPalette::refreshPalette()
102 {
103  qreal currentZoomFactor = mBoardController->currentZoom();
104  bool showAsNoZoom = (currentZoomFactor > 0.9 && currentZoomFactor < 1.1);
105  if (showAsNoZoom)
106  {
107  currentZoomFactor = 1;
108  }
109  QString stringFactor = tr("%1 x").arg(currentZoomFactor, 0, 'f', 1);
110 
111  mCurrentZoomButton->setText(stringFactor);
112  adjustSizeAndPosition(true,false);
113  if (showAsNoZoom)
114  {
115  QTimer::singleShot(500, this, SLOT(hide()));
116  }
117  else
118  {
119  show();
