162 void WBHistoryManager::checkForExpired()
163 {
164  if (m_historyLimit < 0 || m_history.isEmpty())
165  return;
166 
167  QDateTime now = QDateTime::currentDateTime();
168  int nextTimeout = 0;
169 
170  while (!m_history.isEmpty())
171  {
172  QDateTime checkForExpired = m_history.last().dateTime;
173  checkForExpired.setDate(checkForExpired.date().addDays(m_historyLimit));
174  if (now.daysTo(checkForExpired) > 7)
175  {
176  // check at most in a week to prevent int overflows on the timer
177  nextTimeout = 7 * 86400;
178  }
179  else
180  {
181  nextTimeout = now.secsTo(checkForExpired);
182  }
183  if (nextTimeout > 0)
184  break;
185  WBHistoryItem item = m_history.takeLast();
186  // remove from saved file also
187  m_lastSavedUrl = QString();
188  emit entryRemoved(item);
189  }
190 
191  if (nextTimeout > 0)
192  m_expiredTimer.start(nextTimeout * 1000);
193 }

254 void WBHistoryManager::load()
255 {
256  loadSettings();
257 
258  QFile historyFile(UBSettings::userDataDirectory() + QLatin1String("/history"));
259  if (!historyFile.exists())
260  return;
261  if (!historyFile.open(QFile::ReadOnly))
262  {
263  qWarning() << "Unable to open history file" << historyFile.fileName();
264  return;
265  }
266 
267  QList<WBHistoryItem> list;
268  QDataStream in(&historyFile);
269  // Double check that the history file is sorted as it is read in
270  bool needToSort = false;
271  WBHistoryItem lastInsertedItem;
272  QByteArray data;
273  QDataStream stream;
274  QBuffer buffer;
275  stream.setDevice(&buffer);
276  while (!historyFile.atEnd())
277  {
278  in >> data;
279  buffer.close();
280  buffer.setBuffer(&data);
281  buffer.open(QIODevice::ReadOnly);
282  quint32 ver;
283  stream >> ver;
284  if (ver != HISTORY_VERSION)
285  continue;
286  WBHistoryItem item;
287  stream >> item.url;
288  stream >> item.dateTime;
289  stream >> item.title;
290 
291  if (!item.dateTime.isValid())
292  continue;
293 
294  if (item == lastInsertedItem)
295  {
296  if (lastInsertedItem.title.isEmpty() && !list.isEmpty())
297  list[0].title = item.title;
298  continue;
299  }
300 
301  if (!needToSort && !list.isEmpty() && lastInsertedItem < item)
302  needToSort = true;
303 
304  list.prepend(item);
305  lastInsertedItem = item;
306  }
307  if (needToSort)
308  qSort(list.begin(), list.end());
309 
310  setHistory(list, true);
311 
312  // If we had to sort re-write the whole history sorted
313  if (needToSort)
314  {
315  m_lastSavedUrl = QString();
316  m_saveTimer->changeOccurred();
317  }
318 }

320 void WBHistoryManager::save()
321 {
322  QSettings settings;
323  settings.beginGroup(QLatin1String("history"));
324  settings.setValue(QLatin1String("historyLimit"), m_historyLimit);
325 
326  bool saveAll = m_lastSavedUrl.isEmpty();
327  int first = m_history.count() - 1;
328  if (!saveAll)
329  {
330  // find the first one to save
331  for (int i = 0; i < m_history.count(); ++i)
332  {
333  if (m_history.at(i).url == m_lastSavedUrl)
334  {
335  first = i - 1;
336  break;
337  }
338  }
339  }
340  if (first == m_history.count() - 1)
341  saveAll = true;
342 
343  QString directory = UBSettings::userDataDirectory();
344  if (directory.isEmpty())
345  directory = QDir::homePath() + QLatin1String("/.") + QCoreApplication::applicationName();
346  if (!QFile::exists(directory))
347  {
348  QDir dir;
349  dir.mkpath(directory);
350  }
351 
352  QFile historyFile(directory + QLatin1String("/history"));
353  // When saving everything use a temporary file to prevent possible data loss.
354  QTemporaryFile tempFile;
355  tempFile.setAutoRemove(false);
356  bool open = false;
357  if (saveAll)
358  {
359  open = tempFile.open();
360  } else {
361  open = historyFile.open(QFile::Append);
362  }
363 
364  if (!open)
365  {
366  qWarning() << "Unable to open history file for saving"
367  << (saveAll ? tempFile.fileName() : historyFile.fileName());
368  return;
369  }
370 
371  QDataStream out(saveAll ? &tempFile : &historyFile);
372  for (int i = first; i >= 0; --i)
373  {
374  QByteArray data;
375  QDataStream stream(&data, QIODevice::WriteOnly);
376  WBHistoryItem item = m_history.at(i);
377  stream << HISTORY_VERSION << item.url << item.dateTime << item.title;
378  out << data;
379  }
380  tempFile.close();
381 
382  if (saveAll)
383  {
384  if (historyFile.exists() && !historyFile.remove())
385  qWarning() << "History: error removing old history." << historyFile.errorString();
386  if (!tempFile.rename(historyFile.fileName()))
387  qWarning() << "History: error moving new history over old." << tempFile.errorString() << historyFile.fileName();
388  }
389  m_lastSavedUrl = m_history.value(0).url;
390 }

439 QVariant WBHistoryModel::data(const QModelIndex &index, int role) const
440 {
441  QList<WBHistoryItem> lst = m_history->history();
442  if (index.row() < 0 || index.row() >= lst.size())
443  return QVariant();
444 
445  const WBHistoryItem &item = lst.at(index.row());
446  switch (role) {
447  case DateTimeRole:
448  return item.dateTime;
449  case DateRole:
450  return item.dateTime.date();
451  case UrlRole:
452  return QUrl(item.url);
453  case UrlStringRole:
454  return item.url;
455  case Qt::DisplayRole:
456  case Qt::EditRole: {
457  switch (index.column()) {
458  case 0:
459  // when there is no title try to generate one from the url
460  if (item.title.isEmpty()) {
461  QString page = QFileInfo(QUrl(item.url).path()).fileName();
462  if (!page.isEmpty())
463  return page;
464  return item.url;
465  }
466  return item.title;
467  case 1:
468  return item.url;
469  }
470  }
471  case Qt::DecorationRole:
472  if (index.column() == 0) {
473 // return BrowserApplication::instance()->icon(item.url);
474  }
475  }
476  return QVariant();
477 }

530 int WBHistoryMenuModel::rowCount(const QModelIndex &parent) const
531 {
532  if (parent.column() > 0)
533  return 0;
534 
535  if (!parent.isValid())
536  {
537  int folders = sourceModel()->rowCount();
538  int bumpedItems = bumpedRows();

541  --folders;
542  return bumpedItems + folders;
543  }
544 
545  if (parent.internalId() == -1)
546  {

553  if (idx == sourceModel()->index(0, 0))

566 QModelIndex WBHistoryMenuModel::mapToSource(const QModelIndex &proxyIndex) const
567 {

569  return QModelIndex();
570 
571  if (proxyIndex.internalId() == -1)
572  {
573  int bumpedItems = bumpedRows();

576  if (bumpedItems <= MOVEDROWS && bumpedItems == sourceModel()->rowCount(m_treeModel->index(0, 0)))

586 QModelIndex WBHistoryMenuModel::index(int row, int column, const QModelIndex &parent) const
587 {
588  if (row < 0
589  || column < 0 || column >= columnCount(parent)
590  || parent.column() > 0)
591  return QModelIndex();
592  if (!parent.isValid())
593  return createIndex(row, column, -1);
594 
595  QModelIndex treeIndexParent = mapToSource(parent);
596 
597  int bumpedItems = 0;
598  if (treeIndexParent == m_treeModel->index(0, 0))
599  bumpedItems = bumpedRows();
600  QModelIndex treeIndex = m_treeModel->index(row + bumpedItems, column, treeIndexParent);
601  QModelIndex historyIndex = m_treeModel->mapToSource(treeIndex);
602  int historyRow = historyIndex.row();
603  if (historyRow == -1)
604  historyRow = treeIndex.row();
605  return createIndex(row, column, historyRow);
606 }

773 QModelIndex WBHistoryFilterModel::mapFromSource(const QModelIndex &sourceIndex) const
774 {
775  load();
776  QString url = sourceIndex.data(WBHistoryModel::UrlStringRole).toString();
777  if (!m_historyHash.contains(url))
778  return QModelIndex();
779 
780  // This can be done in a binary search, but we can't use qBinary find
781  // because it can't take: qBinaryFind(m_sourceRow.end(), m_sourceRow.begin(), v);
782  // so if this is a performance bottlneck then convert to binary search, until then
783  // the cleaner/easier to read code wins the day.
784  int realRow = -1;
785  int sourceModelRow = sourceModel()->rowCount() - sourceIndex.row();
786 
787  for (int i = 0; i < m_sourceRow.count(); ++i)
788  {
789  if (m_sourceRow.at(i) == sourceModelRow)
790  {
791  realRow = i;
792  break;
793  }
794  }
795  if (realRow == -1)
796  return QModelIndex();
797 
798  return createIndex(realRow, sourceIndex.column(), sourceModel()->rowCount() - sourceIndex.row());

1234 void WBHistoryTreeModel::sourceRowsRemoved(const QModelIndex &parent, int start, int end)
1235 {
1236  Q_UNUSED(parent); // Avoid warnings when compiling release
1237  Q_ASSERT(!parent.isValid());
1238  if (m_sourceRowCache.isEmpty())
1239  return;
1240  for (int i = end; i >= start;)
1241  {
1242  QList<int>::iterator it;
1243  it = qLowerBound(m_sourceRowCache.begin(), m_sourceRowCache.end(), i);
1244  // playing it safe
1245  if (it == m_sourceRowCache.end())
1246  {
1247  m_sourceRowCache.clear();
1248  reset();
1249  return;
1250  }
1251 
1252  if (*it != i)
1253  --it;
1254  int row = qMax(0, it - m_sourceRowCache.begin());
1255  int offset = m_sourceRowCache[row];
1256  QModelIndex dateParent = index(row, 0);
1257  // If we can remove all the rows in the date do that and skip over them
1258  int rc = rowCount(dateParent);
1259  if (i - rc + 1 == offset && start <= i - rc + 1)
1260  {
1261  beginRemoveRows(QModelIndex(), row, row);
1262  m_sourceRowCache.removeAt(row);
1263  i -= rc + 1;
1264  } else {
1265  beginRemoveRows(dateParent, i - offset, i - offset);
1266  ++row;
1267  --i;
1268  }
1269  for (int j = row; j < m_sourceRowCache.count(); ++j)
1270  --m_sourceRowCache[j];
1271  endRemoveRows();
1272  }
